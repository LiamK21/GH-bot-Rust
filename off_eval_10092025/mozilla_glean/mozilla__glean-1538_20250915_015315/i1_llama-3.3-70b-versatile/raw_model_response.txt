<imports> 
use super::initialize_core_metrics;
use glean::Glean;
use glean::client_info::ClientInfoMetrics;
</imports>
<Rust>
#[test]
fn test_os_version_detection() {
  let glean = Glean::new();
  let client_info = ClientInfoMetrics::new();
  let channel = Some("test_channel".to_string());
  initialize_core_metrics(&glean, &client_info, channel);
  let os_version = glean.core_metrics.internal_metrics.os_version.get();
  assert_ne!(os_version, "unknown");
}</Rust>
