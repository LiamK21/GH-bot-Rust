#[test]
fn test_log_pings_management() {
    let mut upload_task = std::mem::MaybeUninit::uninit();

    // Test glean_get_upload_task without log_ping parameter
    let result = unsafe { glean_get_upload_task(upload_task.as_mut_ptr()) };
    assert_eq!(result, 0);

    // Test enabling log pings
    let enable_result = unsafe { glean_set_log_pings(1) };
    assert_eq!(enable_result, 0);

    // Test disabling log pings
    let disable_result = unsafe { glean_set_log_pings(0) };
    assert_eq!(disable_result, 0);
}
