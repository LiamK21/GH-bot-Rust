#[test]
fn test_glean_get_upload_task_and_log_pings() {
  let mut upload_task: *mut FfiPingUploadTask = std::ptr::null_mut();
  let _ = unsafe { glean_get_upload_task(upload_task) };
  let _ = glean_set_log_pings(1);
  let mut upload_task_with_log: *mut FfiPingUploadTask = std::ptr::null_mut();
  let _ = unsafe { glean_get_upload_task(upload_task_with_log) };
  assert_ne!(upload_task, upload_task_with_log);
}</Rust>
