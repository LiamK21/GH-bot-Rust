#[test]
fn test_glean_get_upload_task_without_log_ping() {
    let mut result = FfiPingUploadTask::default();
    let expected = FfiPingUploadTask::from(get_expected_task_without_logging());

    unsafe {
        glean_set_log_pings(0);
        glean_get_upload_task(&mut result);
    }

    assert_eq!(result, expected);
}
