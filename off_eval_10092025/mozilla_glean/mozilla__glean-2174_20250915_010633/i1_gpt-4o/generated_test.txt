#[test]
fn test_initialize_core_metrics_sets_os_version() {
    // Initialize required variables
    let glean = Glean::new();
    let client_info = ClientInfoMetrics {
        os_version: Some("10.0".to_string()),
        architecture: Some("x86_64".to_string()),
        channel: None,
        android_sdk_version: None,
    };

    // Define expected variable
    let expected_os_version = "10.0";

    // Generate actual variables
    initialize_core_metrics(&glean, &client_info);
    let actual_os_version = internal_metrics::os_version.get_value(&glean);

    // Compare expected with actual
    assert_eq!(actual_os_version, expected_os_version);
}
