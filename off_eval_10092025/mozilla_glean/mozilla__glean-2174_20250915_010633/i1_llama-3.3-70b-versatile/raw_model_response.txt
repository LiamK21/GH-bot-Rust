<imports> 
use super::initialize_core_metrics;
use glean_core::client_info::ClientInfoMetrics;
use glean_core::Glean;
</imports>
<Rust>
#[test]
fn test_initialize_core_metrics_os_version() {
  let mut glean = Glean::new();
  let client_info = ClientInfoMetrics {
    os_version: "Linux".to_string(),
    architecture: "x86_64".to_string(),
    ..Default::default()
  };
  initialize_core_metrics(&glean, &client_info);
  assert_eq!(glean.core_metrics.internal_metrics.os_version.get(), "Linux");
}</Rust>
