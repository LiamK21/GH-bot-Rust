#[test]
fn test_labeled_metric_with_cow_labels() {
    let labels = Some(vec![
        Cow::from("status_code_4xx"),
        Cow::from("status_code_5xx"),
        Cow::from("status_code_unknown"),
        Cow::from("unrecoverable"),
        Cow::from("recoverable"),
    ]);
    let meta = CommonMetricData::default();
    let labeled_metric = LabeledMetric::<()>::new(meta, labels.clone());

    assert_eq!(labeled_metric.labels, labels);
}
