<imports> 
use super::LabeledMetric;
use std::borrow::Cow;
</imports>
<Rust>
#[test]
fn test_labeled_metric_labels_memory_usage() {
  let labels = Some(vec![Cow::from("label1"), Cow::from("label2")]);
  let meta = CommonMetricData { ..Default::default() };
  let labeled_metric = LabeledMetric::new(meta, labels);
  let memory_usage_before = std::mem::size_of_val(&labeled_metric);
  let expected_memory_usage = std::mem::size_of_val(&labeled_metric.labels);
  assert_eq!(memory_usage_before, expected_memory_usage);
}</Rust>
