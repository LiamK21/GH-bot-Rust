#[test]
fn test_universal2_builds() {
  let args: Vec<String> = vec!["--config".to_string(), "path/to/config".to_string()];
  let result_before = Command::new("cargo").arg("run").args(&args).output();
  assert!(result_before.unwrap().status.success() == false);

  // Apply patch
  let result_after = Command::new("cargo").arg("run").args(&args).output();
  assert!(result_after.unwrap().status.success() == true);
}
