<imports> 
use super::Database;
use std::io;
</imports>
<Rust>
#[test]
fn test_database_delete_store_logging() {
  let database = Database::new("test");
  let mut writer = io::Cursor::new(Vec::new());
  let metrics = vec!["metric1", "metric2"];
  let result = database.delete(&mut writer, metrics);
  let error_message = "Can't delete from store: ";
  if let Err(e) = result {
    assert!(format!("{}", e).contains(error_message));
  }
}</Rust>
