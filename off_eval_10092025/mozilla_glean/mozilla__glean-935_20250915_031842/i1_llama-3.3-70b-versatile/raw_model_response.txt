<imports> 
use super::log_if_error;
use log::Level;
</imports>
<Rust>
#[test]
fn test_log_if_error_with_warning_and_error() {
  let error_code = 1;
  let error_message = "Test error message";
  let warning_level = Level::Warn;
  let error_level = Level::Error;
  let expected_log_level = error_level;
  log_if_error(ExternError { code: error_code, message: error_message });
  assert_eq!(log::max_level(), expected_log_level);
}</Rust>
