<imports>
use crate::PingRateLimit;
use crate::Configuration;
use crate::Builder;
</imports>
<Rust>
#[test]
fn test_ping_rate_limit_re_export() {
  let rate_limit = PingRateLimit {
    seconds_per_interval: 1,
    pings_per_interval: 10,
  };
  let cfg: Configuration = Builder::new()
    .rate_limit(Some(rate_limit))
    .build();
  assert!(cfg.rate_limit.is_some());
}</Rust>
