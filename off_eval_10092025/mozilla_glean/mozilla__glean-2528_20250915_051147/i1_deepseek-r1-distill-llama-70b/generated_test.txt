#[test]
fn test_ping_rate_limit_reexport() {
    let config = Configuration {
        log_level: None,
        rate_limit: Some(PingRateLimit {
            seconds_per_interval: 1,
            pings_per_interval: 10,
        }),
    };
    // If we reach here, the reexport worked
    assert!(config.rate_limit.is_some());
}
