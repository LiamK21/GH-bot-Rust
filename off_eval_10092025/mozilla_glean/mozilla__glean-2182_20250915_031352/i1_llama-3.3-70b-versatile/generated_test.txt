#[test]
fn test_uuid_encoding() {
  let glean = "glean";
  let meta = "meta";
  let value = "123e4567e89b12d3a456b321";
  let expected = Uuid::parse_str(&value).unwrap().to_hyphenated().to_string();
  let metric = UuidMetric {};
  let actual = Uuid::parse_str(&value).unwrap().to_hyphenated().to_string();
  metric.storage().record(glean, &meta, &Metric::Uuid(expected));
  assert_eq!(actual, expected);
}
