use uuid::Uuid;

<Rust>
#[test]
fn test_uuid_hyphenated_encoding() {
    let glean = Glean::new(); // Assuming a constructor for Glean
    let metric = UuidMetric::new(); // Assuming a constructor for UuidMetric
    let input_uuid = "123e4567e89b12d3a456426655440000";
    let expected_uuid = "123e4567-e89b-12d3-a456-426655440000";

    metric.record(&glean, input_uuid);

    let stored_value = glean.storage().get(&metric.meta).unwrap(); // Assuming a method to retrieve stored value
    assert_eq!(stored_value, expected_uuid);
}
</Rust>
