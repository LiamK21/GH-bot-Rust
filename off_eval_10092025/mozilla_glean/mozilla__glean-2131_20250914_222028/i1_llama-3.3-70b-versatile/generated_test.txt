#[test]
fn test_client_info_fields_populated() {
  let mut glean = Glean::new();
  let client_info = ClientInfoMetrics {
    android_sdk_version: Some("Some Android SDK version".to_string()),
    device_manufacturer: Some("Some device manufacturer".to_string()),
    device_model: Some("Some device model".to_string()),
    locale: Some("Some locale".to_string()),
  };
  initialize_core_metrics(&mut glean, &client_info);
  assert!(glean.get_string_metric("android_sdk_version").is_some());
  assert!(glean.get_string_metric("device_manufacturer").is_some());
  assert!(glean.get_string_metric("device_model").is_some());
  assert!(glean.get_string_metric("locale").is_some());
}
