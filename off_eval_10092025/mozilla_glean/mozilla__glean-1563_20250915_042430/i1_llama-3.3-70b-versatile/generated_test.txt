#[test]
fn test_invalid_tag_ignores_all_tags() {
  let valid_tags = vec!["test1".to_string(), "test2".to_string()];
  let invalid_tags = vec!["test1".to_string(), "!nv@lid-val*e".to_string()];
  let expected_valid_result = true;
  let expected_invalid_result = false;
  let actual_valid_result = validate_source_tags(&valid_tags);
  let actual_invalid_result = validate_source_tags(&invalid_tags);
  assert_eq!(actual_valid_result, expected_valid_result);
  assert_eq!(actual_invalid_result, expected_invalid_result);
}
