#[test]
fn test_glean_enable_logging_does_not_init_env_logger_by_default() {
  // Set the RUST_LOG environment variable to debug
  env::set_var("RUST_LOG", "debug");
  // Call glean_enable_logging
  glean_enable_logging();
  // env_logger should not be initialized, so logging level should not be set
  let logger = Builder::new().filter_level(LevelFilter::Off).try_init();
  assert!(logger.is_err());
}
