<imports>
use super::glean_enable_logging;
use log::LevelFilter;
use std::env;
use env_logger;
</imports>
<Rust>
#[test]
fn test_logging_initialization_on_non_android_ios() {
    // Set the RUST_LOG environment variable to a known value
    env::set_var("RUST_LOG", "debug");

    // Initialize the logger
    glean_enable_logging();

    // Check if the logger is initialized with the correct level
    let log_level = log::max_level();
    assert_eq!(log_level, LevelFilter::Debug);
}
</Rust>
