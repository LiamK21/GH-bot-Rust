#[test]
fn test_upload_task_action() {
  let glean = Glean::new();
  let upload_manager = PingUploadManager::new(&glean);
  let document_id = "test_document_id";
  let upload_result = UploadResult::done();
  let task_action = upload_manager.process_ping_upload_response(&glean, document_id, upload_result);
  assert_eq!(task_action, UploadTaskAction::End);
}
