#[test]
fn test_metric_id_in_database() {
  let db = Database {};
  let mut data: HashMap<String, Metric> = HashMap::new();
  db.iter(&Lifetime::Ping, None, |metric_id, metric| {
    data.insert(metric_id.to_string(), metric.clone());
  });
  assert!(data.iter().all(|(key, _)| key.contains("/")));
}
