#[test]
fn test_get_error_metric_for_metric() {
    let meta = CommonMetricData::new("category", "name", "ping", false, None, None, None);
    let error = ErrorType::InvalidValue;
    let expected = CounterMetric::new("category", "invalid_value", "ping");
    let actual = get_error_metric_for_metric(&meta, error);
    assert_eq!(expected, actual);
}
