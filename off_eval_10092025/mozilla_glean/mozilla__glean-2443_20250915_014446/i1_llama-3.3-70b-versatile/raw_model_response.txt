<imports> 
use super::glean_set_metrics_enabled_config;
use super::glean_test_get_experiment_data;
use serde_json::json;
</imports>
<Rust>
#[test]
fn test_set_metrics_enabled_config() {
  let metrics_enabled_config = json!({
    "category.string_metric": false,
    "category.labeled_string_metric": false,
  })
  .to_string();
  glean_set_metrics_enabled_config(metrics_enabled_config);
  let experiment_id = "test_experiment".to_string();
  let result = glean_test_get_experiment_data(experiment_id);
  assert!(result.is_none());
}</Rust>
