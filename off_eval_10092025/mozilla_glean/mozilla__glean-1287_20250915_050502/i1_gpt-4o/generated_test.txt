#[test]
fn test_get_value_defaults_to_first_send_in_ping() {
    let metric = BooleanMetric::new(
        Arc::new(glean_core::metrics::BooleanMetric::new(
            "category",
            "name",
            "test",
            vec!["ping1", "ping2"],
        )),
    );

    let expected = Some(true);
    metric.set(true);
    let actual = metric.test_get_value(None);

    assert_eq!(expected, actual);
}
