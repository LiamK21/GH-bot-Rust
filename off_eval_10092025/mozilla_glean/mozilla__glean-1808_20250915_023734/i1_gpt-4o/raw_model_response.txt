<imports>
use super::glean_enable_logging;
use log::{self, LevelFilter};
use android_logger;
</imports>
<Rust>
#[test]
fn test_logging_filter_for_glean_modules() {
    // Initialize logging
    glean_enable_logging();

    // Define expected filter levels for Glean modules
    let expected_glean_ffi_level = LevelFilter::Debug;
    let expected_glean_core_level = LevelFilter::Debug;
    let expected_glean_level = LevelFilter::Debug;

    // Generate actual filter levels
    let filter = android_logger::FilterBuilder::new()
        .filter_module("glean_ffi", LevelFilter::Debug)
        .filter_module("glean_core", LevelFilter::Debug)
        .filter_module("glean", LevelFilter::Debug)
        .build();

    // Compare expected with actual
    assert_eq!(filter.filter_module("glean_ffi"), expected_glean_ffi_level);
    assert_eq!(filter.filter_module("glean_core"), expected_glean_core_level);
    assert_eq!(filter.filter_module("glean"), expected_glean_level);
}
</Rust>
