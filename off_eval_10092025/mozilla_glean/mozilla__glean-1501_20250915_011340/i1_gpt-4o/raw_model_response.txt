use super::Timings;
use super::TimerId;

<Rust>
#[test]
fn test_multiple_timers_can_run_simultaneously() {
    let mut timings = Timings::new();
    let start_time1 = 100;
    let start_time2 = 200;

    let timer_id1 = timings.set_start(start_time1);
    let timer_id2 = timings.set_start(start_time2);

    assert_ne!(timer_id1, timer_id2, "Each start should generate a unique TimerId");
}
