#[test]
fn test_labeled_metric_size_of() {
  let label_map = Mutex::new(HashMap::new());
  let submetric = Arc::new(());
  let labels = ();
  let labeled_metric = LabeledMetric { label_map, submetric, labels };
  let ops = &mut malloc_size_of::MallocSizeOfOps::new();
  let size = labeled_metric.size_of(ops);
  assert_ne!(size, 0);
}
