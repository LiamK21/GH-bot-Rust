#[test]
fn test_glean_enable_logging_to_fd_resolves_subprocess_issue() {
  let result = Command::new("python")
    .args(&["-c", "import subprocess; subprocess.check_output(['hg', '--version'])"])
    .output();
  assert!(result.is_ok());
  glean_enable_logging_to_fd(2);
  let result = Command::new("python")
    .args(&["-c", "import subprocess; subprocess.check_output(['hg', '--version'])"])
    .output();
  assert!(result.is_ok());
}
