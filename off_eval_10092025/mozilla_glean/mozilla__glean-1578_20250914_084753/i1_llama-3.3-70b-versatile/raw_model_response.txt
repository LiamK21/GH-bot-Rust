<imports> 
use glean_core::traits::Timespan;
use std::time::Duration;
use super::TimespanMetric;
</imports>
<Rust>
#[test]
fn test_timespan_set_raw() {
  let metric = TimespanMetric::new("test_metric");
  let elapsed = Duration::from_millis(100);
  metric.set_raw(elapsed);
  let actual = metric.test_get_value(None);
  let expected = Some(100);
  assert_eq!(actual, expected);
}</Rust>
