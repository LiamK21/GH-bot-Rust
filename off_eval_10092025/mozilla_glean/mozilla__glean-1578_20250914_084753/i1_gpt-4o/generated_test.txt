#[test]
fn test_set_raw() {
    let timespan_metric = TimespanMetric(Arc::new(RwLock::new(MockTimespanMetric::new())));
    let expected_duration = Duration::new(5, 0);

    timespan_metric.set_raw(expected_duration);

    let actual_duration = launch_with_glean(|glean| {
        let inner = timespan_metric.0.read().unwrap();
        inner.test_get_value(None)
    });

    assert_eq!(actual_duration, Some(expected_duration.as_secs()));
}
