struct TestLogger {
    pub messages: std::sync::Mutex<Vec<String>>,
}

impl log::Log for TestLogger {
    fn enabled(&self, metadata: &Metadata) -> bool {
        metadata.level() <= Level::Info
    }

    fn log(&self, record: &Record) {
        if self.enabled(record.metadata()) {
            self.messages.lock().unwrap().push(format!("{}", record.args()));
        }
    }

    fn flush(&self) {}
}

#[test]
fn test_log_glean_version_on_initialization() {
    let test_logger = TestLogger {
        messages: std::sync::Mutex::new(Vec::new()),
    };
    set_logger(Box::leak(Box::new(test_logger))).unwrap();
    set_max_level(LevelFilter::Info);

    let cfg = Configuration {
        application_id: "test_app".into(),
        upload_enabled: false,
        data_path: std::path::PathBuf::new(),
        max_events: None,
    };

    let _ = Glean::new(cfg);

    let messages = test_logger.messages.lock().unwrap();
    let found = messages.iter().any(|msg| msg.contains("Creating new Glean v"));
    assert!(found, "Expected log message with Glean version not found");
}
