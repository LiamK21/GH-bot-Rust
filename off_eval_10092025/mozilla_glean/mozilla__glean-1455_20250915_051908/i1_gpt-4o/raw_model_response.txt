#[test]
fn test_activity_api() {
    let dir = tempfile::tempdir().unwrap();
    let (mut glean, _) = new_glean(Some(dir));

    glean.handle_client_active();
    assert!(glean.is_dirty_flag_set());

    glean.handle_client_inactive();
    assert!(!glean.is_dirty_flag_set());
}
