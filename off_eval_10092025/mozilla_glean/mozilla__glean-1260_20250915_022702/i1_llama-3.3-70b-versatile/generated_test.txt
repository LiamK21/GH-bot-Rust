#[test]
fn test_database_open_rkv() {
  let temp_dir = TempDir::new().unwrap();
  let path = Path::new(temp_dir.path()).join("db");
  fs::create_dir_all(&path).unwrap();
  let db = Database {
    rkv: rkv::Rkv::new(&path).unwrap(),
  };
  let result = db.open_rkv(&path);
  assert!(result.is_ok());
}
