<imports> 
use super::ObjectMetric;
use serde_json::json;
</imports>
<Rust>
#[test]
fn test_test_get_value_returns_json_object() {
  let metric = ObjectMetric::new("test_metric");
  let test_value = json!({ "key": "value" });
  metric.set(test_value.clone());
  let actual_value = metric.test_get_value(None).unwrap();
  assert_eq!(test_value, actual_value);
}
</Rust>
