#[test]
fn test_url_metric() {
  let (glean, _) = glean_core::tests::new_glean(None);
  let metric: UrlMetric = UrlMetric::new(CommonMetricData {
    name: "url".into(),
    category: "test".into(),
    send_in_pings: vec!["test1".into()],
    lifetime: Lifetime::Application,
    disabled: false,
    dynamic_label: None,
  });
  metric.set("glean://test");
  assert_eq!(metric.test_get_value("test1"), Some("glean://test".into()));
}
