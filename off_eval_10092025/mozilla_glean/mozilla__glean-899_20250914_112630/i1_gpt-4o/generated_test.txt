#[test]
fn test_pretty_body() {
    // Initialize a JSON payload and compress it
    let json_payload = r#"{"key": "value"}"#;
    let mut encoder = GzEncoder::new(Vec::new(), Compression::default());
    encoder.write_all(json_payload.as_bytes()).unwrap();
    let compressed_payload = encoder.finish().unwrap();

    // Create a PingRequest with the compressed payload
    let request = PingRequest {
        body: compressed_payload,
        document_id: String::from("test_id"),
        path: String::from("test_path"),
        headers: Default::default(),
        sent_at: None,
    };

    // Define the expected pretty-printed JSON
    let expected = r#"{
  "key": "value"
}"#;

    // Generate the actual pretty-printed JSON
    let actual = request.pretty_body().unwrap();

    // Compare expected with actual
    assert_eq!(expected, actual);
}
