<imports>
use super::glean_enable_logging;
use log::LevelFilter;
use oslog::OsLogger;
</imports>
<Rust>
#[test]
fn test_logging_initialization() {
    // Initialize the logger
    glean_enable_logging();

    // Define expected logger state
    let expected_logger = OsLogger::new("org.mozilla.glean", "glean_core").level_filter(LevelFilter::Debug);

    // Generate actual logger state
    let actual_logger = OsLogger::current();

    // Compare expected with actual
    assert_eq!(expected_logger, actual_logger);
}
</Rust>
