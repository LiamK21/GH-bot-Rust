<imports> 
use super::*;
use serde::{Deserialize, Serialize};
use serde_json;
</imports>
<Rust>
#[test]
fn test_common_metric_data_deserialize() {
  let json_data = "{\"name\":\"test_metric\"}";
  let expected = CommonMetricData {
    name: "test_metric".to_string(),
  };
  let actual: CommonMetricData = serde_json::from_str(json_data).unwrap();
  assert_eq!(expected, actual);
}</Rust>
