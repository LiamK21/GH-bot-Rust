#[test]
fn test_get_debug_view_tag_and_log_pings() {
    // Set debug view tag
    let tag = "test_tag";
    set_debug_view_tag(tag).unwrap();

    // Enable log pings
    set_log_pings(true);

    // Get values
    let actual_tag = glean_get_debug_view_tag();
    let actual_log_pings = glean_get_log_pings();
    let actual_pings = get_registered_ping_names();

    // Verify results
    assert_eq!(actual_tag, Some(tag.to_string()));
    assert_eq!(actual_log_pings, true);
    assert!(!actual_pings.is_empty());
}
