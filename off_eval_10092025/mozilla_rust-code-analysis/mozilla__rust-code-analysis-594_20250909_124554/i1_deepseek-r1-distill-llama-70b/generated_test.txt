#[test]
fn test_language_retrieval() {
    let languages = [
        LANG::Java,
        LANG::Typescript,
        LANG::Tsx,
        LANG::Javascript,
        LANG::Python,
        LANG::Rust,
        LANG::Preproc,
        LANG::Ccomment,
        LANG::Cpp,
        LANG::Mozjs,
    ];

    for lang in languages {
        let result = get_language(&lang);
        assert!(
            result.is_some(),
            "Failed to retrieve language for {:?}",
            lang
        );
    }
}
