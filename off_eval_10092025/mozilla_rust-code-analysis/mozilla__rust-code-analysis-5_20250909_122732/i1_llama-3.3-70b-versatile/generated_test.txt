#[test]
fn test_comment_removal_service() {
  let mut app = TestApp::new();
  let payload = WebCommentPayload { id: "1234".to_string(), file_name: "prova.cpp".to_string(), code: "int x = 1;".to_string() };
  let req = test::TestRequest::post()
    .uri("/comment")
    .header(header::CONTENT_TYPE, "application/json")
    .set_json(&payload);
  let res = app.service(req).await;
  assert_eq!(res.status(), StatusCode::OK);
}
