#[test]
fn test_nargs_compute_minmax() {
  let mut state = State {
    space: Space {
      metrics: Metrics {
        nargs: Stats::default(),
        ..Default::default()
      },
      ..Default::default()
    },
    ..Default::default()
  };
  state.space.metrics.nargs.fn_nargs = 5;
  state.space.metrics.nargs.closure_nargs = 3;
  compute_minmax(&mut state);
  assert_eq!(state.space.metrics.nargs.fn_nargs_min, 5);
  assert_eq!(state.space.metrics.nargs.fn_nargs_max, 5);
  assert_eq!(state.space.metrics.nargs.closure_nargs_min, 3);
  assert_eq!(state.space.metrics.nargs.closure_nargs_max, 3);
}
