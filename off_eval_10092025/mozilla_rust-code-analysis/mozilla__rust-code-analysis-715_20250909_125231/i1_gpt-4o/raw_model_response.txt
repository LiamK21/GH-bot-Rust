<imports>
use super::compute_minmax;
use crate::State;
use crate::Metrics;
use crate::NArgs;
</imports>
<Rust>
#[test]
fn test_compute_minmax_updates_nargs_minmax() {
    let mut state = State {
        space: Metrics {
            cyclomatic: Default::default(),
            nexits: Default::default(),
            cognitive: Default::default(),
            nargs: NArgs {
                fn_nargs: 3,
                closure_nargs: 2,
                fn_nargs_sum: 0,
                closure_nargs_sum: 0,
                fn_nargs_min: usize::MAX,
                closure_nargs_min: usize::MAX,
                fn_nargs_max: 0,
                closure_nargs_max: 0,
                total_functions: 1,
                total_closures: 1,
            },
        },
    };

    compute_minmax(&mut state);

    assert_eq!(state.space.metrics.nargs.fn_nargs_min, 3.0);
    assert_eq!(state.space.metrics.nargs.fn_nargs_max, 3.0);
    assert_eq!(state.space.metrics.nargs.closure_nargs_min, 2.0);
    assert_eq!(state.space.metrics.nargs.closure_nargs_max, 2.0);
}
</Rust>
