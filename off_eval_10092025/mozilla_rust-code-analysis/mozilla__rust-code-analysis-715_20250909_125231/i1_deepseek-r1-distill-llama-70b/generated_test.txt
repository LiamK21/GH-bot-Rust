#[test]
fn test_nargs_minmax() {
    let sample_code = "
def foo(a, b, c):
    pass

def bar(d, e):
    pass

def baz(f):
    pass
";
    let mut space = Space::new();
    PythonParser.parse(&mut space, sample_code).unwrap();

    let expected = Stats {
        fn_nargs_sum: 6,
        closure_nargs_sum: 0,
        fn_nargs_min: 1,
        closure_nargs_min: 0,
        fn_nargs_max: 3,
        closure_nargs_max: 0,
        total_functions: 3,
        total_closures: 0,
    };

    let actual = space.metrics.nargs.clone();

    assert_eq!(expected.fn_nargs_sum, actual.fn_nargs_sum());
    assert_eq!(expected.closure_nargs_sum, actual.closure_args_sum());
    assert_eq!(expected.fn_nargs_min, actual.fn_args_min() as usize);
    assert_eq!(expected.closure_nargs_min, actual.closure_args_min() as usize);
    assert_eq!(expected.fn_nargs_max, actual.fn_args_max() as usize);
    assert_eq!(expected.closure_nargs_max, actual.closure_args_max() as usize);
}
