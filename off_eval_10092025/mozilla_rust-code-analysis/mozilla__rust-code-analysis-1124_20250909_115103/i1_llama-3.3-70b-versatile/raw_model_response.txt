<imports> 
use super::Exit;
use super::Stats;
use super::Node;
use super::RustCode;
</imports>
<Rust>
#[test]
fn test_nexits_counts_try_expression() {
  let mut stats = Stats { exit: 0 };
  let node = Node { kind_id: Rust::TryExpression, children: vec![] };
  RustCode::compute(&node, &mut stats);
  assert_eq!(stats.exit, 1);
}</Rust>
