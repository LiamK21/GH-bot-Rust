#[test]
fn test_quic_version_for_v2_testcase() {
  let mut args = Args {
    qns_test: Some("v2".to_string()),
    quic_parameters: QuicParameters { quic_version: vec![] },
    ..Default::default()
  };
  let result = main();
  assert!(result.is_ok());
  assert_eq!(args.quic_parameters.quic_version.len(), 0);
}
