<imports>
use super::LossRecoverySpace;
use std::time::{Duration, Instant};
</imports>
<Rust>
#[test]
fn test_detect_lost_packets_no_overflow() {
    let now = Instant::now();
    let loss_delay = Duration::from_secs(1);
    let cleanup_delay = Duration::from_secs(10);
    let mut recovery_space = LossRecoverySpace::new();

    // Simulate sending a packet
    let packet_time_sent = now - Duration::from_secs(2);
    recovery_space.sent_packets.insert(1, Packet { time_sent: packet_time_sent });

    // Simulate acknowledging a packet
    let largest_acked = Some(2);

    // Call the function under test
    recovery_space.detect_lost_packets(now, loss_delay, cleanup_delay, largest_acked);

    // Check that the packet is considered lost
    assert!(recovery_space.lost_packets.contains(&1));
}
</Rust>
