use super::Time;
use std::thread;
use std::time::Duration;

<Rust>
#[test]
fn test_update_multi() {
  let thr = std::thread::spawn(move || {
    let mut hrt = Time::get(Duration::from_millis(4));
    hrt.update(Duration::from_millis(1));
    hrt.update(Duration::from_millis(1));
  });
  thr.join().unwrap();
}</Rust>
