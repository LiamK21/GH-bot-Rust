<imports>
use super::TxBuffer;
use super::RangeTracker;
use super::RangeState;
</imports>
<Rust>
#[test]
fn test_next_bytes_cpu_usage() {
    let mut tx_buffer = TxBuffer {
        ranges: RangeTracker {
            used: [(0, (10, RangeState::default()))].iter().cloned().collect(),
            first_unmarked: None,
        },
        retired: 0,
        buffer: vec![0; 10],
    };

    let expected = Some((0, &[0; 10][..]));
    let actual = tx_buffer.next_bytes();
    assert_eq!(expected, actual);
}
</Rust>
