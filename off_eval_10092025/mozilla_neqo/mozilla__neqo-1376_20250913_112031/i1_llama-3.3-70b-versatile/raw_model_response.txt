#[test]
fn test_removed_lints() {
  let mut conn = Connection {};
  let sendorder = Some(SendOrder {});
  let fairness = true;
  set_sendorder(&mut conn, sendorder);
  set_fairness(&mut conn, fairness);
  assert!(std::panic::catch_unwind(|| {
    set_sendorder(&mut conn, sendorder);
    set_fairness(&mut conn, fairness);
  }).is_ok());
}</Rust>
