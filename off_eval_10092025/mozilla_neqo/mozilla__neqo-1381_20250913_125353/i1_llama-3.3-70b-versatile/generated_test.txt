#[test]
fn test_suppress_unavailable_error() {
  let mut handler = Handler { streams: std::collections::HashMap::new(), url_queue: VecDeque::new() };
  let result = handler.try_create_stream("test_url");
  let expected_output = "";
  let actual_output = "";
  assert_eq!(expected_output, actual_output);
  // simulate the error
  let result = handler.try_create_stream("test_url");
  let expected_output = "";
  let actual_output = "";
  assert_eq!(expected_output, actual_output);
}
