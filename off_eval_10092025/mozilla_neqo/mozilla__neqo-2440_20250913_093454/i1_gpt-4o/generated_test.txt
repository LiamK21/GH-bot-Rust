#[test]
fn test_iptosecn_from_u8() {
    let values = [
        (0b00, IpTosEcn::NotEct),
        (0b01, IpTosEcn::Ect1),
        (0b10, IpTosEcn::Ect0),
        (0b11, IpTosEcn::Ce),
    ];

    for (input, expected) in values.iter() {
        let actual: IpTosEcn = (*input).into();
        assert_eq!(*expected, actual);
    }
}
