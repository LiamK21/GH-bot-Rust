#[test]
fn test_iptos_ecn_from_repr() {
    use IpTosEcn::*;
    // Test all possible ECN values
    let test_cases = [
        (0b00, NotEct),
        (0b01, Ect1),
        (0b10, Ect0),
        (0b11, Ce),
    ];
    for (input, expected) in &test_cases {
        let actual = IpTosEcn::from_repr(*input).expect("all ECN values are covered");
        assert_eq!(actual, *expected);
    }
}
