#[test]
fn test_server_udp_receive_path() {
  let mut server = HttpServer {
    // initialize server
  };
  let dgram = Some(Datagram::new(&[1, 2, 3], "127.0.0.1:8080".parse().unwrap()));
  let now = Instant::now;
  let output = server.process(dgram, now());
  assert!(output != Output::None);
