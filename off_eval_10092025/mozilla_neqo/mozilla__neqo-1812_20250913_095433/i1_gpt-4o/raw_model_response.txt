use super::ClassicCongestionControl;
use super::CongestionControl;

<Rust>
#[test]
fn test_rebinding_does_not_panic() {
    let mut cc = ClassicCongestionControl::new();
    let pkt_size = 1000;
    cc.bytes_in_flight = pkt_size;
    
    // Simulate a rebinding event
    cc.bytes_in_flight = 0;
    
    // Simulate receiving an ACK for a packet sent before rebinding
    cc.on_packets_acked(pkt_size);
    
    // Check that bytes_in_flight does not go negative
    assert_eq!(cc.bytes_in_flight, 0);
}
</Rust>
