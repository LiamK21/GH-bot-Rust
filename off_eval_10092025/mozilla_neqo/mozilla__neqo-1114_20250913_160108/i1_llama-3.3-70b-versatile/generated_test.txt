#[test]
fn test_connection_parameters_initialized() {
  let mut cid_manager = ConnectionIdManager::new(&mut vec![], vec![1, 2, 3]);
  let conn_params = ConnectionParameters::default();
  let role = Role::Server;
  let result = conn_params.create_transport_parameter(role, &mut cid_manager);
  assert!(result.is_ok());
  let tps = result.unwrap();
  assert_eq!(tps.local.get_integer(tparams::INITIAL_MAX_STREAM_DATA_BIDI_REMOTE).unwrap(), 8192);
  assert_eq!(tps.local.get_integer(tparams::IDLE_TIMEOUT).unwrap(), 30000);
  assert_eq!(tps.local.get_integer(tparams::ACTIVE_CONNECTION_ID_LIMIT).unwrap(), 0x100);
}
