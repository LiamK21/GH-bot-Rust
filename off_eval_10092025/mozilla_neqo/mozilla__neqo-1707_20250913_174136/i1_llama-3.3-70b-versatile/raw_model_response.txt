<imports> 
use super::Socket;
use tokio::net::UdpSocket;
use std::net::SocketAddr;
use std::io;
</imports>
<Rust>
#[test]
fn test_recv_mut_socket() {
  let socket = UdpSocket::bind("127.0.0.1:0").expect("bind failed");
  let mut socket = Socket::new(socket).expect("socket creation failed");
  let local_address: SocketAddr = "127.0.0.1:0".parse().unwrap();
  let result = socket.recv(&local_address);
  assert!(result.is_ok());
}</Rust>
