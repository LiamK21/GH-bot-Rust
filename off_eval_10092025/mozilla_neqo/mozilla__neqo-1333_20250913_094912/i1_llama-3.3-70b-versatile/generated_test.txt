#[test]
fn test_do_log_with_four_arguments() {
  let lvl = Level::Info;
  let target = "target";
  let arg = "argument";
  let result_before_patch = std::panic::catch_unwind(|| do_log!(lvl, "{}", arg));
  let result_after_patch = std::panic::catch_unwind(|| do_log!(lvl, "{}", arg));
  assert!(result_before_patch.is_err());
  assert!(result_after_patch.is_ok());
}
