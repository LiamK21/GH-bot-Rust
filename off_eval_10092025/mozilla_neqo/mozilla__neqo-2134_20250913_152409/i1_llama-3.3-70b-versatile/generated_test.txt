#[test]
fn test_resend_0rtt_after_useless_crypto_frame() {
  let mut connection = Connection::new();
  let now = Instant::now();
  connection.crypto.resend_unacked(PacketNumberSpace::Initial);
  let old_len = connection.stats.borrow().saved_datagrams;
  connection.client_0rtt_rejected(now);
  let new_len = connection.stats.borrow().saved_datagrams;
  assert!(new_len > old_len);
}
