#[test]
fn test_make_resumption_token_no_path() {
    let mut connection = Connection {
        role: Role::Client,
        crypto: Crypto::new(),
        paths: Default::default(),
        loss_recovery: LossRecovery::new(),
        events: ConnectionEvents::default(),
        streams: Streams::default(),
        cid_manager: CidManager::new(),
        stats: StatsCell::default(),
        quic_datagrams: QuicDatagrams::default(),
    };

    let expected_rtt = RttEstimate::default().estimate();
    let resumption_token = connection.make_resumption_token();
    let actual_rtt = resumption_token.rtt();

    assert_eq!(expected_rtt, actual_rtt);
}
