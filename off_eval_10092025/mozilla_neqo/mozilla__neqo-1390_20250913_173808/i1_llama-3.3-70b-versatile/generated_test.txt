#[test]
fn test_session_id_validation() {
  let mut conn = Http3Connection::new();
  let session_id: u32 = 5; // example session id
  let stream_id = StreamId::from(session_id);
  let result_before_patch = conn.stream_stop_sending(stream_id, Error::HttpStreamCreation.code());
  assert!(result_before_patch.is_ok());
  // Apply patch
  // Check the validation of the session id
  let result_after_patch = conn.stream_stop_sending(stream_id, Error::HttpId.code());
  assert!(result_after_patch.is_ok());
}
