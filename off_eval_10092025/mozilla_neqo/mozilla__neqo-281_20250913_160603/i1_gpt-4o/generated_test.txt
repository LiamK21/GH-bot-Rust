#[test]
fn test_handle_connection_closing_state_transition() {
    let mut conn = Http3Connection::new();
    let error_code = CloseError::Application(0);

    // Simulate the connection being in an active state
    conn.state = Http3State::Connected;

    // Call the function under test
    conn.handle_connection_closing(error_code).unwrap();

    // Check that the state has transitioned to Closing
    assert!(matches!(conn.state, Http3State::Closing(_)));
}
