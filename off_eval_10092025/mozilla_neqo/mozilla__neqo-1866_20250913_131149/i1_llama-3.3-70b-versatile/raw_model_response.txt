<imports> 
use super::Client;
use neqo_transport::Connection;
use neqo_transport::ConnectionError;
use std::time::Instant;
</imports>
<Rust>
#[test]
fn test_client_is_closed() {
  let mut connection = Connection::new().unwrap();
  connection.close(Instant::now(), 0, "");
  let result = connection.is_closed();
  match result {
    Ok(CloseState::Closed) => assert!(true),
    _ => assert!(false),
  }
}</Rust>
