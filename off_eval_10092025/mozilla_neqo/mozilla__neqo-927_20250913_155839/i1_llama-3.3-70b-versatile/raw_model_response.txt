<imports> 
use super::CongestionControl;
use crate::packet::PacketType;
use crate::packet::PacketNumber;
use crate::tracking::SentPacket;
use std::time::Duration;
use std::time::Instant;
</imports>
<Rust>
#[test]
fn test_persistent_congestion() {
  let mut cc = CongestionControl::default();
  let pto = Duration::from_millis(100);
  let rtt = Duration::from_millis(98);
  let lost_packets = vec![
    SentPacket::new(PacketType::Short, 1, Instant::now(), true, std::rc::Rc::default(), 100),
    SentPacket::new(PacketType::Short, 2, Instant::now() + Duration::from_millis(100 * 5), true, std::rc::Rc::default(), 100),
  ];
  cc.on_packets_lost(Instant::now(), None, pto, &lost_packets);
  assert_eq!(cc.cwnd(), 2);
}</Rust>
