<Rust>
#[test]
fn test_send_arbitrary_amounts_of_data() {
  let mut server = SimpleServer::new(&Args::default(), AntiReplay::default(), Rc::new(RefCell::new(RandomConnectionIdGenerator::default())));
  let mut response = ResponseData::from(vec![1; 10000]);
  let mut stream = Http3OrWebTransportStream::new(1, StreamType::Bidi);
  response.send(&mut stream);
  assert_eq!(response.remaining, 0);
}</Rust>
