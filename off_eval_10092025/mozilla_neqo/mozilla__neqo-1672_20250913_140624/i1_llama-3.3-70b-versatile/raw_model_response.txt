<imports>
use neqo_common::event::Provider;
use neqo_crypto::init;
use neqo_http3::Http3Client;
use neqo_transport::Connection;
use std::thread;
use std::time::Duration;
</imports>
<Rust>
#[test]
fn test_neqo_interop_removal() {
  init();
  let mut client = Http3Client::new("example.com", &[], "127.0.0.1:0".parse().unwrap(), "127.0.0.1:443".parse().unwrap(), Duration::from_secs(1)).unwrap();
  assert!(std::panic::catch_unwind(|| client.conn()).is_err());
}</Rust>
