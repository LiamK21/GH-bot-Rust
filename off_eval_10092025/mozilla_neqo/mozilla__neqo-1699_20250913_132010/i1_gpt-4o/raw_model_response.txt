<imports>
use super::init;
use log::LevelFilter;
use std::sync::Once;
</imports>
<Rust>
#[test]
fn test_init_no_op_in_release() {
    static INIT_ONCE: Once = Once::new();
    INIT_ONCE.call_once(|| {
        // Set the STATIC_MAX_LEVEL to Off to simulate release build behavior
        let original_max_level = log::STATIC_MAX_LEVEL;
        log::STATIC_MAX_LEVEL = LevelFilter::Off;

        // Call init and check if it returns immediately
        init();

        // Restore the original STATIC_MAX_LEVEL
        log::STATIC_MAX_LEVEL = original_max_level;
    });
}
</Rust>
