<imports> 
use std::sync::Once;
use std::time::Duration;
use log::LevelFilter;
use log::Level;
use log::logger;
use log::Record;
use log::max_level;
use log::STATIC_MAX_LEVEL;
use env_logger::Builder;
use super::init;

<Rust>
#[test]
fn test_init_performance() {
  let original_max_level = max_level();
  let _ = env_logger::try_init();
  max_level().set(LevelFilter::Off);
  let start = std::time::Instant::now();
  for _ in 0..10000 {
    init();
  }
  let duration = start.elapsed();
  assert!(duration.as_millis() < 10);
  max_level().set(original_max_level);
}</Rust>
