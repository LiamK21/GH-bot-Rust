use super::SendStream;
use super::Frame;

<Rust>
#[test]
fn test_send_data_blocked() {
  let mut s = SendStream::new(1, 100);
  let buf = [1, 2, 3, 4, 5];
  s.send(&buf).unwrap();
  let expected = Frame::DataBlocked { data_limit: 100 };
  let actual = s.flow_mgr.borrow_mut().next().unwrap();
  assert_eq!(actual, expected);
}</Rust>
