<imports> 
use super::send_inner;
use std::io;
use std::net::UdpSocket;
</imports>
<Rust>
#[test]
fn test_send_inner_propagates_io_error() {
  let socket = UdpSocket::bind("127.0.0.1:0").unwrap();
  let state = quinn_udp::UdpSocketState::new();
  let d = quinn_udp::Datagram::new(&[1, 2, 3], "127.0.0.1:8080");
  let result = send_inner(&state, &socket, &d);
  assert!(result.is_err());
}</Rust>
