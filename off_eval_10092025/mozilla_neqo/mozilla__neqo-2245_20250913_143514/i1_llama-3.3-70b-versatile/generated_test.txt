#[test]
fn test_take_ranges() {
  let mut packets = SentPackets::new();
  let packet_number = 1..101;
  for i in packet_number {
    packets.packets.insert(i, ());
  }
  let acked_ranges = vec![(1, 2)];
  let mut result = Vec::new();
  let original_packets_len = packets.packets.len();
  result = packets.take_ranges(acked_ranges);
  assert_eq!(original_packets_len - acked_ranges.len() * (acked_ranges[0].1 - acked_ranges[0].0 + 1), packets.packets.len());
}
