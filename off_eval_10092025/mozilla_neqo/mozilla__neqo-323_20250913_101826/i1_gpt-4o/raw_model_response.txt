<imports>
use super::Connection;
use std::time::Instant;
use crate::Datagram;
</imports>
<Rust>
#[test]
fn test_initial_packet_padding() {
    let mut conn = Connection::new();
    let now = Instant::now();
    let out = conn.output_pkt_for_path(now).unwrap();
    assert!(out.is_some());
    let datagram = out.unwrap();
    assert_eq!(datagram.len(), conn.path.as_ref().unwrap().mtu());
}
</Rust>
