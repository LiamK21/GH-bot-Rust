<imports> 
use super::Connection;
use neqo_common::{Instant, Role};
use neqo_crypto::Epoch;
use neqo_transport::Error;
</imports>
<Rust>
#[test]
fn test_initial_packet_padded_to_path_mtu() {
    let mut connection = Connection::new(Role::Client, Instant::now());
    let path = connection.path.unwrap();
    let out = connection.output_pkt_for_path(Instant::now()).unwrap().unwrap();
    assert_eq!(out.len(), path.mtu());
}
