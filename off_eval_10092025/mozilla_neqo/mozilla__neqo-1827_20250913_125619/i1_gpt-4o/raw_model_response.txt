#[test]
fn test_padding_frame_logging() {
    let mut qlog = NeqoQlog::default();
    let frame = Frame::Padding(10);
    let expected = QuicFrame::Padding {
        length: None,
        payload_length: 10,
    };

    let actual: QuicFrame = (&frame).into();
    assert_eq!(expected, actual);
}
