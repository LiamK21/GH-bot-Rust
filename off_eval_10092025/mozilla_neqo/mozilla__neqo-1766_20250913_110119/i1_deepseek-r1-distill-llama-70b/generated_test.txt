#[test]
fn test_stats_output() {
    let mut args = Args::default();
    args.stats = true;
    let mut runner = Runner::new(&args, "http".to_string());
    runner.client = Box::new(Http3Client::new(&mut runner.handler));
    runner.process(None).await?;

    let stats = runner.client.stats();
    assert!(stats.packets_tx > 0);
    assert!(stats.packets_rx > 0);
}
