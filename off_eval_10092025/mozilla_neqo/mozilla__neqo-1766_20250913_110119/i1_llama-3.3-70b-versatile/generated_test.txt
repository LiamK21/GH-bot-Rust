#[test]
fn test_client_stats() {
  let mut args = Args { upload_size: 100, stats: true };
  let mut client = Connection { state: State::Open };
  let stats = client.stats();
  let expected_stats = Stats { packets_tx: 0, lost: 0, late_ack: 0, pto_ack: 0, resumed: 0 };
  assert_eq!(stats, expected_stats);
}
