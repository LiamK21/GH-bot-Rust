<imports>
use super::CongestionControl;
use std::time::{Duration, Instant};
use crate::tracking::SentPacket;
use crate::packet::PacketType;
use std::rc::Rc;
</imports>
<Rust>
#[test]
fn test_persistent_congestion_defer_start() {
    let mut cc = CongestionControl::default();
    let now = Instant::now();
    let pto = Duration::from_millis(100);
    let first_rtt_sample_time = Some(now + Duration::from_millis(50));
    let prev_largest_acked_sent = Some(now);
    
    let lost_packets = vec![
        SentPacket::new(PacketType::Short, 1, now + Duration::from_millis(10), true, Rc::default(), 1000),
        SentPacket::new(PacketType::Short, 2, now + Duration::from_millis(110), true, Rc::default(), 1000),
        SentPacket::new(PacketType::Short, 3, now + Duration::from_millis(220), true, Rc::default(), 1000),
    ];
    
    cc.on_packets_lost(now + Duration::from_millis(300), first_rtt_sample_time, prev_largest_acked_sent, pto, &lost_packets);
    
    assert_eq!(cc.cwnd(), super::CWND_INITIAL);
}
</Rust>
