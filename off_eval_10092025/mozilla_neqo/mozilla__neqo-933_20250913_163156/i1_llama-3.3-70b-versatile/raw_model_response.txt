<imports> 
use super::CongestionControl;
use std::time::{Duration, Instant};
use crate::tracking::SentPacket;
</imports>
<Rust>
#[test]
fn test_persistent_congestion_with_rtt_sample() {
  let mut cc = CongestionControl::default();
  let rtt_sample_time = Some(Instant::now());
  let prev_largest_acked = None;
  let pto = Duration::from_millis(100);
  let lost_packets = vec![SentPacket::new(0, 1, Instant::now(), true, std::rc::Rc::default(), 1000)];
  cc.detect_persistent_congestion(None, prev_largest_acked, pto, &lost_packets);
  assert_eq!(cc.cwnd(), CongestionControl::default().cwnd());
  cc.detect_persistent_congestion(Some(Instant::now()), prev_largest_acked, pto, &lost_packets);
  assert_eq!(cc.cwnd(), CongestionControl::default().cwnd());
  cc.detect_persistent_congestion(rtt_sample_time, prev_largest_acked, pto, &lost_packets);
  assert_eq!(cc.cwnd(), CongestionControl::default().cwnd());
}</Rust>
