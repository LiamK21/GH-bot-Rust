<imports> 
use super::convert_to_f64;
use std::time::Duration;
use std::time::Instant;
</imports>
<Rust>
#[test]
fn test_cubic_fast_convergence() {
  let mut cubic = Cubic {
    last_max_cwnd: 1000.0,
    estimated_tcp_cwnd: 500.0,
    k: 0.0,
    w_max: 1000.0,
    ca_epoch_start: None,
    tcp_acked_bytes: 0.0,
  };
  let max_datagram_size = 100;
  let curr_cwnd = 900;
  let expected_last_max_cwnd = 765.0;
  cubic.last_max_cwnd = if convert_to_f64(curr_cwnd) + convert_to_f64(max_datagram_size) < cubic.last_max_cwnd {
    convert_to_f64(curr_cwnd) * 0.85
  } else {
    cubic.last_max_cwnd
  };
  assert_eq!(cubic.last_max_cwnd, expected_last_max_cwnd);
}</Rust>
