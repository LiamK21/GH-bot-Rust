#[test]
fn test_acknowledged_does_not_log() {
  let mut range = PacketRange { smallest: 1, largest: 10, ack_needed: true };
  let other = PacketRange { smallest: 1, largest: 10, ack_needed: false };
  let start = Instant::now();
  range.acknowledged(&other);
  let duration = start.elapsed();
  assert!(duration < Duration::from_millis(10));
}
