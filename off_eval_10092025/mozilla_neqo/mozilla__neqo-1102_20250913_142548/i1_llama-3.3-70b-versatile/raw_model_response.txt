<imports> 
use super::PacketBuilder;
use super::Encoder;
use super::PacketType;
use super::QuicVersion;
use super::ConnectionId;

<Rust>
#[test]
fn test_packet_builder_initial_token() {
  let mut builder = PacketBuilder::long(
    Encoder::new(),
    PacketType::Initial,
    QuicVersion::default(),
    &ConnectionId::from(&[0u8; 8]),
    &ConnectionId::from(&[0u8; 8]),
  );
  let token = [0u8; 1000];
  builder.initial_token(&token);
  assert_eq!(builder.remaining(), 0);
}
</Rust>
