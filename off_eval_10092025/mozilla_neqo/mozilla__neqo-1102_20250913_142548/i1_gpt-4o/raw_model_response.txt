<Rust>
#[test]
fn test_packet_builder_no_internal_error_19() {
    let encoder = Encoder::with_capacity(10);
    let mut builder = PacketBuilder::short(encoder, true, &ConnectionId::from(&[0; 8]));
    builder.pn(1, 4);
    assert_eq!(builder.remaining(), 0);
    let encoder = builder.abort();
    assert!(encoder.is_empty());
}
