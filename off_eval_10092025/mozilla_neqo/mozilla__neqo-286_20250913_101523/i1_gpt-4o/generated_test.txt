#[test]
fn test_decode_stream_frame_fill() {
    let enc = Encoder::from_hex("0905010203");
    let mut dec = enc.as_decoder();
    let frame = decode_frame(&mut dec).unwrap();
    if let Frame::Stream { fill, .. } = frame {
        assert!(fill, "Expected fill to be true for frame without length field");
    } else {
        panic!("Expected a Stream frame");
    }
}
