<imports> 
use super::Connection;
use std::time::Duration;
use std::time::Instant;
</imports>
<Rust>
#[test]
fn test_callback_time_after_pto() {
  let mut connection = Connection::default();
  let now = Instant::now();
  let _ = connection.process(None, now);
  let _ = connection.process(None, now + Duration::from_millis(144));
  let callback_time_before_pto = connection.get_callback_time();
  assert_ne!(callback_time_before_pto, Duration::from_millis(0));
  let _ = connection.process(None, now + Duration::from_millis(170));
  let callback_time_after_pto = connection.get_callback_time();
  assert_ne!(callback_time_after_pto, Duration::from_millis(0));
}</Rust>
