#[test]
fn test_vn_grease_avoids_client_version() {
    fixture_init();
    let client_version = 0x0a0a0a0a;
    let vn = PacketBuilder::version_negotiation(
        SERVER_CID,
        CLIENT_CID,
        client_version,
        &Version::all(),
    );
    let last_byte = vn[vn.len() - 1];
    assert_ne!(last_byte, (client_version & 0xff) as u8);
}
