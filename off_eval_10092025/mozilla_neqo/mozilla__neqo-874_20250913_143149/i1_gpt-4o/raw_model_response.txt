<imports>
use super::SendStreams;
use crate::stream_id::StreamId;
use crate::send_stream::SendStream;
</imports>
<Rust>
#[test]
fn test_send_streams_iteration_order() {
    let mut send_streams = SendStreams::default();
    let stream_id1 = StreamId::from(1);
    let stream_id2 = StreamId::from(2);
    let stream_id3 = StreamId::from(3);

    send_streams.0.insert(stream_id1, SendStream::default());
    send_streams.0.insert(stream_id2, SendStream::default());
    send_streams.0.insert(stream_id3, SendStream::default());

    let mut iter = send_streams.into_iter();
    let first = iter.next().unwrap().0;
    let second = iter.next().unwrap().0;
    let third = iter.next().unwrap().0;

    assert_eq!(*first, stream_id1);
    assert_eq!(*second, stream_id2);
    assert_eq!(*third, stream_id3);
}
</Rust>
