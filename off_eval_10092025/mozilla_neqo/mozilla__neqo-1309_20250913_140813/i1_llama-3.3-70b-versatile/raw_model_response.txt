<imports>
use super::is_interim;
use super::Error;
use neqo_http3::headers::Header;
</imports>
<Rust>
#[test]
fn test_is_interim_returns_error_for_101() {
  let headers = &[Header::new(":status", "101")];
  let result = is_interim(headers);
  assert!(result.is_err());
  assert_eq!(result.unwrap_err(), Error::InvalidHeader);
}
</Rust>
