#[test]
fn test_ecn_count_is_empty() {
  let not_ect = 0;
  let ect1 = 0;
  let ect0 = 0;
  let ce = 0;
  let count = Count::from_array([not_ect, ect1, ect0, ce]);
  assert!(count.is_empty());

  let not_ect = 1;
  let ect1 = 0;
  let ect0 = 0;
  let ce = 0;
  let count = Count::from_array([not_ect, ect1, ect0, ce]);
  assert!(!count.is_empty());

  let not_ect = 0;
  let ect1 = 1;
  let ect0 = 0;
  let ce = 0;
  let count = Count::from_array([not_ect, ect1, ect0, ce]);
  assert!(!count.is_empty());

  let not_ect = 0;
  let ect1 = 0;
  let ect0 = 1;
  let ce = 0;
  let count = Count::from_array([not_ect, ect1, ect0, ce]);
  assert!(!count.is_empty());

  let not_ect = 0;
  let ect1 = 0;
  let ect0 = 0;
  let ce = 1;
  let count = Count::from_array([not_ect, ect1, ect0, ce]);
  assert!(!count.is_empty());
}
