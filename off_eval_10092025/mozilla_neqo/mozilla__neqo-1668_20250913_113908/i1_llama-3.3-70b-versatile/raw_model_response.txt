<imports> 
use neqo_common::Res;
use neqo_transport::{Connection, SendOrder, SendStream};
use super::SendStream;

<Rust>
#[test]
fn test_removed_sendorder_and_fairness_methods() {
  let mut conn = Connection::new();
  let mut stream = WebTransportSession::new();
  let result1 = stream.set_sendorder(&mut conn, Some(SendOrder::Default));
  let result2 = stream.set_fairness(&mut conn, true);
  assert!(result1.is_ok());
  assert!(result2.is_ok());
  assert_eq!(result1.unwrap(), ());
  assert_eq!(result2.unwrap(), ());
}</Rust>
