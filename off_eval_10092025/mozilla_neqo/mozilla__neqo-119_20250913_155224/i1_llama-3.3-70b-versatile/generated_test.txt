#[test]
fn test_borrow_mut_removal() {
    let mut conn = Http3Connection::new(vec![], vec![], None, None, None, None, None, None);
    let state = Http3State::Initializing;
    let expected = state;
    conn.state = state;
    if conn.state == Http3State::Initializing {
        conn.events.connected();
    }
    let actual = conn.state;
    assert_eq!(expected, actual);
}
