#[test]
fn test_qns_read_response() {
    let test_filename = "test_file.txt";
    let test_content = b"Test content";

    // Create a temporary file with known content
    let mut file = fs::File::create(test_filename).expect("Unable to create test file");
    file.write_all(test_content).expect("Unable to write to test file");

    // Call the function under test
    let result = qns_read_response(test_filename);

    // Clean up the temporary file
    fs::remove_file(test_filename).expect("Unable to delete test file");

    // Assert that the function returns the expected content
    assert_eq!(result.expect("Failed to read response"), test_content);
}
