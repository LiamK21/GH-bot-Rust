<imports>
use super::Encoder;
use std::fmt::Write;
</imports>
<Rust>
#[test]
fn test_encoder_write_str() {
    let mut encoder = Encoder::new();
    let input_str = "test";
    let expected = input_str.as_bytes().to_vec();

    write!(encoder, "{}", input_str).expect("write should succeed");

    assert_eq!(encoder.as_ref(), &expected);
}
</Rust>
