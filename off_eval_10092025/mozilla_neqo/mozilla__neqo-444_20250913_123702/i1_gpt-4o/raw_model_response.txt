<imports>
use super::ResponseStream;
use crate::client_events::Http3ClientEvents;
use crate::Header;
use neqo_transport::Connection;
</imports>
<Rust>
#[test]
fn test_response_stream_read_response_headers() {
    let stream_id = 0;
    let conn_events = Http3ClientEvents::default();
    let mut response_stream = ResponseStream::new(stream_id, conn_events);

    let headers = vec![
        (String::from(":status"), String::from("200")),
        (String::from("content-length"), String::from("0")),
    ];
    let header_block = vec![0x01, 0x06, 0x00, 0x00, 0xd9, 0x54, 0x01, 0x30];
    response_stream.handle_headers_frame(header_block, false).unwrap();

    let (actual_headers, fin) = response_stream.read_response_headers().unwrap();
    assert_eq!(actual_headers, headers);
    assert_eq!(fin, false);
}
</Rust>
