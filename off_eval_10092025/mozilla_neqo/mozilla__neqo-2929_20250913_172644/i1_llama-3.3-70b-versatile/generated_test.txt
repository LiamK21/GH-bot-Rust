#[test]
fn test_auto_tuning_approximates_bandwidth_delay_product_runs_faster() {
  let start = Instant::now();
  test_fixture::fixture_init();
  for _ in 0..100 {
    let _bandwidth = u64::from(u16::from_be_bytes([0; 2])) % 1_000 + 1;
  }
  let duration = start.elapsed().as_secs_f64();
  assert!(duration < 3.0);
}
