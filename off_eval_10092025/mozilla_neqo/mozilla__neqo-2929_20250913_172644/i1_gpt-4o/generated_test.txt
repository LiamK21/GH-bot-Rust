#[test]
fn test_auto_tuning_approximates_bandwidth_delay_product_speed() {
    test_fixture::fixture_init();

    let start = Instant::now();
    for _ in 0..100 {
        let bandwidth = u64::from(u16::from_be_bytes(random::<2>()) % 1_000 + 1) * 1_000 * 1_000;
        // Simulate some operation with bandwidth
    }
    let duration = start.elapsed();
    assert!(duration.as_secs_f64() < 0.5, "Test took too long: {:?}", duration);
}
