#[test]
fn test_default_reordering_threshold() {
    let mut rp = RecvdPackets::new(PacketNumberSpace::ApplicationData);
    let now = Instant::now();

    // Set received packets in reverse order
    rp.set_received(now, 2, true);
    rp.set_received(now, 1, true);

    // Without any ACK_FREQUENCY frames, the default reordering_threshold is 1
    // So receiving packet 1 after 2 should trigger an immediate ACK
    assert!(rp.ack_now(now, Duration::from_millis(10)));
}
