#[test]
fn test_main_resume() {
  let args = Args { resume: true, download_in_series: false, use_old_http: false };
  let urls = vec!["url1".to_string(), "url2".to_string()];
  let mut first = true;
  let to_request_first = if args.resume && first { vec!["url1".to_string()] } else { vec![] };
  first = false;
  let to_request_second = if args.resume && first || args.download_in_series { vec!["url2".to_string()] } else { vec![] };
  assert_eq!(to_request_first, vec!["url1".to_string()]);
  assert_eq!(to_request_second, vec!["url2".to_string()]);
}</Rust>
