#[test]
fn test_header_from_str_error_messages() {
    let missing_colon = "namevalue";
    let missing_name = ":value";

    let result_missing_colon = Header::from_str(missing_colon);
    let result_missing_name = Header::from_str(missing_name);

    assert_eq!(result_missing_colon, Err(FromStrError::MissingColon));
    assert_eq!(result_missing_name, Err(FromStrError::MissingName));
}
