#[test]
fn test_pto_after_confirmed() {
  let mut lr = LossRecovery::new();
  let now = Instant::now();
  let sent_pkt = super::SentPacket::new(
      super::PacketType::Handshake,
      0,
      now,
      true,
      std::rc::Rc::default(),
      100,
  );
  lr.on_packet_sent(sent_pkt);
  lr.discard(super::PNSpace::Initial, now);
  lr.discard(super::PNSpace::Handshake, now);
  assert!(lr.pto_time(super::PNSpace::ApplicationData).is_some());
}
