#[test]
fn test_datagram_into_vec() {
    let local_addr = "127.0.0.1:1234".parse().unwrap();
    let remote_addr = "127.0.0.1:5678".parse().unwrap();
    let data = b"Hello, world!".to_vec();
    let datagram = Datagram::new(
        local_addr,
        remote_addr,
        IpTos::default(),
        None,
        data.clone(),
    );

    let vec: Vec<u8> = datagram.into();
    assert_eq!(data, vec);
}
