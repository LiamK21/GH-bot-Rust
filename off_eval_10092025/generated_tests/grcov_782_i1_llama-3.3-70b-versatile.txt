#[test]
fn test_get_percentage_of_covered_lines_fixes_coverage_json_issue() {
  let covered_lines: usize = 5;
  let total_lines: usize = 10;
  let expected_coverage: f64 = 50.0;
  let actual_coverage = get_percentage_of_covered_lines(covered_lines, total_lines);
  assert_eq!(actual_coverage, expected_coverage);
}
