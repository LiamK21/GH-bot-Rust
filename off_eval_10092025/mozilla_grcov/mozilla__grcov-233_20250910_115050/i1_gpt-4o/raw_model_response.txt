<imports>
use super::run_gcov;
use std::path::PathBuf;
use std::env;
</imports>
<Rust>
#[test]
fn test_gcov_failure_output() {
    // Set up a non-existent gcno path to force gcov to fail
    let gcno_path = PathBuf::from("non_existent_file.gcno");
    let working_dir = PathBuf::from(".");
    let branch_enabled = false;

    // Capture the panic message
    let result = std::panic::catch_unwind(|| {
        run_gcov(&gcno_path, branch_enabled, &working_dir);
    });

    // Check that the panic message contains the expected output
    assert!(result.is_err());
    let panic_message = result.unwrap_err().downcast_ref::<String>().unwrap();
    assert!(panic_message.contains("gcov execution failed on"));
}
</Rust>
