#[test]
fn test_version_parsing_error() {
  let buffer = [97, 100, 99, 103, 255]; // 'adcg' followed by an invalid UTF-8 byte
  let mut reader = GcovReaderBuf { buffer, index: 0 };
  let result = reader.parse_version();
  assert!(result.is_err());
  let err = result.unwrap_err();
  assert!(matches!(err, GcovError::Str(_)));
}
