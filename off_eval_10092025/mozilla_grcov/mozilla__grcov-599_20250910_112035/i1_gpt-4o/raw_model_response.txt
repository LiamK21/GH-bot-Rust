use super::main;
use std::process::Command;

<Rust>
#[test]
fn test_cobertura_output_type() {
    let output = Command::new("cargo")
        .arg("run")
        .arg("--")
        .arg("--output-type")
        .arg("cobertura")
        .output()
        .expect("Failed to execute command");

    let stderr = String::from_utf8_lossy(&output.stderr);
    assert!(!stderr.contains("is not a supported output type"), "Cobertura output type should be supported");
}
</Rust>
