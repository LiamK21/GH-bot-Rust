#[test]
fn test_producer_channel_update() {
    let tmp_dir = tempdir().unwrap().path().to_path_buf();
    let paths = vec!["test.rs".to_string()];
    let (sender, _) = crossbeam::channel::bounded(1);
    let result = producer(&tmp_dir, &paths, &sender, false, false);

}
