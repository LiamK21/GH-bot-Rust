<imports> 
use super::output_html;
use std::fs;
use std::path::Path;
use std::thread;
</imports>
<Rust>
#[test]
fn test_output_html_with_subpath() {
  let results: &[(&str, &str, &str)] = &[];
  let output_dir = Some(Path::new("./coverage/subpath"));
  let num_threads = 1;
  let branch_enabled = true;
  let output_config_file = None;

  fs::remove_dir_all("./coverage").ok();

  output_html(results, output_dir, num_threads, branch_enabled, output_config_file);

  assert!(Path::new("./coverage/subpath").is_dir());
}</Rust>
