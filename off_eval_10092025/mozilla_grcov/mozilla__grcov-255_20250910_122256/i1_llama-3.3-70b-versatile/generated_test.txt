#[test]
fn test_output_activedata_etl_with_file() {
  let output_file = "test_output.txt";
  let results = vec![(1, Path::new("test.txt"), vec![1, 0, 1])];
  output_activedata_etl(results.into_iter(), Some(&output_file.to_string()));
  let mut file = File::open(output_file).unwrap();
  let mut contents = String::new();
  file.read_to_string(&mut contents).unwrap();
 assert!(contents.len() > 0);
  std::fs::remove_file(output_file).unwrap();
}
