<imports>
use super::is_covered;
use std::collections::HashMap;
</imports>
<Rust>
#[test]
fn test_is_covered_with_all_lines_uncovered() {
    struct CovResult {
        lines: HashMap<String, u32>,
        functions: HashMap<String, Function>,
    }

    struct Function {
        executed: bool,
    }

    let mut lines = HashMap::new();
    lines.insert("line1".to_string(), 0);
    lines.insert("line2".to_string(), 0);

    let mut functions = HashMap::new();
    functions.insert("func1".to_string(), Function { executed: false });

    let result = CovResult { lines, functions };

    let expected = false;
    let actual = is_covered(&result);

    assert_eq!(expected, actual);
}
</Rust>
