<imports> 
use super::profraws_to_lcov;
use std::path::PathBuf;
use std::thread;
use std::time::Duration;
use rayon::prelude::{IntoParallelIterator, ParallelIterator};
</imports>
<Rust>
#[test]
fn test_profraws_to_lcov_performance() {
  let profraw_paths = vec![PathBuf::from("test.profraw")];
  let binary_path = PathBuf::from("test_binary");
  let working_dir = PathBuf::from("/tmp");

  // Before patch, this should take a long time
  let start = std::time::Instant::now();
  let _ = profraws_to_lcov(&profraw_paths, &binary_path, &working_dir).unwrap();
  let duration = start.elapsed();

  // With patch, this should be significantly faster
  assert!(duration < Duration::from_secs(10));
}</Rust>
