#[test]
fn test_get_xml_attribute() {
    let xml = r#"<element key="value" />"#;
    let mut reader = Reader::from_str(xml);
    reader.trim_text(true);
    let mut buf = Vec::new();
    if let Ok(Event::Start(ref e)) = reader.read_event_into(&mut buf) {
        let result = get_xml_attribute(&reader, e, "key");
        assert_eq!(result.unwrap(), "value");
    } else {
        panic!("Failed to read XML event");
    }
}
