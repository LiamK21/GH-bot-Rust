#[test]
fn test_rewrite_paths_skips_symlinks() {
    let result_map = HashMap::new();
    let source_dir = Some(PathBuf::from("/path/to/source"));
    let path_mapping = None;
    let prefix_dir = None;
    let ignore_not_existing = false;
    let to_ignore_dirs = vec![];
    let filter_option = None;

    let result: Vec<_> = rewrite_paths(
        result_map,
        path_mapping,
        source_dir,
        prefix_dir,
        ignore_not_existing,
        to_ignore_dirs,
        filter_option,
    ).collect();

    assert!(result.is_empty(), "Expected no paths to be rewritten due to symlink filtering");
}
