#[test]
fn test_bundled_resources() {
  let output_config_file = None;
  let branch_enabled = false;
  let precision = 0;
  let no_date = true;
  let html_resources = HtmlResources::Bundled;
  let (tera, config) = get_config(output_config_file, branch_enabled, precision, no_date, html_resources);
  let output = Path::new("/tmp/output");
  fs::create_dir_all(output).unwrap();
  super::gen_bundled_resources(&config, output);
  assert!(output.join("bulma.min.css").exists());
}
