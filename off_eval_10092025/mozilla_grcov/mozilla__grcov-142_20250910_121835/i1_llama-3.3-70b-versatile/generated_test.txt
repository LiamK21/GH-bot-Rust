#[test]
fn test_producer_with_jacoco_xml() {
  let queue: Arc<std::sync::mpsc::Sender<String>> = Arc::new(std::sync::mpsc::sender());
  let tmp_dir = TempDir::new("grcov").expect("Failed to create temporary directory");
  let tmp_path = tmp_dir.path().to_owned();
  let paths = vec!["test/jacoco1.zip".to_string()];
  producer(&tmp_path, &paths, &queue, false, false);
  let result = queue.try_recv().unwrap();
  assert!(result.contains("jacoco/basic-jacoco.xml"));
}
