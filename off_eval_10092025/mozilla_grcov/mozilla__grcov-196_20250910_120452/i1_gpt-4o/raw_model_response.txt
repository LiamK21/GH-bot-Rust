<imports>
use super::rewrite_paths;
use std::collections::HashMap;
use std::path::PathBuf;
use crate::defs::CovResultMap;
use crate::empty_result;
use crate::canonicalize_path;
</imports>
<Rust>
#[test]
fn test_rewrite_paths_with_rewrite_partial_relative_paths() {
    let mut result_map: CovResultMap = HashMap::new();
    result_map.insert("main.cpp".to_string(), empty_result!());
    let results = rewrite_paths(
        result_map,
        None,
        Some(canonicalize_path("tests").unwrap()),
        None,
        true,
        Vec::new(),
        None,
    );
    let mut count = 0;
    for (abs_path, rel_path, result) in results {
        count += 1;
        assert!(abs_path.is_absolute());
        assert!(abs_path.ends_with("tests/main.cpp"));
        assert_eq!(rel_path, PathBuf::from("main.cpp"));
        assert_eq!(result, empty_result!());
    }
    assert_eq!(count, 1);
}
</Rust>
