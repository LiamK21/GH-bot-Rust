<imports>
use super::make_context;
use tera::Context;
</imports>
<Rust>
#[test]
fn test_make_context() {
    let context = make_context();
    let expected_context = {
        let mut ctx = Context::new();
        let ver = std::env::var("BULMA_VERSION").unwrap_or_else(|_| "0.9.3".into());
        ctx.insert("bulma_version", &ver);
        ctx
    };
    assert_eq!(context.get("bulma_version"), expected_context.get("bulma_version"));
}
</Rust>
