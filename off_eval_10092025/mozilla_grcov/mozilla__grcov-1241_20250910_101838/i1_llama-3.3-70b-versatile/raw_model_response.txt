#[test]
fn test_lcov_parser_malformed_data() {
  let buf = "SF:http_3a_2f_2fweb_2dplatform_2etest_2a8000_2fjs_2dcontext_2dmenu_2d01_2dhtml_2c_20about_3a_blank
DA:152,4
DA:153,4
DA:154,8
DA:156,12
DA
TN:http_3a_2f_2fweb_2dplatform_2etest_2a8000_2freferrer_2dpolicy_2fgen_2fsrcdoc_2dinherit_2emeta_2funset_2fiframe_2dtag_2ehtml_2c_20about_3a_blank".as_bytes().to_vec();
  let result = parse_lcov(buf, true);
  assert!(result.is_err());
  let error = result.unwrap_err();
  assert_eq!(error.to_string(), "Invalid record: 'DA at line 6'");
}</Rust>
