#[test]
fn test_lcov_parser_malformed_key() {
    let buf = "SF:source_file.rs
DA:1,1
DA:2,1
DA:3,1
XX:4,1
DA:5,1"
    .as_bytes()
    .to_vec();
    let result = parse_lcov(buf, true);
    assert!(result.is_err());
    let error = result.unwrap_err();
    assert_eq!(error.to_string(), "Invalid key at line 4");
}
