#[test]
fn test_nargs_compute_minmax() {
    let mut state = State::default();
    let mut metrics = Metrics::default();
    
    // Initialize with example values
    metrics.nargs = Stats {
        fn_nargs: 2,
        closure_nargs: 3,
        fn_nargs_sum: 4,
        closure_nargs_sum: 5,
        fn_nargs_min: 1,
        closure_nargs_min: 2,
        fn_nargs_max: 3,
        closure_nargs_max: 4,
        total_functions: 2,
        total_closures: 2,
    };
    
    state.space.metrics = metrics;
    compute_minmax(&mut state);
    
    let expected_min = 1;
    let expected_max = 3;
    assert_eq!(state.space.metrics.nargs.fn_args_min(), expected_min as f64);
    assert_eq!(state.space.metrics.nargs.fn_args_max(), expected_max as f64);
}
