<imports> 
use crate::metrics::nom::Stats;
use std::usize;
</imports>
<Rust>
#[test]
fn test_compute_minmax() {
  let mut stats = Stats::default();
  stats.functions = 5;
  stats.closures = 3;
  stats.compute_minmax();
  assert_eq!(stats.functions_min, 5);
  assert_eq!(stats.functions_max, 5);
  assert_eq!(stats.closures_min, 3);
  assert_eq!(stats.closures_max, 3);
  let mut other_stats = Stats::default();
  other_stats.functions = 10;
  other_stats.closures = 2;
  stats.merge(&other_stats);
  stats.compute_minmax();
  assert_eq!(stats.functions_min, 5);
  assert_eq!(stats.functions_max, 10);
  assert_eq!(stats.closures_min, 2);
  assert_eq!(stats.closures_max, 3);
}</Rust>
