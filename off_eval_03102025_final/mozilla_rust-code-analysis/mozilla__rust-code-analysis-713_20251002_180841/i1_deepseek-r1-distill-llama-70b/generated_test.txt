#[test]
fn test_cognitive_min_max() {
    let mut stats = Stats::default();
    stats.structural = 5;
    stats.compute_minmax();

    let mut other = Stats::default();
    other.structural = 3;
    stats.merge(&other);
    stats.compute_minmax();

    assert_eq!(stats.cognitive_min(), 3);
    assert_eq!(stats.cognitive_max(), 5);
    assert_eq!(stats.cognitive_sum(), 8);
}
