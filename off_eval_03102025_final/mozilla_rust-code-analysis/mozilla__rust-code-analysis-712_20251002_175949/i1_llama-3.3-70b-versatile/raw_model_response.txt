<imports> 
use crate::metrics::exit::Stats;
use std::fmt;
</imports>
<Rust>
#[test]
fn test_exit_minmax() {
  let mut stats = Stats::default();
  stats.exit = 5;
  stats.compute_minmax();
  assert_eq!(stats.exit_min, 5);
  assert_eq!(stats.exit_max, 5);
  let mut stats2 = Stats::default();
  stats2.exit = 3;
  stats2.compute_minmax();
  stats.merge(&stats2);
  assert_eq!(stats.exit_min, 3);
  assert_eq!(stats.exit_max, 5);
}</Rust>
