<imports> 
use crate::metrics::loc::Stats;
use std::collections::HashSet;
</imports>
<Rust>
#[test]
fn test_loc_average() {
  let mut stats = Stats::default();
  stats.sloc = 10.0;
  stats.ploc = 5.0;
  stats.lloc = 3.0;
  stats.cloc = 2.0;
  stats.blank = 1.0;
  stats.space_count = 2;

  let sloc_average = stats.sloc_average();
  let ploc_average = stats.ploc_average();
  let lloc_average = stats.lloc_average();
  let cloc_average = stats.cloc_average();
  let blank_average = stats.blank_average();

  assert_eq!(sloc_average, 5.0);
  assert_eq!(ploc_average, 2.5);
  assert_eq!(lloc_average, 1.5);
  assert_eq!(cloc_average, 1.0);
  assert_eq!(blank_average, 0.5);
}
</Rust>
