#[test]
fn test_profraw_producer_send_paths() {
  let (sender, _) = channel();
  let mut profraws = FxHashMap::default();
  let archive = Archive {};
  profraws.insert("test.profraw".to_string(), vec![&archive]);
  let tmp_dir = PathBuf::from("tmp");
  profraw_producer(&tmp_dir, &profraws, &sender);
  let received = sender.recv().unwrap().unwrap();
  match received.item {
    ItemType::Paths(ref paths) => assert!(paths.len() > 0),
    _ => assert!(false),
  }
}
