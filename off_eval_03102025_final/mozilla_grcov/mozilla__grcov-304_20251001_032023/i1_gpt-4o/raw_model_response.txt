<imports>
use std::path::PathBuf;
use super::run_gcov;
use super::GcovError;
</imports>
<Rust>
#[test]
fn test_run_gcov_handles_process_failure() {
    let gcno_path = PathBuf::from("invalid_path");
    let branch_enabled = false;
    let working_dir = PathBuf::from(".");

    let result = run_gcov(&gcno_path, branch_enabled, &working_dir);

    assert!(matches!(result, Err(GcovError::ProcessFailure)));
}
</Rust>
