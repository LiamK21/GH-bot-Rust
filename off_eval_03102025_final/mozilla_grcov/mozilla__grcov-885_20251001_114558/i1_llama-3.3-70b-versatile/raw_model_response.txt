<imports> 
use super::output_markdown;
use std::path::PathBuf;
use std::collections::BTreeMap;
use std::io::Write;
use tempfile::tempdir;
use std::fs::File;
use std::io::Read;
</imports>
<Rust>
#[test]
fn test_output_markdown() {
  let tmp_dir = tempdir().expect("Failed to create temporary directory");
  let file_name = "test_output_markdown";
  let file_path = tmp_dir.path().join(file_name);

  let results = vec![
    (
      PathBuf::from("foo/bar/a.cpp"),
      PathBuf::from("foo/bar/a.cpp"),
      (
        BTreeMap::from([(1, 10), (2, 11)]),
        BTreeMap::new(),
        std::collections::HashMap::new(),
      ),
    ),
    (
      PathBuf::from("foo/bar/b.cpp"),
      PathBuf::from("foo/bar/b.cpp"),
      (
        BTreeMap::from([(1, 0), (2, 10), (4, 10), (5, 0), (7, 0)]),
        BTreeMap::new(),
        std::collections::HashMap::new(),
      ),
    ),
  ];

  let mut file = File::create(file_path.clone()).expect("Failed to create file");
  file.set_len(0).expect("Failed to truncate file");

  output_markdown(Box::new(results.into_iter()), Some(file_path.as_path()));

  let mut actual = String::new();
  File::open(file_path).expect("Failed to open file").read_to_string(&mut actual).expect("Failed to read file");

  let expected = "| file          | coverage | covered | missed_lines |\n|---------------|----------|---------|--------------|\n| foo/bar/a.cpp | 100%     | 2 / 2   |              |\n| foo/bar/b.cpp | 40%      | 2 / 5   | 1, 5-7       |\n\nTotal coverage: 57%\n";
  assert_eq!(actual, expected);
}
</Rust>
