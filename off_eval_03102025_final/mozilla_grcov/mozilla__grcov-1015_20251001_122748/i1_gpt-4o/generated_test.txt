#[test]
fn test_profraws_to_lcov_parallel_execution() {
    let profraw_paths = vec![PathBuf::from("dummy.profraw")];
    let binary_path = Path::new("dummy_binary");
    let working_dir = Path::new(".");

    fs::write("dummy.profraw", b"dummy data").unwrap();
    fs::write("dummy_binary", b"dummy binary data").unwrap();

    let result = profraws_to_lcov(&profraw_paths, &binary_path, &working_dir);

    assert!(result.is_ok());

    fs::remove_file("dummy.profraw").unwrap();
    fs::remove_file("dummy_binary").unwrap();
}
