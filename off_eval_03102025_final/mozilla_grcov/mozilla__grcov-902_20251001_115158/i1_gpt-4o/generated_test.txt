#[test]
fn test_profraws_to_lcov_handles_long_argument_list() {
    let profraw_paths = vec![
        PathBuf::from("/path/to/profraw1"),
        PathBuf::from("/path/to/profraw2"),
        PathBuf::from("/path/to/profraw3"),
    ];
    let binary_path = PathBuf::from("/path/to/binary");
    let working_dir = PathBuf::from("/path/to/working_dir");

    let result = profraws_to_lcov(&profraw_paths, &binary_path, &working_dir);

    assert!(result.is_ok());
}
