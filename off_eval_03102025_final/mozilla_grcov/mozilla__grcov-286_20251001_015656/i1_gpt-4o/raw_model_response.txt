<imports>
use super::get_abs_path;
use std::path::PathBuf;
</imports>
<Rust>
#[test]
fn test_get_abs_path_normalizes_path() {
    let source_dir = Some(PathBuf::from("/source"));
    let rel_path = PathBuf::from("./foo/./bar/./oof/");
    let mut cache = None;
    
    let (_abs_path, normalized_rel_path) = get_abs_path(&source_dir, rel_path, &mut cache);
    
    assert_eq!(normalized_rel_path, PathBuf::from("foo/bar/oof"));
}
</Rust>
