<imports> 
use std::env;
use anyhow::{bail, Context, Result};
use uniffi::{generate_bindings, TargetLanguage};
</imports>
<Rust>
#[test]
fn test_parse_language() {
  let valid_language = "kotlin";
  let invalid_language = "ruby";
  
  let expected_valid = Ok(TargetLanguage::Kotlin);
  let expected_invalid = Err(anyhow::Error::msg("Unknown language"));
  
  let actual_valid = parse_language(valid_language);
  let actual_invalid = parse_language(invalid_language);
  
  assert_eq!(actual_valid, expected_valid);
  assert_eq!(actual_invalid, expected_invalid);
}
</Rust>
