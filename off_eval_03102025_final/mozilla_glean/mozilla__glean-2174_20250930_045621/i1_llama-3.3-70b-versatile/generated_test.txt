#[test]
fn test_initialize_core_metrics_sets_os_version() {
  let mut glean = Glean::new();
  let client_info = ClientInfoMetrics {
    os_version: Some("Test OS Version".to_string()),
    ..Default::default()
  };
  initialize_core_metrics(&glean, &client_info);
  assert_eq!(glean.get("internal_metrics.os_version"), "Test OS Version");
}
