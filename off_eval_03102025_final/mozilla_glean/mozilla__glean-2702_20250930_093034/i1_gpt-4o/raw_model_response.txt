#[test]
fn test_set_experimentation_id() {
    let new_experimentation_id = "new_experiment_id".to_string();
    set_experimentation_id(new_experimentation_id.clone());
    let actual_experimentation_id = test_get_experimentation_id();
    assert_eq!(Some(new_experimentation_id), actual_experimentation_id);
}
