<imports> 
use std::time::Duration;
use glean_core::traits::Timespan;
</imports>
<Rust>
#[test]
fn test_set_raw() {
  let timespan: Box<dyn Timespan> = Box::new(TimespanMetric::new("test_timespan"));
  let elapsed = Duration::from_millis(100);
  timespan.set_raw(elapsed);
  let actual = timespan.test_get_value(None).unwrap();
  let expected = elapsed.as_millis() as u64;
  assert_eq!(actual, expected);
}
</Rust>
