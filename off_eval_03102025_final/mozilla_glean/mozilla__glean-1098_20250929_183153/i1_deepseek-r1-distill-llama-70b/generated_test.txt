#[test]
fn test_ping_body_size_validation() {
    let body = vec![b'a'; 1024 * 1024 + 1];
    let result = PingRequest::builder("test", 1024 * 1024)
        .document_id("test-id")
        .path("test-path")
        .body(body)
        .build();

    assert!(matches!(result, Err(ErrorKind::PingBodyOverflow(_))));
}
