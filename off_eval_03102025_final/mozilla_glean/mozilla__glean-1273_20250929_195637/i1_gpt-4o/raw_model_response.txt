<imports>
use super::BooleanMetric;
use crate::dispatcher;
use glean_core::CommonMetricData;
use glean_core::metrics::BooleanMetric as CoreBooleanMetric;
use std::sync::Arc;
</imports>
<Rust>
#[test]
fn test_boolean_metric_set_dispatch() {
    let meta = CommonMetricData {
        name: "test_metric".into(),
        category: "test_category".into(),
        send_in_pings: vec!["test_ping".into()],
        ..Default::default()
    };
    let metric = BooleanMetric::new(meta);
    let core_metric = Arc::new(CoreBooleanMetric::new(meta));
    let expected_value = true;

    metric.set(expected_value);

    dispatcher::block_on_queue();

    let actual_value = core_metric.test_get_value("test_ping");
    assert_eq!(actual_value, Some(expected_value));
}
</Rust>
