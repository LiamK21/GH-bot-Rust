#[test]
fn test_first_run_hour_resolution() {
  let dir = tempdir().unwrap();
  let tmpname = dir.path().display().to_string();
  {
    let (mut glean, _) = new_glean(Some(tmpname.clone()));
    glean.core_metrics.first_run_date.set(&glean, Some("2022-01-01".to_string()));
    glean.core_metrics.first_run_hour.set(&glean, Some("2022-01-01T00:00:00".to_string()));
    assert_eq!(glean.core_metrics.first_run_hour.get_value(&glean, "metrics"), Some("2022-01-01T00:00:00".to_string()));
  }
}
