<imports>
use super::glean_enable_logging;
use log::{self, Level};
use android_logger;
</imports>
<Rust>
#[test]
fn test_logging_filter_for_glean_modules() {
    glean_enable_logging();
    
    let logger = android_logger::Config::default()
        .with_min_level(Level::Debug)
        .with_tag("libglean_ffi");
    
    let filter = android_logger::FilterBuilder::new()
        .filter_module("glean_ffi", log::LevelFilter::Debug)
        .filter_module("glean_core", log::LevelFilter::Debug)
        .filter_module("glean", log::LevelFilter::Debug)
        .build();
    
    assert_eq!(logger.filter(), filter);
}
</Rust>
