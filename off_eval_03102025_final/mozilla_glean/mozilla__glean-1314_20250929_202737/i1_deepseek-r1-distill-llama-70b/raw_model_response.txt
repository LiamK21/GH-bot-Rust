#[test]
fn test_experiment_extra_data() {
    let experiment_id = "test_experiment".to_string();
    let branch = "test_branch".to_string();
    let mut extra = HashMap::new();
    extra.insert("key".to_string(), "value".to_string());

    set_experiment_active(experiment_id.clone(), branch, Some(extra));

    assert!(test_is_experiment_active(experiment_id.clone()));

    let data = test_get_experiment_data(experiment_id.clone());
    assert_eq!(data.branch, branch);
    assert_eq!(data.extra.unwrap()["key"], "value");

    set_experiment_inactive(experiment_id.clone());
    assert!(!test_is_experiment_active(experiment_id));
}
