#[test]
fn test_timing_distribution_methods() {
    let metric = TimingDistributionMetric(Arc::new(RwLock::new(MockTimingDistribution::new())));

    let timer_id = metric.start();
    sleep(Duration::from_millis(10));
    metric.stop_and_accumulate(timer_id);

    let stored_value = metric.0.read().unwrap().get_stored_value();
    assert!(stored_value > 0);

    let timer_id = metric.start();
    metric.cancel(timer_id);
    let stored_value_after_cancel = metric.0.read().unwrap().get_stored_value();
    assert_eq!(stored_value, stored_value_after_cancel);
}
