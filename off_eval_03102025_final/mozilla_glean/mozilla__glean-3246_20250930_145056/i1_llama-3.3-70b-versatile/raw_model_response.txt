<imports> 
use super::Histogram;
use std::collections::HashMap;
</imports>
<Rust>
#[test]
fn test_snapshot_values_removes_infill_0_buckets() {
  let mut histogram = Histogram::new();
  histogram.add(1);
  histogram.add(3);
  let actual = histogram.snapshot_values();
  let expected: HashMap<u64, u64> = [(1, 1), (3, 1)].iter().cloned().collect();
  assert_eq!(actual, expected);
}</Rust>
