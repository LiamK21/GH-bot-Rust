#[test]
fn test_get_os_version_initialized_correctly() {
  let mut glean = Glean::new();
  let client_info = ClientInfoMetrics::new();
  let channel = Some("test_channel".to_string());
  initialize_core_metrics(&mut glean, &client_info, channel);
  let os_version = system::get_os_version();
  assert_ne!(os_version, "unknown");
}
