<imports> 
use std::collections::HashMap;
use super::EventMetric;
use std::sync::Arc;
use std::thread;
use std::time::Duration;
</imports>
<Rust>
#[test]
fn test_record_with_time() {
  let event = EventMetric {
    inner: Arc::new(()),
    extra_keys: std::marker::PhantomData,
  };
  let timestamp = 1643723900;
  let mut extra = HashMap::new();
  extra.insert(1, "value1".to_string());
  event.record_with_time(timestamp, extra.clone());
  thread::sleep(Duration::from_millis(10));
  let mut another_extra = HashMap::new();
  another_extra.insert(2, "value2".to_string());
  event.record_with_time(timestamp, another_extra);
}</Rust>
