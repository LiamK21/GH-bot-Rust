#[test]
fn test_global_glean_returns_none_if_not_initialized() {
  let glean = global_glean();
  assert!(glean.is_none());
  let result = setup_glean(Glean {});
  assert!(result.is_ok());
  let glean = global_glean();
  assert!(glean.is_some());
}
