<imports>
use super::CustomDistributionMetric;
use crate::Glean;
use crate::CommonMetricData;
use crate::metrics::Metric;
</imports>
<Rust>
#[test]
fn test_accumulate_samples_signed_doesnt_record_when_should_not() {
    let glean = Glean::new("test_app_id", "test_app_version", "test_channel", false, None);
    let metric_data = CommonMetricData::new("category", "name", "test-storage");
    let custom_distribution = CustomDistributionMetric::new(metric_data);

    // Attempt to accumulate samples when should_record is false
    custom_distribution.accumulate_samples_signed(&glean, vec![1, 2, 3]);

    // Verify that no data was recorded
    let recorded_data = glean.database().get_metric(&metric_data.identifier(&glean));
    assert!(recorded_data.is_none());
}
</Rust>
