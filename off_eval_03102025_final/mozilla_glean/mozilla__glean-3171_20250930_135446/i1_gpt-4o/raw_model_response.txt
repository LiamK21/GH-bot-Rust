use super::DualLabeledCounterMetric;
use crate::metrics::MetricType;

#[test]
fn test_dual_labeled_counter_metric_implements_metric_type() {
    let dummy_metric_data = CommonMetricDataInternal::default();
    let dual_labeled_counter = DualLabeledCounterMetric::new(dummy_metric_data, None);
    
    let expected_meta = dual_labeled_counter.counter.meta();
    let actual_meta = dual_labeled_counter.meta();
    
    assert_eq!(expected_meta, actual_meta);
}
