#[test]
fn test_glean_initialization() {
    let glean_cfg = glean_core::Configuration::new("test", "0.0.1");
    let glean = Glean::new(glean_cfg).unwrap();
    let glean_mutex = Mutex::new(glean);
    let _ = glean_core::setup_glean(glean_mutex.lock().unwrap()).unwrap();
    let glean = glean_core::global_glean().lock().unwrap();
    assert!(glean.is_first_run());
}
