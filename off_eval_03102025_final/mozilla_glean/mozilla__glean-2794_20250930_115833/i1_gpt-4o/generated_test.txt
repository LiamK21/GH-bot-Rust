#[test]
fn test_exponential_range_memory_usage() {
    let min = 1;
    let max = 100;
    let bucket_count = 10;
    let result = exponential_range(min, max, bucket_count);
    let expected_size = std::mem::size_of::<Vec<u64>>() + result.capacity() * std::mem::size_of::<u64>();
    let actual_size = malloc_size_of::malloc_size_of_val(&result);
    assert_eq!(expected_size, actual_size);
}
