<imports>
use std::mem;
use malloc_size_of::MallocSizeOf;
use malloc_size_of_derive::MallocSizeOf;
use crate::common_metric_data::CommonMetricData;
</imports>
<Rust>
#[test]
fn test_common_metric_data_size_of() {
  let data = CommonMetricData {
    name: "test_name".to_string(),
    ..Default::default()
  };
  let size = mem::size_of_val(&data);
  let malloc_size = data.size_of(&mut malloc_size_of::MallocSizeOfOps::new());
  assert_eq!(size, malloc_size);
}
</Rust>
