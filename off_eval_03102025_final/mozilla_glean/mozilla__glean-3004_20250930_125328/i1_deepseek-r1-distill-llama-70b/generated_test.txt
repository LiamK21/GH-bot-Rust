#[test]
fn test_get_debug_view_tag_and_log_pings() {
    // Test debug view tag
    let expected_tag = "test_tag";
    let result = glean_set_debug_view_tag(expected_tag);
    assert!(result);

    let actual_tag = glean_get_debug_view_tag();
    assert!(actual_tag.is_some());
    assert_eq!(actual_tag.unwrap(), expected_tag);

    // Test log pings
    let expected_log_pings = true;
    glean_set_log_pings(expected_log_pings);
    let actual_log_pings = glean_get_log_pings();
    assert_eq!(actual_log_pings, expected_log_pings);

    // Test registered ping names
    let pings = get_registered_ping_names();
    assert!(!pings.is_empty());
    assert!(pings.contains(&"baseline".to_string()));
}
