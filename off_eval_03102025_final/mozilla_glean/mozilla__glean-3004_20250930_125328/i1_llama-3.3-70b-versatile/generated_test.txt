#[test]
fn test_get_current_debug_view_tag_and_log_pings() {
  let _ = set_debug_view_tag("test_tag");
  let _ = set_log_pings(true);
  let debug_view_tag = glean_get_debug_view_tag().unwrap();
  let log_pings = glean_get_log_pings();
  let registered_ping_names = get_registered_ping_names();
  assert_eq!(debug_view_tag, "test_tag");
  assert_eq!(log_pings, true);
  assert!(!registered_ping_names.is_empty());
}
