<imports> 
use crate::Glean;
use crate::additional_metrics::AdditionalMetrics;
use crate::metrics::CounterMetric;
use tempfile::tempdir;

<Rust>
#[test]
fn test_invalid_timezone_offset_not_recorded() {
  let dir = tempdir().unwrap();
  let (glean, _) = new_glean(Some(dir));
  let metric = &glean.additional_metrics.invalid_timezone_offset;
  assert_eq!(None, metric.test_get_value(&glean, "metrics"));
}</Rust>
