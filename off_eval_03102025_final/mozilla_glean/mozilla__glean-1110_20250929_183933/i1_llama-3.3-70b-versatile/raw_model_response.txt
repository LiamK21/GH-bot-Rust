<imports> 
use super::PingDirectoryManager;
use std::fs;
use std::path::PathBuf;
</imports>
<Rust>
#[test]
fn test_pending_pings_directory_size_is_calculated_correctly() {
  let mut dir = tempfile::TempDir::new().unwrap();
  let mut ping_dir = dir.path().join("pending_pings");
  fs::create_dir(&ping_dir).unwrap();

  let directory_manager = PingDirectoryManager::new(dir.path());
  let ping_file_name = "ping_file.txt";
  let ping_path = ping_dir.join(ping_file_name);
  fs::File::create(&ping_path).unwrap();

  let ping_data = directory_manager.process_dirs();
  let calculated_size = ping_data.pending_pings.iter().map(|(size, _)| size).sum::<u64>();
  assert_eq!(calculated_size, 0);
}</Rust>
