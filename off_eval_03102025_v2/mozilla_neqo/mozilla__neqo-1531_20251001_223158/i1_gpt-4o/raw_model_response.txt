<imports>
use super::server_version_information_failed;
use crate::NeqoQlog;
use crate::Version;
use crate::WireVersion;
</imports>
<Rust>
#[test]
fn test_server_version_information_failed() {
    let mut qlog = NeqoQlog::new();
    let server_versions = vec![Version::default()];
    let client_version = WireVersion::default();
    
    server_version_information_failed(&mut qlog, &server_versions, client_version);
    
    let events = qlog.events();
    assert_eq!(events.len(), 1);
    if let Some(event) = events.first() {
        if let Some(event_data) = &event.data {
            if let EventData::VersionInformation(version_info) = event_data {
                assert_eq!(version_info.client_versions, Some(vec![format!("{:02x}", client_version)]));
                assert_eq!(version_info.server_versions, Some(server_versions.iter().map(|v| format!("{:02x}", v.wire_version())).collect::<Vec<_>>()));
                assert!(version_info.chosen_version.is_none());
            } else {
                panic!("Expected VersionInformation event data");
            }
        } else {
            panic!("Event data is missing");
        }
    } else {
        panic!("No events logged");
    }
}
</Rust>
