<imports> 
use std::os::raw::c_char;
use std::ffi::CString;
use neqo_crypto::Aead;
</imports>
<Rust>
#[test]
fn test_aead_init() {
  let version: u32 = 1;
  let cipher: u32 = 2;
  let secret: u32 = 3;
  let p = CString::new("test").unwrap();
  let ctx: u32 = 4;
  let result_before = Aead::init(version, cipher, secret, p.as_ptr() as *const i8, p.len() as u32, &ctx);
  let result_after = Aead::init(version, cipher, secret, p.as_ptr() as *const c_char, p.len() as u32, &ctx);
  assert_ne!(result_before, result_after);
}</Rust>
