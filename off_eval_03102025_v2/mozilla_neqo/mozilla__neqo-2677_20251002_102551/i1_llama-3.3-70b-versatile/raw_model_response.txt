use std::io::Cursor;
use super::{Encoder, Buffer};

<Rust>
#[test]
fn test_encoder_does_not_require_ownership() {
    let mut buf = [0; 16];
    let mut enc = Encoder::new_borrowed_slice(&mut buf[..]);
    enc.encode_byte(41);
    assert_eq!(enc.len(), 1);
    assert_eq!(buf[0], 41);
}
</Rust>
