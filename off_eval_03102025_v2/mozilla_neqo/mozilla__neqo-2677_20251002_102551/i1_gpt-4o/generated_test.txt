#[test]
fn test_encoder_does_not_require_ownership() {
    let mut handler = TransportParametersHandler::default();
    let mut buffer = [0u8; 128];
    let result = handler.write_params(&mut buffer, true, TLS_HS_CLIENT_HELLO);
    assert!(result.is_ok());
    assert!(result.unwrap() <= buffer.len());
}
