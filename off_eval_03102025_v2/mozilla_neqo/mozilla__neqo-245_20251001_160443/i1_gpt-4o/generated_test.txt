#[test]
fn test_pmtu_calculation() {
    let mut conn = Connection::new();

    // Test for IPv4
    conn.paths = Some(Path { local: IpAddr::V4(Ipv4Addr::new(192, 168, 0, 1)), ..Default::default() });
    assert_eq!(conn.pmtu(), 1252);

    // Test for IPv6
    conn.paths = Some(Path { local: IpAddr::V6(Ipv6Addr::new(0, 0, 0, 0, 0, 0, 0, 1)), ..Default::default() });
    assert_eq!(conn.pmtu(), 1232);

    // Test for no path
    conn.paths = None;
    assert_eq!(conn.pmtu(), 1280);
}
