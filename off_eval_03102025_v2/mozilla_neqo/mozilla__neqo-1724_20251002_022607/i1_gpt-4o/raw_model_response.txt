<imports>
use super::QuicParameters;
use neqo_transport::Version;
</imports>
<Rust>
#[test]
fn test_quic_parameters_get() {
    let quic_params = QuicParameters {
        quic_version: vec![Version::Draft29, Version::Version1],
        max_streams_bidi: 16,
        max_streams_uni: 16,
        idle_timeout: 30,
        congestion_control: neqo_transport::CongestionControlAlgorithm::NewReno,
        pacing: false,
        preferred_address_v4: None,
        preferred_address_v6: None,
    };

    let connection_params = quic_params.get("h3");
    let versions = connection_params.get_versions();

    assert_eq!(versions.0, Version::Draft29);
    assert_eq!(versions.1, vec![Version::Version1]);
}
</Rust>
