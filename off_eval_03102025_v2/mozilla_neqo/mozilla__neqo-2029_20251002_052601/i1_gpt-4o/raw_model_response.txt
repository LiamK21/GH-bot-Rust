<imports>
use super::Connection;
use crate::packet::Packet;
use crate::path::Path;
use crate::time::Instant;
use crate::Error;
</imports>
<Rust>
#[test]
fn test_input_path_untracked_space() {
    let mut connection = Connection::new();
    let path = Path::new();
    let packet = Packet::new_untracked_space();
    let now = Instant::now();

    let result = connection.input_path(&path, &packet, now);

    assert_eq!(result, Err(Error::ProtocolViolation));
}
</Rust>
