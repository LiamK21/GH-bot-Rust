#[test]
fn test_hframe_reader_state_before_frame() {
  let mut hframe_reader = HFrameReader::new();
  assert_eq!(hframe_reader.state, HFrameReaderState::BeforeFrame);
  let mut conn_s = None; // assuming this is initialized properly
  let mut buf = [0; 0];
  let stream_id = 0; // assuming this is a valid stream id
  let result = hframe_reader.receive(&mut conn_s, stream_id);
  assert_eq!(result, Err(Error::MalformedFrame(0xff)));
}
