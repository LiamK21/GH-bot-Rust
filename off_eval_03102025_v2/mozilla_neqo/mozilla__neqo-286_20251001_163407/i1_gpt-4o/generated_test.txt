#[test]
fn test_decode_stream_frame_fill() {
    let enc = Encoder::from_hex("0905010203");
    let mut dec = enc.as_decoder();
    let expected_frame = Frame::Stream {
        stream_id: 5,
        offset: 0,
        data: vec![1, 2, 3],
        fin: true,
        fill: true,
    };
    let actual_frame = decode_frame(&mut dec).unwrap();
    assert_eq!(expected_frame, actual_frame);
}
