#[test]
fn test_sendstream_methods_removed() {
    let mut session = WebTransportSession::new();
    let mut conn = Connection::new(1);

    // Perform operations that would have used the removed methods
    session.set_sendorder(&mut conn, None);
    session.set_fairness(&mut conn, false);

    // Assert that the session still functions correctly
    assert!(session.done());
}
