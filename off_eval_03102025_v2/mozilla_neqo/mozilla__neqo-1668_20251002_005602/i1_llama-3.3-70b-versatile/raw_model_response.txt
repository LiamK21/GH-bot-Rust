<imports> 
use neqo_transport::{Connection, StreamId, streams::SendOrder};
use super::SendStream;
</imports>
<Rust>
#[test]
fn test_sendstream_methods_removed() {
  let mut conn = Connection::new();
  let mut stream: Box<dyn SendStream> = Box::new(WebTransportSession::new());
  let sendorder: Option<SendOrder> = None;
  let fairness = true;
  let result = std::panic::catch_unwind(|| stream.set_sendorder(&mut conn, sendorder));
  assert!(result.is_err());
  let result = std::panic::catch_unwind(|| stream.set_fairness(&mut conn, fairness));
  assert!(result.is_err());
}</Rust>
