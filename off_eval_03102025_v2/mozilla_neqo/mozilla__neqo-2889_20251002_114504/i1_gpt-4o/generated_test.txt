#[test]
fn test_header_from_str_error_messages() {
    let result: Result<Header, FromStrError> = Header::from_str("no_colon");
    assert_eq!(result.unwrap_err().to_string(), "Header string missing colon");

    let result: Result<Header, FromStrError> = Header::from_str(":no_name");
    assert_eq!(result.unwrap_err().to_string(), "Header string missing name");
}
