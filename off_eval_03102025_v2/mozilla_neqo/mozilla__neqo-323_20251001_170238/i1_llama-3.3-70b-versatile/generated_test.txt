#[test]
fn test_initial_packet_padding() {
  let mut conn = Connection::default();
  conn.role = neqo_transport::Role::Client;
  let path = Path::default();
  let epoch: u16 = 0;
  let now = Instant::now();
  let mut out = conn.output_pkt_for_path(now);
  assert_eq!(out.unwrap().as_dgram_ref().unwrap().len(), path.mtu());
}
