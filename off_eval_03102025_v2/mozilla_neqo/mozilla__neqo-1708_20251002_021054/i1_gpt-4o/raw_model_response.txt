<imports>
use super::Socket;
use std::net::SocketAddr;
use std::io;
</imports>

<Rust>
#[test]
fn test_recv_multiple_datagrams() {
    let receiver_addr: SocketAddr = "127.0.0.1:0".parse().unwrap();
    let mut receiver = Socket::bind(receiver_addr).expect("bind to succeed");

    let datagrams = receiver.recv(&receiver_addr).expect("receive to succeed");

    assert!(datagrams.is_empty(), "Expected no datagrams to be received initially");
}
</Rust>
