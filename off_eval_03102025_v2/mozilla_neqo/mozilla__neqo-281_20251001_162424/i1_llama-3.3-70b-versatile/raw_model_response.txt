<imports> 
use neqo_http3::connection::Http3Connection;
use neqo_http3::connection::Http3State;
use neqo_common::CloseError;
</imports>
<Rust>
#[test]
fn test_handle_connection_closing() {
  let mut connection = Http3Connection::default();
  connection.state = Http3State::Closing(CloseError::Application(0));
  let result = connection.handle_connection_closing(CloseError::Application(0));
  assert!(result.is_ok());
}</Rust>
