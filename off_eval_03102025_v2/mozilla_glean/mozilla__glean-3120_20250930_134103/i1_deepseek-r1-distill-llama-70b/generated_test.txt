#[test]
fn test_alloc_size_returns_positive_value() {
    // Initialize Glean
    let _glean = initialize_inner(
        InternalConfiguration::default(),
        ClientInfoMetrics::default(),
        Box::new(|_| ()),
    );

    let ops = &mut MallocSizeOfOps::default();
    let size = alloc_size(ops);
    assert!(size > 0);
}
