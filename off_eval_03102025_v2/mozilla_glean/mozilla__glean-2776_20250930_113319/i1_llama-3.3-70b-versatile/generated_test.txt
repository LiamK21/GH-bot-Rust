#[test]
fn test_event_metric_extra_glean_timestamp_removed() {
  let tmp = TempDir::new().unwrap();
  let config = ConfigurationBuilder::new(true, tmp.path(), "test").build();
  let glean = super::new_glean(Some(config), true);
  let event = EventMetric::new("test", "test_event", "test_description", "".to_string(), None, None);
  let _ = event.set(None);
  let events = event.test_get_value("test");
  assert!(events.unwrap().into_iter().all(|ev| ev.extra.map_or(true, |ex| !ex.contains_key("glean_timestamp"))));
}
