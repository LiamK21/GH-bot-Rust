<imports>
use super::Glean;
use super::PingType;
use log::Level;
use log::Log;
use log::Metadata;
use log::Record;
use log::set_logger;
use log::set_max_level;
</imports>
<Rust>
struct TestLogger {
    pub messages: Vec<String>,
}

impl Log for TestLogger {
    fn enabled(&self, metadata: &Metadata) -> bool {
        metadata.level() <= Level::Info
    }

    fn log(&self, record: &Record) {
        if self.enabled(record.metadata()) {
            self.messages.push(format!("{}", record.args()));
        }
    }

    fn flush(&self) {}
}

#[test]
fn test_glean_disabled_message_logged_as_info() {
    let logger = TestLogger { messages: Vec::new() };
    set_logger(Box::leak(Box::new(logger))).unwrap();
    set_max_level(Level::Info.to_level_filter());

    let glean = Glean::new(); // Assuming a constructor or method to create a Glean instance
    let ping_type = PingType::new(); // Assuming a constructor or method to create a PingType instance

    glean.submit_ping(&ping_type, None).unwrap();

    let logger = unsafe { &*Box::leak(Box::new(logger)) };
    assert!(logger.messages.contains(&"Glean disabled: not submitting any pings.".to_string()));
}
