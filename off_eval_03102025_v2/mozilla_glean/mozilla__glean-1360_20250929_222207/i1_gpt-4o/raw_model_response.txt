<imports>
use std::thread;
use super::Dispatcher;
</imports>
<Rust>
#[test]
fn test_thread_naming() {
    let builder = thread::Builder::new().name("glean.dispatcher".into());
    let handle = builder.spawn(|| {}).expect("Failed to spawn thread");
    let thread_name = handle.thread().name().unwrap_or("");
    assert_eq!(thread_name, "glean.dispatcher");
}
</Rust>
