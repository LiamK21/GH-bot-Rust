<imports> 
use std::thread;
use std::thread::Builder;
</imports>
<Rust>
#[test]
fn test_thread_name_set() {
  let handle = thread::Builder::new()
    .name("test_thread".into())
    .spawn(|| ());
  assert!(handle.is_ok());
  let handle = thread::Builder::new()
    .name("test_thread".into())
    .spawn(move || {
      let thread_name = thread::current().name().unwrap();
      assert_eq!(thread_name, "test_thread");
    });
  handle.unwrap().join().unwrap();
}</Rust>
