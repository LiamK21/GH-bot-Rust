<imports>
use super::TimespanMetric;
use std::sync::{Arc, RwLock};
use std::time::Duration;
</imports>
<Rust>
#[test]
fn test_set_raw() {
    let metric = Arc::new(RwLock::new(TimespanMetric::new()));
    let timespan = TimespanMetric(metric.clone());

    let expected_duration = Duration::new(5, 0);
    timespan.set_raw(expected_duration);

    let actual_duration = metric.read().unwrap().test_get_value(None);
    assert_eq!(actual_duration, Some(expected_duration.as_secs()));
}
</Rust>
