#[test]
fn test_builder_generates_correct_rust_code() {
    let builder = Builder {};
    let output = Command::new("glean_parser")
        .arg("--version")
        .output()
        .expect("Failed to execute command");
    let version = String::from_utf8_lossy(&output.stdout);
    assert!(version.contains("15.0.1"));
}
