<imports>
use super::UuidMetric;
use uuid::Uuid;
use crate::glean::Glean;
use crate::metrics::Metric;
</imports>
<Rust>
#[test]
fn test_uuid_hyphenated_encoding() {
    let glean = Glean::new();
    let uuid_metric = UuidMetric::new();
    let input_uuid = "123e4567e89b12d3a456426655440000";
    let expected_uuid = "123e4567-e89b-12d3-a456-426655440000";

    uuid_metric.record(&glean, input_uuid);

    let stored_value = glean.storage().get(&uuid_metric.meta).unwrap();
    if let Metric::Uuid(actual_uuid) = stored_value {
        assert_eq!(actual_uuid, expected_uuid);
    } else {
        panic!("Stored value is not a UUID");
    }
}
</Rust>
