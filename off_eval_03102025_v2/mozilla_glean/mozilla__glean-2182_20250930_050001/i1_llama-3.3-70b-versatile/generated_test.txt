#[test]
fn test_uuid_hyphenation() {
  let glean = Glean::new();
  let meta = glean_core::metrics::uuid::UuidMetric::new("test_metric");
  let uuid = Uuid::new_v4();
  let hyphenated_uuid = uuid.to_hyphenated();
  let value = hyphenated_uuid.to_string();
  let metric = Metric::Uuid(value.clone());
  glean_core::metrics::uuid::UuidMetric::record(&glean, &meta, &metric);
  let stored_value = glean.storage().get(&meta).unwrap();
  assert_eq!(stored_value, metric);
}
