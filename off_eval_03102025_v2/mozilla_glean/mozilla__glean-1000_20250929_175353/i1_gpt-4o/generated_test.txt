#[test]
fn test_ping_request_creation_without_user_agent() {
    let document_id = "test_document_id";
    let path = "/test/path";
    let body = json!({"key": "value"});

    let ping_request = PingRequest::new(document_id, path, body);

    assert!(!ping_request.headers.contains_key("userAgent"));
}
