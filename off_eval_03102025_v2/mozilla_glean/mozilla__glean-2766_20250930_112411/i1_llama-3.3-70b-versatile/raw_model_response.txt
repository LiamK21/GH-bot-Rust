<imports> 
use crate::metrics::MetricType;
use crate::error_recording;
use crate::Glean;
</imports>
<Rust>
#[test]
fn test_metric_error_recording() {
  let glean = Glean::new("my_app");
  let metric = glean.counter("my_metric");
  metric.record_error(crate::ErrorType::InvalidValue, 1);
  let errors = error_recording::get_errors(glean);
  assert!(errors.len() > 0);
}</Rust>
