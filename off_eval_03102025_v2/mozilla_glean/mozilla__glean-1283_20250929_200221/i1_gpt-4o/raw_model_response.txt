<imports>
use super::PingUploadManager;
use log::Level;
use log::Log;
use log::Metadata;
use log::Record;
use std::sync::Mutex;
</imports>
<Rust>
struct TestLogger {
    pub messages: Mutex<Vec<(Level, String)>>,
}

impl Log for TestLogger {
    fn enabled(&self, metadata: &Metadata) -> bool {
        metadata.level() <= Level::Info
    }

    fn log(&self, record: &Record) {
        if self.enabled(record.metadata()) {
            let mut messages = self.messages.lock().unwrap();
            messages.push((record.level(), record.args().to_string()));
        }
    }

    fn flush(&self) {}
}

#[test]
fn test_recoverable_failure_logging_level() {
    let test_logger = TestLogger {
        messages: Mutex::new(vec![]),
    };
    log::set_boxed_logger(Box::new(test_logger)).unwrap();
    log::set_max_level(log::LevelFilter::Info);

    let manager = PingUploadManager {};
    let document_id = "957090b3-4e26-4cfc-bfd8-00cbbac42524";
    let status = RecoverableFailure;

    manager.handle_upload_failure(document_id, status);

    let messages = test_logger.messages.lock().unwrap();
    let expected_message = format!(
        "Recoverable upload failure while attempting to send ping {}, will retry. Error was {:?}",
        document_id, status
    );

    assert!(messages.iter().any(|(level, message)| {
        *level == Level::Info && message == &expected_message
    }));
}
</Rust>
