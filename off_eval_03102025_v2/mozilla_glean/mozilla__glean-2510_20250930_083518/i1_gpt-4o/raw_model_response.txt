<imports>
use super::parse_language;
use anyhow::Result;
</imports>
<Rust>
#[test]
fn test_parse_language() {
    let result_kotlin = parse_language("kotlin");
    assert!(result_kotlin.is_ok());
    assert_eq!(result_kotlin.unwrap(), uniffi::TargetLanguage::Kotlin);

    let result_python = parse_language("python");
    assert!(result_python.is_ok());
    assert_eq!(result_python.unwrap(), uniffi::TargetLanguage::Python);

    let result_swift = parse_language("swift");
    assert!(result_swift.is_ok());
    assert_eq!(result_swift.unwrap(), uniffi::TargetLanguage::Swift);

    let result_unknown = parse_language("unknown");
    assert!(result_unknown.is_err());
}
</Rust>
