<imports> 
use super::system;
use glean_core::ClientInfoMetrics;
</imports>
<Rust>
#[test]
fn test_windows_build_number_included_in_client_info() {
  let client_info_metrics = ClientInfoMetrics {
    windows_build_number: system::get_windows_build_number(),
    ..Default::default()
  };
  let test_client_info = ClientInfoMetrics::from(client_info_metrics);
  assert!(test_client_info.windows_build_number.is_some());
}</Rust>
