<imports> 
use crate::error::ErrorKind;
use crate::upload::request::PingRequest;
use crate::upload::request::Builder;
</imports>
<Rust>
#[test]
fn test_ping_request_body_size_overflow() {
  let builder = Builder::new("Rust", 10);
  let ping_request = builder.document_id("id").path("/path").body("this is a long string that exceeds the max size of 10").build();
  assert!(ping_request.is_err());
  assert_eq!(ping_request.unwrap_err().kind(), &ErrorKind::PingBodyOverflow(10 + 41));
}
</Rust>
