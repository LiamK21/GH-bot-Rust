#[test]
fn test_ping_request_body_exceeds_max_size() {
    let request = Builder::new("Rust", 1)
        .document_id("test_doc")
        .path("/test/path")
        .body("test_body")
        .build();

    assert!(matches!(request, Err(e) if matches!(e.kind(), ErrorKind::PingBodyOverflow(_))));
}
