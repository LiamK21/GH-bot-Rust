#[test]
fn test_remove_ping_name_from_test_get_num_recorded_errors() {
  let meta = glean_core::metrics::common_metric_data::CommonMetricData {
    base_identifier: "metric1".to_string(),
    send_in_pings: vec!["metrics".to_string()],
  };
  let result = test_get_num_recorded_errors("glean", &meta, ErrorType::InvalidValue);
  assert_eq!(result.unwrap_or(0), 0);
}
