#[test]
fn test_create_configuration_with_ping_rate_limit() {
  let pings_per_interval = 10;
  let rate_limit = Some(PingRateLimit {
    seconds_per_interval: 1,
    pings_per_interval,
  });
  let cfg = ConfigurationBuilder::new()
    .rate_limit(rate_limit)
    .build();
  assert_eq!(cfg.rate_limit, rate_limit);
}
