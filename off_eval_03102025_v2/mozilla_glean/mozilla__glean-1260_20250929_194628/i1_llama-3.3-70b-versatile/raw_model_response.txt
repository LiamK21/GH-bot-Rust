<imports> 
use std::path::Path;
use rkv::StoreError;
use super::Database;
use std::fs;
use tempdir::TempDir;
</imports>
<Rust>
#[test]
fn test_database_open_rkv() {
  let temp_dir = TempDir::new("test_database").unwrap();
  let path = Path::new(temp_dir.path()).join("db");
  fs::create_dir(&path).unwrap();
  let db = Database { rkv: rkv::Rkv::new(&path).unwrap() };
  assert!(db.open_rkv(&path).is_ok());
}</Rust>
