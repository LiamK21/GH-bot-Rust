<imports> 
use env_logger::Builder;
use log::Level;
use std::env;
use super::glean_enable_logging;
</imports>
<Rust>
#[test]
fn test_glean_enable_logging_logging_initialized() {
  env::set_var("RUST_LOG", "debug");
  glean_enable_logging();
  let logger = Builder::from_env("RUST_LOG").build().unwrap();
  assert_eq!(logger.filter().to_level(), Level::Debug);
}</Rust>
