#[test]
fn test_object_metric_set_string() {
  let glean = Glean::new();
  let metric = ObjectMetric::new("test_metric", glean);
  let object = json!({"key": "value"});
  metric.set_sync(&glean, object.clone());
  let stored_value = glean.storage().get(&metric.meta);
  assert_eq!(stored_value.unwrap(), Metric::Object(serde_json::to_string(&object).unwrap()));
}
