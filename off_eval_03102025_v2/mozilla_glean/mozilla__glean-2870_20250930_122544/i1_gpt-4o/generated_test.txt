#[test]
fn test_rkv_new_initializes_database_without_migration() {
    let dir = tempdir().unwrap();
    let database_dir = dir.path().join("db");
    fs::create_dir_all(&database_dir).expect("create dir");

    let (rkv, load_state) = rkv_new(&database_dir).expect("rkv_new should succeed");

    assert!(rkv.is_some(), "Rkv should be initialized");
    assert!(load_state.is_some(), "RkvLoadState should be initialized");
}
