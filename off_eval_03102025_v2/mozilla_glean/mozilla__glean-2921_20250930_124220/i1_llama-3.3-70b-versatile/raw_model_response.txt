<imports> 
use glean_core::rlb::src::private::object::ObjectMetric;
use serde_json::json;
use std::collections::HashMap;
</imports>
<Rust>
#[test]
fn test_object_metric_null_values() {
  let metric: ObjectMetric<HashMap<String, String>> = ObjectMetric::new(Default::default());
  let arr_str = json!({
    "key": null,
    "key2": "value"
  })
  .to_string();
  metric.set_string(arr_str);
  let data = metric.test_get_value(None).expect("no object recorded");
  let expected = json!({
    "key2": "value"
  });
  assert_eq!(expected, data);
  assert_eq!(0, metric.test_get_num_recorded_errors(super::ErrorType::InvalidValue));
}
</Rust>
