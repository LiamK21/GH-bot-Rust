#[test]
fn test_get_pings_dir_deletion_request() {
    let ping_maker = PingMaker {};
    let data_path = Path::new("/test/path");
    let ping_type = Some("deletion-request");

    let expected = data_path.join("deletion-request-pings");
    let actual = ping_maker.get_pings_dir(data_path, ping_type).unwrap();

    assert_eq!(expected, actual);
}
