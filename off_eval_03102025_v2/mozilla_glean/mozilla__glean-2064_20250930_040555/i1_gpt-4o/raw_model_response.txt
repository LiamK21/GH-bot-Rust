<imports>
use super::glean_enable_logging_to_fd;
use std::sync::Once;
use log::{self, LevelFilter};
</imports>
<Rust>
#[test]
fn test_glean_enable_logging_to_fd() {
    static INIT: Once = Once::new();
    INIT.call_once(|| {
        let fd: u64 = 1; // Using stdout for testing purposes
        glean_enable_logging_to_fd(fd);
        assert_eq!(log::max_level(), LevelFilter::Debug);
    });
}
</Rust>
