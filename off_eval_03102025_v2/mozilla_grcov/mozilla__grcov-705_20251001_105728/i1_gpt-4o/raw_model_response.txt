<imports>
use super::make_context;
use tera::Context;
</imports>
<Rust>
#[test]
fn test_make_context_bulma_version() {
    std::env::set_var("BULMA_VERSION", "0.9.3");
    let context = make_context();
    let expected_version = "0.9.3";
    let actual_version: &str = context.get("bulma_version").unwrap();
    assert_eq!(expected_version, actual_version);
}
</Rust>
