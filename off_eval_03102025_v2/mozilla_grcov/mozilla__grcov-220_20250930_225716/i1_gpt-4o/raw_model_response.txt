<Rust>
#[test]
fn test_thread_naming() {
    let builder = thread::Builder::new().name(String::from("TestThread"));
    let handle = builder.spawn(|| {
        assert_eq!(thread::current().name(), Some("TestThread"));
    }).unwrap();
    handle.join().unwrap();
}
</Rust>
