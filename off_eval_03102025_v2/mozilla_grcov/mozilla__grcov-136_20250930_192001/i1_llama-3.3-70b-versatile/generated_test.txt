#[test]
fn test_parse_jacoco_xml_report_supports_parsing() {
  let file = File::open("./test/jacoco/basic-report.xml").expect("Failed to open xml file");
  let file_reader = BufReader::new(&file);
  let result = parse_jacoco_xml_report(file_reader);
  assert!(result.is_ok());
  let coverage_result = result.unwrap();
  assert!(!coverage_result.is_empty());
}
