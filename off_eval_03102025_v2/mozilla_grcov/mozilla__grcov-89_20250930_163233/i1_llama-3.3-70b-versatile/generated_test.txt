#[test]
fn test_rewrite_paths_with_wildcard_exclusion() {
  let mut result_map: HashMap<String, ()> = HashMap::new();
  result_map.insert("vs2017_15.6.6/main.cpp".to_string(), ());
  result_map.insert("vs2017_15.6.6/mydir/prova.h".to_string(), ());
  result_map.insert("vs2017_15.7.7/main.cpp".to_string(), ());
  let expected_count = 0;
  let actual_count = rewrite_paths(result_map, None, "", "", false, false, vec!["vs2017_*/".to_string()]).count();
  assert_eq!(expected_count, actual_count);
}
