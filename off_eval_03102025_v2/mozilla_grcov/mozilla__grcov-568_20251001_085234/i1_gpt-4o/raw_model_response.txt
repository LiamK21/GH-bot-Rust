<Rust>
#[test]
fn test_parse_version() {
    let gcov_output = "gcov (GCC) 4.9.0\n";
    let expected_version = Version {
        major: 4,
        minor: 9,
        patch: 0,
        pre: vec![],
        build: vec![],
    };
    let actual_version = parse_version(gcov_output);
    assert_eq!(expected_version, actual_version);
}
</Rust>
