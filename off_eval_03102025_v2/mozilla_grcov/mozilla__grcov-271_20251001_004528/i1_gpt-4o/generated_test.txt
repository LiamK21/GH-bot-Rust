#[test]
fn test_rewrite_paths_without_sorting() {
    let result_map = std::collections::HashMap::new();
    let path_mapping = None;
    let source_dir = None;
    let prefix_dir = None;
    let ignore_not_existing = false;
    let to_ignore_dirs = vec!["foo/*".to_string(), "bar/*".to_string()];
    let filter_option = None;

    let mut iter = rewrite_paths(
        result_map,
        path_mapping,
        source_dir,
        prefix_dir,
        ignore_not_existing,
        to_ignore_dirs,
        filter_option,
    );

    assert!(iter.next().is_none());
}
