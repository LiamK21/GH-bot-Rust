#[test]
fn test_profraws_to_lcov_parallelism() {
  let profraw_paths = vec![PathBuf::from("test.profraw")];
  let binary_path = PathBuf::from("/path/to/binary");
  let working_dir = PathBuf::from("/path/to/working/dir");
  let result = profraws_to_lcov(&profraw_paths, &binary_path, &working_dir);
  assert!(result.is_ok());
}
