<imports>
use super::parse_gcov;
use std::fs::File;
use std::io::Write;
use std::path::Path;
</imports>

<Rust>
#[test]
fn test_parse_gcov_handles_empty_file() {
    let path = Path::new("empty_gcov_file.txt");
    let mut file = File::create(&path).unwrap();
    file.write_all(b"").unwrap();

    let result = parse_gcov(&path);

    assert!(result.is_ok());
    assert_eq!(result.unwrap().len(), 0);

    std::fs::remove_file(&path).unwrap();
}
</Rust>
