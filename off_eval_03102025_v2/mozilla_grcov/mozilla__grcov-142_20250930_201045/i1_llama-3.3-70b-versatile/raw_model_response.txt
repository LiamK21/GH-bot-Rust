#[test]
fn test_producer_with_jacoco_xml() {
  let queue: Arc<WorkQueue> = Arc::new(WorkQueue::new());
  let tmp_dir = TempDir::new("grcov").expect("Failed to create temporary directory");
  let paths = vec!["test/jacoco1.zip".to_string()];
  producer(tmp_dir.path(), &paths, &queue, false, false);
  let expected = vec![
    (super::ItemFormat::JACOCO_XML, false, "jacoco/basic-jacoco.xml", true),
    (super::ItemFormat::JACOCO_XML, false, "inner-classes.xml", true),
  ];
  // assuming check_produced is defined elsewhere
  super::check_produced(tmp_dir.path().to_owned(), &queue, expected);
}</Rust>
