#[test]
fn test_unexpected_version_error() {
  let data = r#"
  <?xml version="1.0" encoding="UTF-8"?>
  <package name="test">
    <sourcefiles>
      <sourcefile name="test.java">
        <lines>
          <line number="1" ci="0" cb="0" mb="0" />
        </lines>
      </sourcefile>
    </sourcefiles>
  </package>
  "#;
  let reader = BufReader::new(data.as_bytes());
  let mut buf = Vec::new();
  let mut parser = Reader::from_reader(reader);
  let package = parse_jacoco_report_package(&mut parser, &mut buf, "test").unwrap();
  assert_eq!(package.len(), 1);
}
