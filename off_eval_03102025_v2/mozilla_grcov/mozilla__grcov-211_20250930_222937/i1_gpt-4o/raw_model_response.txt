<imports>
use super::rewrite_paths;
use std::collections::HashMap;
use std::path::PathBuf;
use serde_json::Value;
</imports>
<Rust>
#[test]
fn test_rewrite_paths_skips_symlinks() {
    let result_map = HashMap::new();
    let path_mapping = None;
    let source_dir = Some(PathBuf::from("test_data/source"));
    let prefix_dir = None;
    let ignore_not_existing = false;
    let to_ignore_dirs = vec![];
    let filter_option = None;

    let mut actual = rewrite_paths(
        result_map,
        path_mapping,
        source_dir,
        prefix_dir,
        ignore_not_existing,
        to_ignore_dirs,
        filter_option,
    );

    let expected: HashMap<String, String> = HashMap::new();
    assert_eq!(actual.next(), expected.into_iter().next());
}
</Rust>
