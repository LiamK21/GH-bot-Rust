#[test]
fn test_alterate_function_expression() {
  let node = Node { kind_id: Mozjs::FunctionExpression as i32 };
  let code = b"function foo() {}";
  let span = true;
  let children = vec![];
  let expected = AstNode::default();
  let actual = MozjsCode::alterate(&node, code, span, children);
  assert_eq!(actual, expected);
}
