<imports>
use super::compute_minmax;
use crate::State;
use crate::metrics::nom::Stats;
</imports>
<Rust>
#[test]
fn test_compute_minmax_updates_nom_minmax() {
    let mut state = State::default();
    state.space.metrics.nom = Stats {
        functions: 2,
        closures: 1,
        functions_sum: 0,
        closures_sum: 0,
        functions_min: usize::MAX,
        functions_max: 0,
        closures_min: usize::MAX,
        closures_max: 0,
    };

    compute_minmax(&mut state);

    assert_eq!(state.space.metrics.nom.functions_min(), 2.0);
    assert_eq!(state.space.metrics.nom.functions_max(), 2.0);
    assert_eq!(state.space.metrics.nom.closures_min(), 1.0);
    assert_eq!(state.space.metrics.nom.closures_max(), 1.0);
}
</Rust>
