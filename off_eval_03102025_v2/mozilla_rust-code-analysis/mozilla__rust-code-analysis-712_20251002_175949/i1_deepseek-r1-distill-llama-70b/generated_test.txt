#[test]
fn test_nexits_min_max() {
    let mut stats1 = Stats::default();
    stats1.exit = 5;
    let mut stats2 = Stats::default();
    stats2.exit = 3;

    stats1.merge(&stats2);
    stats1.compute_minmax();

    let expected_min = 3;
    let expected_max = 5;

    assert_eq!(stats1.exit_min(), expected_min as f64);
    assert_eq!(stats1.exit_max(), expected_max as f64);
}
