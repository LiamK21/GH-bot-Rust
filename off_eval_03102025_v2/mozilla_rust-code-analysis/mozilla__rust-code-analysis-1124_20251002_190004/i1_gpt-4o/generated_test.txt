#[test]
fn test_try_expression_as_exit() {
    let source_code = r#"
    fn example() -> Result<(), Error> {
        let value = some_function()?;
        Ok(())
    }
    "#;
    let node = parse_to_node(source_code, "function_item");
    let mut stats = Stats::default();
    RustCode::compute(&node, &mut stats);
    assert_eq!(stats.exit, 2);
}
