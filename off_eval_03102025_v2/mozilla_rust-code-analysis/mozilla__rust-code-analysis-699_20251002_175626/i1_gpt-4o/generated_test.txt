#[test]
fn test_cyclomatic_min_max() {
    let mut stats = Stats::default();
    stats.cyclomatic = 3.0;
    stats.compute_minmax();

    let expected_min = 1.0;
    let expected_max = 3.0;

    assert_eq!(stats.cyclomatic_min(), expected_min);
    assert_eq!(stats.cyclomatic_max(), expected_max);
}
