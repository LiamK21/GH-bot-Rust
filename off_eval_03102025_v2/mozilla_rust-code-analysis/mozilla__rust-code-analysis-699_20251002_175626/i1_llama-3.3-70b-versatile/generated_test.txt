#[test]
fn test_cyclomatic_min_max() {
  let mut stats = Stats::default();
  stats.cyclomatic = 5.0;
  stats.compute_minmax();
  assert_eq!(stats.cyclomatic_min(), 5.0);
  assert_eq!(stats.cyclomatic_max(), 5.0);
  let mut other_stats = Stats::default();
  other_stats.cyclomatic = 3.0;
  stats.merge(&other_stats);
  assert_eq!(stats.cyclomatic_min(), 3.0);
  assert_eq!(stats.cyclomatic_max(), 5.0);
}
