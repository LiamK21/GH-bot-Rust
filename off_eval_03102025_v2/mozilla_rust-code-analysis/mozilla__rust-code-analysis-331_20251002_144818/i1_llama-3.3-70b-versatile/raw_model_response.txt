#[test]
fn test_rust_lloc_computation() {
  let code = "let foo = Foo { 42 }; foo.field;";
  let expected_lloc = 1;
  let actual_lloc = check_metrics!(code, "foo.rs", RustParser, loc, [(lloc, 1, usize)]);
  assert_eq!(expected_lloc, actual_lloc[0].1);
}
