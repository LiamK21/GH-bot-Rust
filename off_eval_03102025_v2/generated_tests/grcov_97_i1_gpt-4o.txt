#[test]
fn test_rewrite_paths_ignore_completely_uncovered_files() {
    let mut result_map: HashMap<String, CovResult> = HashMap::new();
    result_map.insert("uncovered.cpp".to_string(), CovResult {
        lines: [(1, 0), (2, 0)].iter().cloned().collect(),
        branches: [].iter().cloned().collect(),
        functions: HashMap::new(),
    });
    let results = rewrite_paths(result_map, None, "", "", false, false, Vec::new(), Some(true));
    let mut count = 0;
    for _ in results {
        count += 1;
    }
    assert_eq!(count, 0);
}
