<imports>
use super::output_html;
use std::path::Path;
use std::sync::{Arc, Mutex};
use crate::html::{HtmlGlobalStats, HtmlJobReceiver};
use crate::html::HtmlStats;
use crate::html::HtmlFileStats;
use crate::html::HtmlDirStats;
use std::collections::BTreeMap;
</imports>
<Rust>
#[test]
fn test_output_html_with_abs_link_prefix() {
    let results = vec![];
    let output_dir = Some(Path::new("output"));
    let num_threads = 1;
    let branch_enabled = false;
    let output_config_file = None;
    let precision = 2;
    let abs_link_prefix = Some(String::from("https://example.com/coverage/"));
    let no_date = false;

    let global_stats = HtmlGlobalStats {
        abs_prefix: abs_link_prefix.clone().map(PathBuf::from),
        ..Default::default()
    };
    let stats = Arc::new(Mutex::new(global_stats));

    output_html(
        &results,
        output_dir,
        num_threads,
        branch_enabled,
        output_config_file,
        precision,
        &abs_link_prefix,
        no_date,
    );

    let locked_stats = stats.lock().unwrap();
    assert_eq!(
        locked_stats.abs_prefix,
        Some(PathBuf::from("https://example.com/coverage/"))
    );
}
</Rust>
