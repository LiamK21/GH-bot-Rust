<imports>
use super::main;
use std::process::Command;
use std::path::Path;
</imports>
<Rust>
#[test]
fn test_output_path_option_with_default_output_type() {
  let result_before_patch = Command::new("grcov").arg("target/coverage").arg("--binary-path").arg("target/debug").arg("-o").arg("lcov.info").arg("--output-type").arg("lcov").output().unwrap();
  assert_ne!(result_before_patch.status.code(), Some(0));

  let result_after_patch = Command::new("grcov").arg("target/coverage").arg("--binary-path").arg("target/debug").arg("-o").arg("lcov.info").output().unwrap();
  assert_eq!(result_after_patch.status.code(), Some(0));
  assert!(Path::new("lcov.info").exists());
}</Rust>
