#[test]
fn test_output_path_creation() {
    // Setup: Clean up any existing lcov.info file
    let _ = fs::remove_file("lcov.info");

    // Execute the command that should create the lcov.info file
    let output = Command::new("grcov")
        .args(&["target/coverage", "--binary-path", "target/debug/", "--source-dir", ".", "-o", "lcov.info"])
        .output()
        .expect("Failed to execute grcov command");

    // Check that the command executed successfully
    assert!(output.status.success());

    // Verify that the lcov.info file was created
    assert!(fs::metadata("lcov.info").is_ok());

    // Cleanup: Remove the lcov.info file after the test
    let _ = fs::remove_file("lcov.info");
}
