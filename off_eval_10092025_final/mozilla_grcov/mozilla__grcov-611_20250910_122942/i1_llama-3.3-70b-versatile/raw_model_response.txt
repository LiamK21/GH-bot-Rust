<imports> 
use super::add_results;
use log::Level;
use std::io::Cursor;
use std::collections::Vec;
</imports>
<Rust>
#[test]
fn test_execution_count_overflow_logging() {
  let mut results: Vec<(String, ())> = Vec::new();
  let warn_overflow = true;
  let handler = |record: &log::Record| {
    assert_eq!(record.level(), Level::Warn);
    assert_eq!(record.args(), &format_args!("Execution count overflow detected."));
  };
  let _ = log::set_max_level(log::LevelFilter::Trace);
  log::set_logger(&handler).unwrap();
  add_results(results, warn_overflow);
}</Rust>
