#[test]
fn test_known_binary_detection() {
  let temp_file = PathBuf::from("temp_file");
  let mut file = File::create(temp_file.clone()).unwrap();
  file.write_all(&[0x7F, b'E', b'L', b'F']).unwrap();
  file.sync_all().unwrap();
  let binaries = find_binaries(&temp_file);
  assert!(binaries.contains(&temp_file));
  std::fs::remove_file(temp_file).unwrap();
}
