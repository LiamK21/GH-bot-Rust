<imports> 
use super::get_coverage;
use super::CovResultIter;
use super::DemangleOptions;
</imports>
<Rust>
#[test]
fn test_cobertura_class_line_coverage() {
  let results: CovResultIter = vec![].into_iter();
  let sources: Vec<String> = vec!["src/test.rs".to_string()];
  let demangle: bool = true;
  let demangle_options: DemangleOptions = DemangleOptions::default();
  let coverage_before = get_coverage(results.clone(), sources.clone(), demangle, demangle_options);
  let coverage_after = get_coverage(results, sources, demangle, demangle_options);
  assert_ne!(coverage_before, coverage_after);
}</Rust>
