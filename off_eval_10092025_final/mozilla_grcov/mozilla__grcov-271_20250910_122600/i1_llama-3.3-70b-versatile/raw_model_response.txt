<imports> 
use super::rewrite_paths;
use std::path::PathBuf;
use std::collections::HashMap;
</imports>
<Rust>
#[test]
fn test_rewrite_paths_without_sorting() {
  let result_map: HashMap<String, String> = HashMap::new();
  let path_mapping: Option<String> = None;
  let source_dir: Option<PathBuf> = None;
  let prefix_dir: Option<PathBuf> = None;
  let ignore_not_existing: bool = false;
  let mut to_ignore_dirs: Vec<String> = vec!["foo/bar".to_string(), "foo/aaa".to_string()];
  let filter_option: Option<bool> = None;
  let expected_to_ignore_dirs: Vec<String> = vec!["foo/aaa".to_string(), "foo/bar".to_string()];
  
  let actual = rewrite_paths(result_map, path_mapping, source_dir, prefix_dir, ignore_not_existing, to_ignore_dirs.clone(), filter_option).collect::<Vec<_>>();
  let expected = rewrite_paths(result_map, path_mapping, source_dir, prefix_dir, ignore_not_existing, expected_to_ignore_dirs, filter_option).collect::<Vec<_>>();
  
  assert_ne!(actual, expected);
}</Rust>
