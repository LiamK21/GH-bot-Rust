#[test]
fn test_jacoco_xml_parsing() {
  let file = File::open("./test/jacoco/basic-report.xml").expect("Failed to open xml file");
  let file_before = File::open("./test/jacoco/basic-report.xml").expect("Failed to open xml file");
  let reader_before = BufReader::new(&file_before);
  let result_before = parse_jacoco_xml_report(reader_before);
  assert!(result_before.is_err());

  let reader = BufReader::new(&file);
  let result = parse_jacoco_xml_report(reader).unwrap();
  assert_eq!(result.len(), 1);
}
