#[test]
fn test_rewrite_paths_with_to_ignore_dirs_is_mutable() {
  let mut result_map: CovResultMap = FxHashMap::default();
  result_map.insert("main.cpp".to_string(), empty_result!());
  let mut to_ignore_dirs = vec!["mydir/*"];
  let results = rewrite_paths(result_map, None, None, None, false, &mut to_ignore_dirs, None);
  let mut count = 0;
  for (abs_path, rel_path, result) in results {
    count += 1;
  }
  assert_eq!(count, 1);
}
