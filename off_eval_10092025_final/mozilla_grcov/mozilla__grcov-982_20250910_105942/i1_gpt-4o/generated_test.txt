#[test]
fn test_output_html_creates_directory() {
    let output_path = Path::new("coverage/subpath");
    if output_path.exists() {
        fs::remove_dir_all(output_path).unwrap();
    }

    output_html(&[], Some(output_path), 1, false, None);

    assert!(output_path.exists());

    fs::remove_dir_all("coverage").unwrap();
}
