#[test]
fn test_guess_abs_path_subfolder_same_as_root() {
  let prefix_dir = PathBuf::from("memflow");
  let path = PathBuf::from("memflow/src/main.rs");
  let expected = PathBuf::from("memflow/memflow/src/main.rs");
  let actual = guess_abs_path(&prefix_dir, &path);
  assert_eq!(actual, expected);
}
