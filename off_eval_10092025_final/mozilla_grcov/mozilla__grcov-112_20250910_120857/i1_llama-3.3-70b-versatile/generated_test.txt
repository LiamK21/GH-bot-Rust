#[test]
fn test_output_activedata_etl_format() {
  let mut cursor = Cursor::new(Vec::new());
  output_activedata_etl(vec![].into_iter());
  let actual = String::from_utf8(cursor.into_inner()).unwrap();
  let expected = "{\"is_file\":true,\"file\":{\"name\":\"\",\"covered\":[],\"uncovered\":[],\"total_covered\":0,\"total_uncovered\":0,\"percentage_covered\":0.0},\"method\":{\"covered\":[],\"uncovered\":[],\"total_covered\":0,\"total_uncovered\":0}}";
  assert_eq!(actual, expected);
}
