use std::io::{self, Write};

use std::collections::HashSet;

<Rust>
#[test]
fn test_output_activedata_etl_removes_line_key() {
    let results = vec![
        // Mock data for CovResultIter
    ];
    let mut buffer = Vec::new();
    {
        let stdout = io::stdout();
        let _lock = stdout.lock();
        let mut writer = io::BufWriter::new(&mut buffer);
        output_activedata_etl(results.into_iter());
    }
    let output = String::from_utf8(buffer).unwrap();
    let json_output: Value = serde_json::from_str(&output).unwrap();
    
    // Check that the "line" key is not present in the output
    assert!(json_output["file"]["covered"].as_array().unwrap().iter().all(|v| v.is_u64()));
    assert!(json_output["method"]["covered"].as_array().unwrap().iter().all(|v| v.is_u64()));
}
</Rust>
