#[test]
fn test_branch_coverage_in_html_report() {
  let results = CovResultIter::empty();
  let output_dir = Some("target/html");
  let num_threads = 1;
  let branch_enabled = true;
  let config = get_config();
  output_html(results, output_dir, num_threads, branch_enabled);
  // Check the generated HTML report for branch coverage information
  let output_file = PathBuf::from("target/html/index.html");
  assert!(output_file.exists());
  let html_content = std::fs::read_to_string(output_file).unwrap();
  assert!(html_content.contains("Branches"));
}
