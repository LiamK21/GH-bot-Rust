#[test]
fn test_make_context_returns_valid_context() {
    let context = make_context();
    let expected_bulma_version = env!("BULMA_VERSION").as_static_str();
    let actual_bulma_version = context.get::<String>("bulma_version").unwrap();

    assert_eq!(expected_bulma_version, actual_bulma_version);
}
