<imports>
use std::thread;
</imports>
<Rust>
#[test]
fn test_thread_naming() {
    let producer_thread = thread::Builder::new().name(String::from("Producer")).spawn(|| {}).unwrap();
    assert_eq!(producer_thread.thread().name(), Some("Producer"));

    let consumer_thread = thread::Builder::new().name(String::from("Consumer 1")).spawn(|| {}).unwrap();
    assert_eq!(consumer_thread.thread().name(), Some("Consumer 1"));
}
</Rust>
