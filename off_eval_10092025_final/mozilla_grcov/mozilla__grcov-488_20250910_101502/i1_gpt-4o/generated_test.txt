#[test]
fn test_handle_file_with_none() {
    let archive = Archive {
        name: String::from("test"),
        archive_type: ArchiveType::Plain(vec![]),
    };
    let path = PathBuf::from("non_existent_file");
    let gcno_stem_archives = RefCell::new(HashMap::new());
    let gcda_stem_archives = RefCell::new(HashMap::new());

    let result = std::panic::catch_unwind(|| {
        archive.handle_file(
            None,
            &path,
            &gcno_stem_archives,
            &gcda_stem_archives,
        );
    });

    assert!(result.is_ok(), "handle_file should handle None without panicking");
}
