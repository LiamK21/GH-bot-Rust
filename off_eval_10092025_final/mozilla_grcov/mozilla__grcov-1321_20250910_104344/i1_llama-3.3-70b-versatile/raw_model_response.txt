<imports> 
use super::rewrite_paths;
use std::path::Path;
use std::collections::HashMap;
</imports>
<Rust>
#[test]
fn test_rewrite_paths_with_partial_path_mapping() {
  let result_map: HashMap<String, String> = HashMap::new();
  let path_mapping = None;
  let source_dir = Some(Path::new("/path/to/source"));
  let prefix_dir = None;
  let ignore_not_existing = false;
  let to_ignore_dirs: &[&str] = &[];
  let to_keep_dirs: &[&str] = &[];
  let filter_option = None;
  let file_filter = crate::FileFilter::default();

  let result = rewrite_paths(result_map, path_mapping, source_dir, prefix_dir, ignore_not_existing, to_ignore_dirs, to_keep_dirs, filter_option, file_filter);
  assert!(result.len() == 0);
}</Rust>
