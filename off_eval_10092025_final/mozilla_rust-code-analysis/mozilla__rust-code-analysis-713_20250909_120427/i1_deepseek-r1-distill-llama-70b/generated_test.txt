#[test]
fn test_cognitive_minmax() {
    let mut stats = Stats::default();
    stats.structural = 5;
    stats.compute_minmax();

    let mut other = Stats::default();
    other.structural = 3;
    stats.merge(&other);
    stats.compute_minmax();

    assert_eq!(stats.cognitive_min(), 0.0);
    assert_eq!(stats.cognitive_max(), 5.0);
}
