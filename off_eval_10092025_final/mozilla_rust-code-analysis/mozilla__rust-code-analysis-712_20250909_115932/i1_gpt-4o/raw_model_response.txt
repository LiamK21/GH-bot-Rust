<imports>
use super::compute_minmax;
use crate::State;
use crate::metrics::exit::Stats as ExitStats;
use crate::metrics::cyclomatic::Stats as CyclomaticStats;
</imports>
<Rust>
#[test]
fn test_compute_minmax_updates_minmax_values() {
    let mut state = State {
        space: Default::default(),
    };

    state.space.metrics.cyclomatic = CyclomaticStats {
        cyclomatic_sum: 5,
        cyclomatic_min: 2,
        cyclomatic_max: 3,
    };

    state.space.metrics.nexits = ExitStats {
        exit: 1,
        exit_sum: 5,
        exit_min: 2,
        exit_max: 3,
        total_space_functions: 1,
    };

    compute_minmax(&mut state);

    assert_eq!(state.space.metrics.cyclomatic.cyclomatic_min, 2);
    assert_eq!(state.space.metrics.cyclomatic.cyclomatic_max, 3);
    assert_eq!(state.space.metrics.nexits.exit_min, 1);
    assert_eq!(state.space.metrics.nexits.exit_max, 3);
}
</Rust>
