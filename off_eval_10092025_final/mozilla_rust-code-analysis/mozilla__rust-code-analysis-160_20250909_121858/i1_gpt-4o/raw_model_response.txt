<imports>
use super::dump_formats;
use super::Format;
use super::FuncSpace;
use std::path::PathBuf;
</imports>
<Rust>
#[test]
fn test_dump_json_to_stdout() {
    let space = FuncSpace {
        name: Some("test_function"),
        start_line: 1,
        end_line: 10,
        kind: super::SpaceKind::Function,
        spaces: vec![],
        metrics: super::CodeMetrics::default(),
    };
    let path = PathBuf::from("test.rs");
    let output_format = Format::Json;
    let output_path = None;
    let pretty = false;

    let result = dump_formats(&space, &path, &output_path, output_format, pretty);
    assert!(result.is_ok());
}
</Rust>
