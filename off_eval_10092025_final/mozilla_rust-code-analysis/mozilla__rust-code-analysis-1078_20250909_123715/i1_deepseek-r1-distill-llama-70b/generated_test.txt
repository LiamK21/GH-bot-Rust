#[test]
fn test_qualified_function_name() {
    check_func_space::<CppParser, _>(
        "void Foo::bar(){
            return;
        }",
        "foo.c",
        |func_space| {
            insta::assert_json_snapshot!(
                func_space.spaces[0].name,
                @r###""Foo::bar""###
            );
        },
    );
}
