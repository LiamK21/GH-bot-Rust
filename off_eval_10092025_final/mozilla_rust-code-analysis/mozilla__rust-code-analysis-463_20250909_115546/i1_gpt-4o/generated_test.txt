#[test]
fn test_javascript_language_binding() {
    enum LANG {
        Javascript,
    }

    mk_get_language!(LANG);

    let expected: Language = tree_sitter_javascript::language();
    let actual: Language = match LANG::Javascript {
        LANG::Javascript => tree_sitter_javascript::language(),
    };

    assert_eq!(expected, actual);
}
