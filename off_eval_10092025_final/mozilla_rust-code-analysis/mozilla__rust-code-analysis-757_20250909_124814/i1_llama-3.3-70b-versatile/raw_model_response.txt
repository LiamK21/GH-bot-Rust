<imports> 
use super::Cpp;
use std::collections::HashSet;
</imports>
<Rust>
#[test]
fn test_cpp_enum_values() {
  let enum_values: HashSet<i32> = (154..491).collect();
  let cpp_enum_values: HashSet<i32> = Cpp::Catch as i32..Cpp::Error as i32 + 1
    .filter_map(|value| match value {
      157 => Some(value), // OperatorName is 445 in the patch
      158 => None, // This is 161 in the patch
      159 => None, // Nullptr is 162 in the patch
      160 => None, // LiteralSuffix is 163 in the patch
      _ => Some(value),
    })
    .collect();
  let difference: Vec<i32> = enum_values.symmetric_difference(&cpp_enum_values).collect();
  assert!(difference.is_empty());
}</Rust>
