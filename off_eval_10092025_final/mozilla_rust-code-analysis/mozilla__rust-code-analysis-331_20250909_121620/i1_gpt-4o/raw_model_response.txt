use super::Loc;

<Rust>
#[test]
fn test_rust_expression_lloc() {
    let code = "
        struct Foo {
            field: usize,
        }
        let foo = Foo { field: 42 };
        foo.field;
        let a = (42 + 0);
        let b = [0; 42];
        let c = (0, 42);
        let d = ();
        let e = foo();
        foo();
        k!(foo());
    ";
    let expected_lloc = 3;
    let mut stats = RustCode::default();
    RustParser::parse(&mut stats, code);
    assert_eq!(stats.logical_lines, expected_lloc);
}
</Rust>
