#[test]
fn test_dump_formats() {
  let language = rust_code_analysis::LANG::Cpp;
  let source_code = "int a = 42;\n int b = 43;".as_bytes().to_vec();
  let path = PathBuf::from("foo.c");
  let space = get_function_spaces(&language, source_code, &path, None).unwrap();
  let output_path = None;
  let pretty = false;
  let format = Format::Json;
  format.dump_formats(&space, &path, &output_path, pretty).unwrap();
}
