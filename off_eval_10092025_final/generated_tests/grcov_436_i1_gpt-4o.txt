#[test]
#[should_panic(expected = "The output file ./target/debug/coverage/ is a directory, but must be a regular file.")]
fn test_output_file_is_directory() {
    let output_path = "./target/debug/coverage/";
    if !Path::new(output_path).exists() {
        std::fs::create_dir_all(output_path).unwrap();
    }
    let _ = get_target_output_writable(Some(output_path));
}
