#[test]
fn test_deletion_request_pings_are_processed() {
  let temp_dir = TempDir::new().unwrap();
  let ping_directory_manager = PingDirectoryManager::new(temp_dir.path());
  let deletion_request_dir = temp_dir.path().join("deletion_request");
  std::fs::create_dir_all(&deletion_request_dir).unwrap();
  let file_path = deletion_request_dir.join(Uuid::new_v4().to_string());
  std::fs::File::create(&file_path).unwrap();
  let requests = ping_directory_manager.process_dir();
  assert_eq!(requests.len(), 1);
}
