<imports> 
use super::glean_labeled_counter_metric_get;
use std::ffi::CString;
use std::ptr;
</imports>
<Rust>
#[test]
fn test_labeled_metric_get() {
  let handle: u64 = 123;
  let label = CString::new("test_label").unwrap();
  let label_ptr = label.as_ptr();
  let result_before_patch = unsafe { glean_labeled_counter_metric_get(handle, label_ptr) };
  let result_after_patch = unsafe { glean_labeled_counter_metric_get(handle, label_ptr) };
  assert_ne!(result_before_patch, 0);
  assert_eq!(result_before_patch, result_after_patch);
}</Rust>
