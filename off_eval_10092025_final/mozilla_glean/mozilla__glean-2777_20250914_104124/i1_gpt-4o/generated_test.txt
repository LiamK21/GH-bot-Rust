#[test]
fn test_timer_id_starts_at_one() {
    let metric = TimingDistributionMetric {
        meta: Arc::new(Default::default()),
        time_unit: Default::default(),
        next_id: Arc::new(AtomicUsize::new(1)),
        start_times: Arc::new(Mutex::new(Default::default())),
    };

    let initial_id = metric.next_id.load(std::sync::atomic::Ordering::SeqCst);
    assert_eq!(initial_id, 1, "Timer ID should start at 1");
}
