struct TestLogger {
    pub messages: Mutex<Vec<String>>,
}

impl Log for TestLogger {
    fn enabled(&self, metadata: &Metadata) -> bool {
        metadata.level() <= Level::Error
    }

    fn log(&self, record: &Record) {
        if self.enabled(record.metadata()) {
            let mut messages = self.messages.lock().unwrap();
            messages.push(format!("{}", record.args()));
        }
    }

    fn flush(&self) {}
}

#[test]
fn test_log_if_error_logs_error() {
    let test_logger = TestLogger {
        messages: Mutex::new(vec![]),
    };
    log::set_boxed_logger(Box::new(test_logger)).unwrap();
    log::set_max_level(Level::Error.to_level_filter());

    let error = ExternError::new_error(1, "Test error".into());
    log_if_error(error);

    let messages = log::logger().downcast_ref::<TestLogger>().unwrap().messages.lock().unwrap();
    assert!(messages.iter().any(|msg| msg.contains("Glean failed (Error(1)): Test error")));
}
