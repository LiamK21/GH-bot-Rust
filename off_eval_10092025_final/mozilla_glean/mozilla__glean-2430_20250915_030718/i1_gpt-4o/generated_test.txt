#[test]
fn test_memory_distribution_no_storage() {
    let glean = Glean::new("test_app_id", true, None, None, None).unwrap();
    let metric = MemoryDistributionMetric::new("category", "name", "test");

    // Simulate the condition where the database is not available
    glean.reset_glean(true);

    // Attempt to record a memory distribution
    metric.accumulate(&glean, MAX_BYTES);

    // Since the storage is not available, we expect no panic and no data recorded
    assert!(true); // If no panic occurs, the test passes
}
