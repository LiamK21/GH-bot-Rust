#[test]
fn test_initialize_for_subprocess() {
    let config = FfiConfiguration {
        data_path: FfiStr::from("test_path"),
        application_id: FfiStr::from("test_app"),
        upload_enabled: true,
        max_events: Some(100),
    };
    let result = unsafe { glean_initialize_for_subprocess(&config as *const _) };
    assert_eq!(result, 1);
}
