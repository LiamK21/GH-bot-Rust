<imports>
use super::EventMetric;
use glean_core::metrics::event::EventMetric;
use glean_core::metrics::Configuration;
use glean_core::metrics::ConfigurationBuilder;
use tempfile::TempDir;
</imports>
<Rust>
#[test]
fn test_event_timestamp_removal() {
  let tmp_dir = TempDir::new().unwrap();
  let configuration = ConfigurationBuilder::new(true, tmp_dir.path(), "test_app").build();
  let glean = new_glean(Some(configuration), true);
  let event = EventMetric::new("test_event", "test_category", "test_name");
  event.set("test_value");
  let events = event.test_get_value("test_ping");
  let expected: Option<Vec<_>> = None;
  assert_eq!(events, expected);
}
</Rust>
