<imports> 
use super::glean_enable_logging;
use log::LevelFilter;
use oslog::OsLogger;
</imports>
<Rust>
#[test]
fn test_glean_enable_logging() {
  let level = LevelFilter::Info;
  let logger = OsLogger::new("org.mozilla.glean", "glean_core").level_filter(level);
  glean_enable_logging();
  let result = logger.init();
  assert!(result.is_ok());
}</Rust>
