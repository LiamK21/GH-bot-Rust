<imports>
use super::test_get_value;
use serde_json::json;
use crate::common_test::{lock_test, new_glean};
use crate::CommonMetricData;
</imports>
<Rust>
#[test]
fn test_get_value_returns_json() {
    let _lock = lock_test();
    let (glean, _t) = new_glean(None, true);

    let metric = ObjectMetric::<serde_json::Value>::new(
        CommonMetricData {
            name: "test_metric".into(),
            category: "test".into(),
            send_in_pings: vec!["store1".into()],
            ..Default::default()
        },
        glean,
    );

    let arr = json!([1, 2, 3]);
    metric.set(arr.clone());

    let data = metric.test_get_value(None).expect("no object recorded");
    assert_eq!(arr, data);
}
</Rust>
