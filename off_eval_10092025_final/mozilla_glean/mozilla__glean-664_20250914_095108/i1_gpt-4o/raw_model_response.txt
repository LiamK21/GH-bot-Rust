#[test]
fn test_glean_initialize_and_upload_enabled() {
    unsafe {
        let config = ptr::null();
        let result = glean_initialize(config);
        assert_eq!(result, 0);

        let upload_enabled = glean_is_upload_enabled();
        assert_eq!(upload_enabled, 0);
    }
}
