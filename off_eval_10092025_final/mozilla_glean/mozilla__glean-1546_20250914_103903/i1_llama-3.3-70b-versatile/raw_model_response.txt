<imports> 
use super::get_timestamp_ms;
use std::thread;
use std::time::Duration;
</imports>
<Rust>
#[test]
fn test_get_timestamp_ms_monotonic() {
  let start = get_timestamp_ms();
  thread::sleep(Duration::from_millis(1));
  let end = get_timestamp_ms();
  assert!(end >= start);
}</Rust>
