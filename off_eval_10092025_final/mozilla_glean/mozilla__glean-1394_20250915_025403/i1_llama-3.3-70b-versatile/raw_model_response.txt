<imports> 
use super::CustomDistributionMetric;
use glean_core::metrics::{DistributionData, HistogramType};
use glean_core::{CommonMetricData, ErrorType};
</imports>
<Rust>
#[test]
fn test_custom_distribution_accumulate_samples_signed() {
  let meta = CommonMetricData {
    name: "custom_distribution_test".to_string(),
    ..Default::default()
  };
  let metric = CustomDistributionMetric::new(
    meta.clone(),
    0,
    100,
    10,
    HistogramType::Custom,
  );
  let samples = vec![10, 20, -30, 40];
  metric.accumulate_samples_signed(samples);
  let expected_errors = 1;
  let actual_errors = metric.test_get_num_recorded_errors(ErrorType::InvalidValue, None);
  assert_eq!(actual_errors, expected_errors);
}
</Rust>
