<imports> 
use super::PingType;
use glean_core::metrics::PingType as InnerPingType;

<Rust>
#[test]
fn test_ping_metadata_access() {
  let ping_type = PingType::new("test_ping", true, true, true, true, true);
  let name = ping_type.name();
  let includes_client_id = ping_type.include_client_id();
  let send_if_empty = ping_type.send_if_empty();
  let precise_timestamps = ping_type.precise_timestamps();
  let include_info_sections = ping_type.include_info_sections();
  let naively_enabled = ping_type.naively_enabled();
  let follows_collection_enabled = ping_type.follows_collection_enabled();
  let schedules_pings = ping_type.schedules_pings();
  let reason_codes = ping_type.reason_codes();

  assert_eq!(name, "test_ping");
  assert_eq!(includes_client_id, true);
  assert_eq!(send_if_empty, true);
  assert_eq!(precise_timestamps, true);
  assert_eq!(include_info_sections, true);
  assert_eq!(naively_enabled, true);
  assert_eq!(follows_collection_enabled, true);
  assert_eq!(schedules_pings.len(), 0);
  assert_eq!(reason_codes.len(), 0);
}</Rust>
