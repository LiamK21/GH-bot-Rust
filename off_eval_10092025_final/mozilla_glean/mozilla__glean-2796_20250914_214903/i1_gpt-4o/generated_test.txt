#[test]
fn test_set_string_records_valid_json() {
    let glean = Glean::new();
    let metric = ObjectMetric::new();
    let json_string = json!({"key": "value"}).to_string();

    metric.set_string(json_string.clone());

    let recorded_value = glean.storage().get(&metric.meta).unwrap();
    assert_eq!(recorded_value, json_string);
}
