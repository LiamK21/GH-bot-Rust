#[test]
fn test_ping_rate_limit_reexport() {
    let mut cfg = configure_ping_throttling();
    let pings_per_interval = 10;
    cfg.rate_limit = Some(PingRateLimit {
        seconds_per_interval: 1,
        pings_per_interval,
    });

    assert_eq!(cfg.rate_limit.unwrap().pings_per_interval, pings_per_interval);
}
