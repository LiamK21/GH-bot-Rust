#[test]
fn test_recoverable_failure_logging() {
  let document_id = "957090b3-4e26-4cfc-bfd8-00cbbac42524";
  let status = "RecoverableFailure";
  let expected_log_level = Level::Info;
  let actual_log_level = log::max_level();
  log::set_max_level(log::LevelFilter::Info);
  PingUploadManager::upload(document_id, status);
  assert_eq!(actual_log_level, expected_log_level);
}
