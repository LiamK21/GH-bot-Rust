#[test]
fn test_event_metric_type() {
  let event = glean_parser.yaml_event("autotrack.fafo", "event");
  let generated_rust = rust.generate_rust(event);
  let expected_type = "NoExtraKeys";
  let actual_type = generated_rust.split("EventMetric<").nth(1).unwrap().split(">[].*").nth(0).unwrap();
  assert_eq!(actual_type, expected_type);
}
