#[test]
fn test_ping_submission_on_disabled_ping_schedule() {
  let mut ping = PingType::new("test_ping", true, true, true, true, true, vec![], vec![], false, vec![]);
  let mut glean = Glean::new();
  glean.ping_schedule = HashMap::from([("test_ping".to_string(), vec![" ride_along".to_string()])]);
  ping.set_enabled(false);
  glean.submit_ping(&ping, None);
  assert!(glean.ping_schedule.get("test_ping").unwrap().len() > 0);
}
