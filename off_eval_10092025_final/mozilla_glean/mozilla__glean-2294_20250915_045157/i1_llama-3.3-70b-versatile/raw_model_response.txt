<imports> 
use std::path::Path;
use super::destroy_glean;
use tempfile::tempdir;
use std::fs;
</imports>
<Rust>
#[test]
fn test_destroy_glean_clears_stores() {
  let dir = tempdir().unwrap();
  let tmpname = dir.path().to_path_buf();
  fs::create_dir_all(tmpname.join("store")).unwrap();
  destroy_glean(true, &tmpname);
  assert!(!tmpname.join("store").exists());
}
</Rust>
