#[test]
fn test_iter_over_all_error_types() {
    let expected = vec![
        ErrorType::InvalidValue,
        ErrorType::InvalidLabel,
        ErrorType::InvalidState,
        ErrorType::InvalidOverflow,
    ];
    let actual = ErrorType::iter().collect::<Vec<_>>();
    assert_eq!(expected, actual);
}
