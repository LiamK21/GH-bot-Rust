<imports>
use super::matches_label_regex;
</imports>
<Rust>
#[test]
fn test_matches_label_regex() {
    let valid_labels = vec![
        "valid_label",
        "valid.label",
        "valid_label_123",
        "a.b.c",
        "a_b-c.d_e-f",
    ];
    let invalid_labels = vec![
        "1invalid",
        "$invalid",
        "invalid$",
        "invalid..label",
        "toolongtoolongtoolongtoolongtoolongtoolongtoolong",
    ];

    for label in valid_labels {
        assert!(matches_label_regex(label), "Expected '{}' to be valid", label);
    }

    for label in invalid_labels {
        assert!(!matches_label_regex(label), "Expected '{}' to be invalid", label);
    }
}
</Rust>
