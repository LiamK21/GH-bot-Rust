#[test]
fn test_destroy_glean_before_init() {
  let cfg = super::configuration::Configuration {
    // initialize with default configuration
    ..Default::default()
  };
  let client_info = super::configuration::ClientInfoMetrics {
    // initialize with default client info
    ..Default::default()
  };
  let handle = thread::spawn(move || {
    initialize(cfg, client_info);
  });
  destroy_glean(true);
  handle.join().unwrap();
  // If no panic occurred, the test passes
  assert!(true);
}
