<imports> 
use super::initialize;
use super::register_ping_type;
use crate::private::PingType;
</imports>
<Rust>
#[test]
fn test_registering_pings_before_init() {
  let sample_ping = PingType::new("test_ping", true, true, vec![]);
  register_ping_type(&sample_ping);
  let cfg = crate::Configuration {
    data_path: "/tmp".into(),
    application_id: "test_app".into(),
    upload_enabled: true,
    max_events: None,
    delay_ping_lifetime_io: false,
    channel: None,
    server_endpoint: None,
    uploader: None,
  };
  let _ = initialize(cfg, crate::ClientInfoMetrics::default());
  // Verify that the ping is registered
  assert!(crate::private::get_registered_pings().contains(&sample_ping));
}</Rust>
