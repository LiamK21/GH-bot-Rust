#[test]
fn test_in_flight_pings_dropped_metric() {
  let mut upload_manager = PingUploadManager::new(UploadMetrics::new());
  let document_id = "test_document_id";
  let path = "test_path";
  upload_manager.track_ping_in_flight(document_id, path);
  upload_manager.upload_ping(document_id, path);
  let expected = 1;
  let actual = upload_manager.upload_metrics.in_flight_pings_dropped.get();
  assert_eq!(expected, actual);
}
