#[test]
fn test_parse_language() {
  let result_kotlin = super::parse_language("kotlin");
  let result_python = super::parse_language("python");
  let result_swift = super::parse_language("swift");
  let result_unknown = super::parse_language("unknown");
  assert!(result_kotlin.is_ok());
  assert!(result_python.is_ok());
  assert!(result_swift.is_ok());
  assert!(result_unknown.is_err());
}
