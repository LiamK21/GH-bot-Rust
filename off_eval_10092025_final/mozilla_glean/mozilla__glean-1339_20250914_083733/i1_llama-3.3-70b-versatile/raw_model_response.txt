<imports> 
use super::shutdown;
use super::initialize;
use std::panic;
use std::thread;
</imports>
<Rust>
#[test]
fn test_shutdown_functionality() {
  initialize("test_cfg".to_string(), "client_info_metrics".to_string());
  std::thread::spawn(move || {
    shutdown();
  });
  let handler = panic::catch_unwind(|| {
    shutdown();
  });
  assert!(handler.is_ok());
}</Rust>
