use super::new_glean;
use super::Glean;

<Rust>
#[test]
fn test_first_run_hour_resolution_increased() {
  let (mut glean, _) = new_glean(None);
  let original_first_run_date = glean.core_metrics.first_run_date.get_value(&glean, "glean_client_info");
  let original_first_run_hour = glean.core_metrics.first_run_hour.get_value(&glean, "metrics");
  assert_eq!(original_first_run_date, original_first_run_hour);
  glean.core_metrics.first_run_hour.set(&glean, Some("2022-01-01T01:00:00+00:00".to_string()));
  let new_first_run_hour = glean.core_metrics.first_run_hour.get_value(&glean, "metrics");
  assert_ne!(original_first_run_date, new_first_run_hour);
}</Rust>
