<imports> 
use super::Histogram;
use std::collections::HashMap;
use super::PrecomputedExponential;
</imports>
<Rust>
#[test]
fn test_histogram_snapshot_values() {
  let bucketing = PrecomputedExponential { min: 1, max: 100, bucket_count: 10 };
  let mut histogram = Histogram { values: HashMap::new(), bucketing };
  histogram.values.insert(10, 5);
  histogram.values.insert(50, 10);
  let expected = histogram.values.clone();
  let actual = histogram.snapshot_values();
  assert_eq!(expected, actual);
}</Rust>
