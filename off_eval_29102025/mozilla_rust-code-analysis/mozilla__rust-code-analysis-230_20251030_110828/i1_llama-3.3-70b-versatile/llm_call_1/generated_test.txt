#[test]
fn test_dump_formats() {
  let space = FuncSpace { /* initialize with some data */ };
  let path = PathBuf::from("test_file");
  let output_path = Some(PathBuf::from("output_file"));
  let format = Format::Json;
  let pretty = true;

  let result = format.dump_formats(&space, &path, &output_path, pretty);
  assert!(result.is_ok());
}
