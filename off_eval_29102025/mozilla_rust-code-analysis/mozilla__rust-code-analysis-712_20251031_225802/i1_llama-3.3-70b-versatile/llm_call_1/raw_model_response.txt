<imports> 
use crate::metrics::exit::Stats;
use std::f64;
</imports>
<Rust>
#[test]
fn test_nexits_min_max() {
  let mut stats = Stats::default();
  stats.exit = 5;
  stats.compute_minmax();
  let expected_min = 5.0;
  let expected_max = 5.0;
  assert_eq!(stats.exit_min(), expected_min);
  assert_eq!(stats.exit_max(), expected_max);
}</Rust>
