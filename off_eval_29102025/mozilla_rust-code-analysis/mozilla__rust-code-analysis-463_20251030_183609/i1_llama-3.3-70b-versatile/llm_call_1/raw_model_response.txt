<imports> 
use tree_sitter_javascript::language;
use tree_sitter_java::language as java_language;
use tree_sitter_typescript::language_typescript;
use tree_sitter_typescript::language_tsx;
use tree_sitter_preproc::language as preproc_language;
use tree_sitter_ccomment::language as ccomment_language;
use tree_sitter_mozcpp::language as cpp_language;
</imports>
<Rust>
#[test]
fn test_get_language() {
  let expected_java = java_language();
  let expected_typescript = language_typescript();
  let expected_tsx = language_tsx();
  let expected_javascript = language();
  let expected_preproc = preproc_language();
  let expected_ccomment = ccomment_language();
  let expected_cpp = cpp_language();

  let actual_java = super::mk_get_language(LANG::Java);
  let actual_typescript = super::mk_get_language(LANG::Typescript);
  let actual_tsx = super::mk_get_language(LANG::Tsx);
  let actual_javascript = super::mk_get_language(LANG::Javascript);
  let actual_preproc = super::mk_get_language(LANG::Preproc);
  let actual_ccomment = super::mk_get_language(LANG::Ccomment);
  let actual_cpp = super::mk_get_language(LANG::Cpp);

  assert_eq!(expected_java, actual_java);
  assert_eq!(expected_typescript, actual_typescript);
  assert_eq!(expected_tsx, actual_tsx);
  assert_eq!(expected_javascript, actual_javascript);
  assert_eq!(expected_preproc, actual_preproc);
  assert_eq!(expected_ccomment, actual_ccomment);
  assert_eq!(expected_cpp, actual_cpp);
}
</Rust>
