#[test]
fn test_loc_min_max() {
  let mut stats = Stats::default();
  stats.sloc.sloc_min = 10;
  stats.sloc.sloc_max = 20;
  stats.compute_minmax();
  let expected_min: f64 = 10.0;
  let expected_max: f64 = 20.0;
  let actual_min = stats.sloc_min();
  let actual_max = stats.sloc_max();
  assert_eq!(actual_min, expected_min);
  assert_eq!(actual_max, expected_max);
}
