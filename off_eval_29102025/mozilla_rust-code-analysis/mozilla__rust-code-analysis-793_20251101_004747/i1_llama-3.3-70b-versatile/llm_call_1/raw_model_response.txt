use super::Stats;
use std::collections::FxHashSet;

<Rust>
#[test]
fn test_loc_min_max() {
  let mut stats = Stats::default();
  let mut other = Stats::default();
  stats.sloc.sloc_min = 10;
  stats.sloc.sloc_max = 20;
  other.sloc.sloc_min = 5;
  other.sloc.sloc_max = 15;
  stats.merge(&other);
  assert_eq!(stats.sloc.sloc_min, 5.0);
  assert_eq!(stats.sloc.sloc_max, 20.0);
}</Rust>
