<imports>
use std::collections::HashSet;
use rustc_hash::FxHashSet;
use super::Stats;
</imports>
<Rust>
#[test]
fn test_compute_minmax_for_loc() {
    let mut stats = Stats::default();
    stats.sloc.start = 0;
    stats.sloc.end = 10;
    stats.ploc.lines = FxHashSet::from_iter([1, 2, 3, 4, 5, 6, 7]);
    stats.lloc.logical_lines = 6;
    stats.cloc.only_comment_lines = 2;
    stats.cloc.code_comment_lines = 2;
    stats.space_count = 1;

    let expected_sloc_min = 10.0;
    let expected_sloc_max = 10.0;
    let expected_ploc_min = 7.0;
    let expected_ploc_max = 7.0;
    let expected_lloc_min = 6.0;
    let expected_lloc_max = 6.0;
    let expected_cloc_min = 4.0;
    let expected_cloc_max = 4.0;
    let expected_blank_min = 1.0;
    let expected_blank_max = 1.0;

    stats.compute_minmax();

    assert_eq!(stats.sloc_min(), expected_sloc_min);
    assert_eq!(stats.sloc_max(), expected_sloc_max);
    assert_eq!(stats.ploc_min(), expected_ploc_min);
    assert_eq!(stats.ploc_max(), expected_ploc_max);
    assert_eq!(stats.lloc_min(), expected_lloc_min);
    assert_eq!(stats.lloc_max(), expected_lloc_max);
    assert_eq!(stats.cloc_min(), expected_cloc_min);
    assert_eq!(stats.cloc_max(), expected_cloc_max);
    assert_eq!(stats.blank_min(), expected_blank_min);
    assert_eq!(stats.blank_max(), expected_blank_max);
}
</Rust>
