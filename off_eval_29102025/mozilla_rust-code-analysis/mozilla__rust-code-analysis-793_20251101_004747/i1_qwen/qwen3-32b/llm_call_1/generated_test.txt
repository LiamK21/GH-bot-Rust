#[test]
fn test_blank_minmax_merge() {
    let mut stats1 = Stats::default();
    stats1.sloc = Sloc {
        start: 1,
        end: 3,
        unit: false,
        sloc_min: usize::MAX,
        sloc_max: 0,
    };
    stats1.ploc = Ploc {
        lines: FxHashSet::from_iter([1usize]),
        ploc_min: usize::MAX,
        ploc_max: 0,
    };
    stats1.cloc = Cloc {
        only_comment_lines: 1,
        code_comment_lines: 0,
        comment_line_end: None,
        cloc_min: usize::MAX,
        cloc_max: 0,
    };
    stats1.compute_minmax();

    let mut stats2 = Stats::default();
    stats2.sloc = Sloc {
        start: 1,
        end: 6,
        unit: false,
        sloc_min: usize::MAX,
        sloc_max: 0,
    };
    stats2.ploc = Ploc {
        lines: FxHashSet::from_iter([1, 2, 3]),
        ploc_min: usize::MAX,
        ploc_max: 0,
    };
    stats2.cloc = Cloc {
        only_comment_lines: 1,
        code_comment_lines: 0,
        comment_line_end: None,
        cloc_min: usize::MAX,
        cloc_max: 0,
    };
    stats2.compute_minmax();

    stats1.merge(&stats2);

    assert_eq!(stats1.blank_min(), 1.0);
    assert_eq!(stats1.blank_max(), 2.0);
}
