use super::RustCode;

<Rust>
#[test]
fn test_rust_lloc_computation() {
  let code = "struct Foo { field: usize, } let foo = Foo { 42 }; foo.field";
  let expected_lloc = 1;
  let stats = RustCode::calculate_loc(code);
  let actual_lloc = stats.logical_lines;
  assert_eq!(expected_lloc, actual_lloc);
}
</Rust>
