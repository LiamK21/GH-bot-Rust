#[test]
fn test_compute_minmax_with_cognitive() {
    let mut state = State::default();
    state.space.metrics.cognitive.structural = 5;
    state.space.metrics.cognitive.total_space_functions = 1;
    compute_minmax(&mut state);
    let expected_sum = 5.0;
    let expected_min = 5.0;
    let expected_max = 5.0;
    let actual_sum = state.space.metrics.cognitive.cognitive_sum();
    let actual_min = state.space.metrics.cognitive.cognitive_min();
    let actual_max = state.space.metrics.cognitive.cognitive_max();
    assert_eq!(expected_sum, actual_sum);
    assert_eq!(expected_min, actual_min);
    assert_eq!(expected_max, actual_max);
}
