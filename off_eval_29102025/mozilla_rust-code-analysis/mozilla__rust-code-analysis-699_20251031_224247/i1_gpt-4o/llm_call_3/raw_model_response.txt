use super::Stats;

<Rust>
#[test]
fn test_cyclomatic_min_max() {
    let mut stats = Stats::default();
    stats.cyclomatic = 3.0;
    stats.compute_minmax();

    let expected_min = 3.0;
    let expected_max = 3.0;

    let actual_min = stats.cyclomatic_min();
    let actual_max = stats.cyclomatic_max();

    assert_eq!(actual_min, expected_min);
    assert_eq!(actual_max, expected_max);
}
</Rust>
