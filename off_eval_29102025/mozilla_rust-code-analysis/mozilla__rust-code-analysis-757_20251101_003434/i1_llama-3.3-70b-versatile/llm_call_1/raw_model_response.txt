#[test]
fn test_cpp_enum_values() {
  let operator_name_value: i32 = 157;
  let this_value: i32 = 161;
  let nullptr_value: i32 = 162;
  let literal_suffix_value: i32 = 163;
  let alone_macro_value: i32 = 164;
  let alone_macro_call_token1_value: i32 = 165;

  let dashgtstar_value: i32 = 157;
  let lparenrparen_value: i32 = 158;
  let lbrackrbrack_value: i32 = 159;
  let dquotedquote_value: i32 = 160;

  assert_eq!(Cpp::DASHGTSTAR as i32, dashgtstar_value);
  assert_eq!(Cpp::LPARENRPAREN as i32, lparenrparen_value);
  assert_eq!(Cpp::LBRACKRBRACK as i32, lbrackrbrack_value);
  assert_eq!(Cpp::DQUOTEDQUOTE as i32, dquotedquote_value);
  assert_eq!(Cpp::This as i32, this_value);
  assert_eq!(Cpp::Nullptr as i32, nullptr_value);
  assert_eq!(Cpp::LiteralSuffix as i32, literal_suffix_value);
  assert_eq!(Cpp::AloneMacro as i32, alone_macro_value);
  assert_eq!(Cpp::AloneMacroCallToken1 as i32, alone_macro_call_token1_value);
}</Rust>
