<imports>
use super::Cpp;
use std::convert::From;
</imports>
<Rust>
#[test]
fn test_operator_name_removal() {
    let operator_name_before_patch = Cpp::OperatorName;
    let operator_name_after_patch = Cpp::DASHGTSTAR;
    
    let expected_before_patch = "operator_name";
    let expected_after_patch = "->*";
    
    let actual_before_patch = <&str>::from(operator_name_before_patch);
    let actual_after_patch = <&str>::from(operator_name_after_patch);
    
    assert_eq!(expected_before_patch, actual_before_patch);
    assert_eq!(expected_after_patch, actual_after_patch);
}
</Rust>
