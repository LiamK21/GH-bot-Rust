#[test]
fn test_class_specifier_with_multiple_macros() {
    let input = "class MOZ_STACK_CLASS JS_PUBLIC_API AutoEntryMonitor";
    let expected = Cpp::ClassSpecifier;
    let actual = KEYS.get("class_specifier").unwrap();
    assert_eq!(expected, *actual);

    let input_with_macros = "class MOZ_STACK_CLASS JS_PUBLIC_API AutoEntryMonitor";
    let expected_with_macros = Cpp::ClassSpecifierRepeat1;
    let actual_with_macros = KEYS.get("class_specifier_repeat1").unwrap();
    assert_eq!(expected_with_macros, *actual_with_macros);
}
</Rust>
