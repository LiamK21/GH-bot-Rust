#[test]
fn test_class_specifier_with_macro_annotations() {
  let class_specifier = Cpp::ClassSpecifier;
  let macro_annotation = Cpp::MacroAnnotation;
  let type_identifier = Cpp::TypeIdentifier;
  let expected = vec![class_specifier, macro_annotation, type_identifier];
  let actual = vec![class_specifier, macro_annotation, type_identifier];
  assert_eq!(expected, actual);
}
