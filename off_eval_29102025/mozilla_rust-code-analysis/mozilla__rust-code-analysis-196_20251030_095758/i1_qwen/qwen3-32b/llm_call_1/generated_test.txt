#[test]
fn test_findcfg_path_initialization() {
    let expected_path = PathBuf::from("test_file.rs");
    let filters = vec!["function".to_string()];
    let cfg = FindCfg {
        path: expected_path.clone(),
        filters,
        line_start: None,
        line_end: None,
    };
    assert_eq!(cfg.path, expected_path);
}
