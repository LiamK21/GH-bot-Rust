<imports> 
use crate::metrics::nom::Stats;
use std::usize;
</imports>
<Rust>
#[test]
fn test_compute_minmax() {
  let mut stats = Stats::default();
  stats.functions = 5;
  stats.closures = 3;
  let expected_functions_min = 5;
  let expected_functions_max = 5;
  let expected_closures_min = 3;
  let expected_closures_max = 3;
  stats.compute_minmax();
  assert_eq!(stats.functions_min() as usize, expected_functions_min);
  assert_eq!(stats.functions_max() as usize, expected_functions_max);
  assert_eq!(stats.closures_min() as usize, expected_closures_min);
  assert_eq!(stats.closures_max() as usize, expected_closures_max);
}</Rust>
