<imports>
use super::compute_minmax;
use crate::metrics::nom::Stats;
use crate::spaces::State;
use crate::spaces::Space;
use crate::spaces::Metrics;
</imports>
<Rust>
#[test]
fn test_compute_minmax_updates_nom_minmax() {
    let mut stats = Stats::default();
    stats.functions_sum = 3;
    stats.closures_sum = 2;
    stats.compute_minmax();

    let mut state = State {
        space: Space {
            metrics: Metrics {
                nom: stats,
                ..Default::default()
            },
            ..Default::default()
        },
        ..Default::default()
    };

    compute_minmax(&mut state);

    assert_eq!(state.space.metrics.nom.functions_min(), 3.0);
    assert_eq!(state.space.metrics.nom.functions_max(), 3.0);
    assert_eq!(state.space.metrics.nom.closures_min(), 2.0);
    assert_eq!(state.space.metrics.nom.closures_max(), 2.0);
}
</Rust>
