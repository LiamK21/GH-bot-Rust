#[test]
fn test_dump_json_to_stdout() {
    let test_file_path = "test_file.rs";
    let mut file = File::create(test_file_path).expect("Failed to create test file");
    writeln!(file, "fn test() {}").expect("Failed to write to test file");

    let output = Command::new("cargo")
        .arg("run")
        .arg("--")
        .arg("--function")
        .arg("--output-format")
        .arg("json")
        .arg("--output")
        .arg("")
        .arg(test_file_path)
        .output()
        .expect("Failed to execute command");

    let stdout = from_utf8(&output.stdout).expect("Failed to convert stdout to string");
    assert!(stdout.contains("\"kind\":\"function\""));
    assert!(stdout.contains("\"name\":\"test\""));
    assert!(stdout.contains("\"start_line\":1"));
    assert!(stdout.contains("\"end_line\":1"));
    assert!(stdout.contains("\"metrics\":")); // This line ensures the test fails before the patch

    std::fs::remove_file(test_file_path).expect("Failed to delete test file");
}
</Rust>
