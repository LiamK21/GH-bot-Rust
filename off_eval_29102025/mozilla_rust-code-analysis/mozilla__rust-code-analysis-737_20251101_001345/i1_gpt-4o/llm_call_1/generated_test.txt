#[test]
fn test_loc_average_calculation() {
    let mut stats = Stats::default();
    stats.only_comment_lines = 4;
    stats.logical_lines = 6;
    stats.space_count = 2;

    let expected_sloc_average = 5.0;
    let expected_ploc_average = 3.5;
    let expected_lloc_average = 3.0;
    let expected_cloc_average = 2.0;
    let expected_blank_average = 0.5;

    let actual_sloc_average = stats.sloc_average();
    let actual_ploc_average = stats.ploc_average();
    let actual_lloc_average = stats.lloc_average();
    let actual_cloc_average = stats.cloc_average();
    let actual_blank_average = stats.blank_average();

    assert_eq!(expected_sloc_average, actual_sloc_average);
    assert_eq!(expected_ploc_average, actual_ploc_average);
    assert_eq!(expected_lloc_average, actual_lloc_average);
    assert_eq!(expected_cloc_average, actual_cloc_average);
    assert_eq!(expected_blank_average, actual_blank_average);
}
