#[test]
fn test_nexits_metric_with_qmark_operator() {
  let mut stats = Stats { exit: 0, exit_max: 0, exit_min: 0, exit_sum: 0, total_space_functions: 0 };
  let node = Node(Rust::TryExpression);
  Exit::compute(&node, &mut stats);
  assert_eq!(stats.exit, 1);
  let mut stats_qmark = Stats { exit: 0, exit_max: 0, exit_min: 0, exit_sum: 0, total_space_functions: 0 };
  let qmark_node = Node(Rust::QMARK);
  Exit::compute(&qmark_node, &mut stats_qmark);
  assert_eq!(stats_qmark.exit, 0);
}
