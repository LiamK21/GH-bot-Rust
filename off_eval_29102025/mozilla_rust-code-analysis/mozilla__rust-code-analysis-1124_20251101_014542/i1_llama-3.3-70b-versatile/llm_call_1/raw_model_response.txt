<imports> 
use super::Exit;
use super::Node;
use super::Rust;
use super::Stats;
</imports>
<Rust>
#[test]
fn test_nexits_metric_with_qmark_operator() {
  let mut stats = Stats { exit: 0 };
  let node = Node::new(Rust::QMARK);
  let func_node = Node::new(Rust::Function);
  Exit::compute(&node, &mut stats);
  let mut stats_func = Stats { exit: 0 };
  Exit::compute(&func_node, &mut stats_func);
  assert_eq!(stats.exit, 0);
  assert_eq!(stats_func.exit, 0);
  let try_node = Node::new(Rust::TryExpression);
  let mut stats_try = Stats { exit: 0 };
  Exit::compute(&try_node, &mut stats_try);
  assert_eq!(stats_try.exit, 1);
}</Rust>
