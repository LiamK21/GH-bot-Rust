#[test]
fn test_try_expression_counted_as_exit() {
    let source_code = r#"
    fn example() -> Result<(), Error> {
        let value = some_function()?;
        Ok(())
    }
    "#;
    let node = parse_to_node(source_code, "source_file");
    let mut stats = Stats::default();
    RustCode::compute(&node, &mut stats);
    assert_eq!(stats.exit, 1);
}
