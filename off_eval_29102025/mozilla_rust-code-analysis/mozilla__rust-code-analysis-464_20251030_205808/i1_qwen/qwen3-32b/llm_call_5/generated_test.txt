mod test_python {
    use tree_sitter::Language;

    get_language!(tree_sitter_python);

    #[test]
    fn test_get_language_python() {
        let lang = get_language();
        let name = lang.name();
        let c_str = unsafe { std::ffi::CStr::from_ptr(name) };
        assert_eq!(c_str.to_str().unwrap(), "python");
    }
}
