#[test]
fn test_remove_trailing_blank_lines() {
  let mut data = "fn main() {\n\n}".as_bytes().to_vec();
  let expected_data = "fn main() {\n\n}".as_bytes().to_vec();
  let mut data_with_trailing_blank_lines = "fn main() {\n\n\n\n".as_bytes().to_vec();
  let expected_data_with_trailing_blank_lines = "fn main() {\n\n\n".as_bytes().to_vec();
  super::remove_blank_lines(&mut data_with_trailing_blank_lines);
  assert_eq!(data_with_trailing_blank_lines, expected_data_with_trailing_blank_lines);
  super::remove_blank_lines(&mut data);
  assert_eq!(data, expected_data);
}
