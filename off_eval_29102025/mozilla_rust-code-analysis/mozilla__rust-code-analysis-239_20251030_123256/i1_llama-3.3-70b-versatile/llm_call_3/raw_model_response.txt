<imports>
use std::path::PathBuf;
use std::io;
use super::remove_blank_lines;
</imports>
<Rust>
#[test]
fn test_remove_trailing_blank_lines() {
  let mut data_with_trailing_blank_lines = "fn main() {\n\n\n\n".as_bytes().to_vec();
  let expected_data_with_trailing_blank_lines = "fn main() {\n".as_bytes().to_vec();
  remove_blank_lines(&mut data_with_trailing_blank_lines);
  assert_eq!(data_with_trailing_blank_lines, expected_data_with_trailing_blank_lines);

  let mut data_without_trailing_blank_lines = "fn main() {\n\n".as_bytes().to_vec();
  let expected_data_without_trailing_blank_lines = "fn main() {\n\n".as_bytes().to_vec();
  remove_blank_lines(&mut data_without_trailing_blank_lines);
  assert_eq!(data_without_trailing_blank_lines, expected_data_without_trailing_blank_lines);
}
</Rust>
