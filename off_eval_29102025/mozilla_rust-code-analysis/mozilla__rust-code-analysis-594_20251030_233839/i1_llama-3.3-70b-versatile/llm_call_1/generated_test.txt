#[test]
fn test_get_language() {
  let lang = LANG::Java;
  let expected = java_language();
  let actual = get_language(lang);
  assert_eq!(expected, actual);

  let lang = LANG::Typescript;
  let expected = typescript_language();
  let actual = get_language(lang);
  assert_eq!(expected, actual);

  let lang = LANG::Tsx;
  let expected = tsx_language();
  let actual = get_language(lang);
  assert_eq!(expected, actual);

  let lang = LANG::Javascript;
  let expected = javascript_language();
  let actual = get_language(lang);
  assert_eq!(expected, actual);

  let lang = LANG::Python;
  let expected = python_language();
  let actual = get_language(lang);
  assert_eq!(expected, actual);

  let lang = LANG::Rust;
  let expected = rust_language();
  let actual = get_language(lang);
  assert_eq!(expected, actual);

  let lang = LANG::Preproc;
  let expected = preproc_language();
  let actual = get_language(lang);
  assert_eq!(expected, actual);

  let lang = LANG::Ccomment;
  let expected = ccomment_language();
  let actual = get_language(lang);
  assert_eq!(expected, actual);

  let lang = LANG::Cpp;
  let expected = cpp_language();
  let actual = get_language(lang);
  assert_eq!(expected, actual);

  let lang = LANG::Mozjs;
  let expected = mozjs_language();
  let actual = get_language(lang);
  assert_eq!(expected, actual);
}
