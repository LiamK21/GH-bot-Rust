<imports>
use super::compute_minmax;
use crate::spaces::State;
use crate::metrics::Stats;
use crate::metrics::Metrics;
</imports>
<Rust>
#[test]
fn test_compute_minmax_updates_nargs_minmax() {
    let mut state = State {
        space: Metrics {
            cyclomatic: Default::default(),
            nexits: Default::default(),
            cognitive: Default::default(),
            nargs: Stats {
                fn_nargs: 3,
                closure_nargs: 2,
                fn_nargs_sum: 0,
                closure_nargs_sum: 0,
                fn_nargs_min: usize::MAX,
                closure_nargs_min: usize::MAX,
                fn_nargs_max: 0,
                closure_nargs_max: 0,
                total_functions: 1,
                total_closures: 1,
            },
        },
        halstead_maps: Default::default(),
    };

    let expected_fn_nargs_min = 3.0;
    let expected_fn_nargs_max = 3.0;
    let expected_closure_nargs_min = 2.0;
    let expected_closure_nargs_max = 2.0;

    compute_minmax(&mut state);

    assert_eq!(state.space.metrics.nargs.fn_args_min(), expected_fn_nargs_min);
    assert_eq!(state.space.metrics.nargs.fn_args_max(), expected_fn_nargs_max);
    assert_eq!(state.space.metrics.nargs.closure_args_min(), expected_closure_nargs_min);
    assert_eq!(state.space.metrics.nargs.closure_args_max(), expected_closure_nargs_max);
}
</Rust>
