#[test]
fn test_alterate_function_expression() {
  let node = Mozjs::Function;
  let code = b"function foo() {}";
  let span = true;
  let children = vec![];
  let expected = false;
  let actual = match Mozjs::from(node) {
    Mozjs::String | Mozjs::String2 => true,
    _ => false,
  };
  assert_eq!(actual, expected);
}
