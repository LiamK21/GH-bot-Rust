#[test]
fn test_alterate_function_expression() {
  let node = Mozjs::FunctionExpression;
  let code = b"function foo() {}";
  let span = true;
  let children = vec![];
  let expected = Mozjs::String | Mozjs::String2;
  let actual = match Mozjs::from(node) {
    Mozjs::String | Mozjs::String2 => true,
    _ => false,
  };
  assert!(actual);
}
