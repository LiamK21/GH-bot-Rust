<imports>
use super::shutdown;
use std::sync::mpsc;
use std::thread;
use std::time::Duration;
</imports>
<Rust>
#[test]
fn test_shutdown_timeout() {
    let (tx, rx) = mpsc::channel();
    thread::spawn(move || {
        shutdown();
        tx.send(()).unwrap();
    });

    let result = rx.recv_timeout(Duration::from_secs(15));
    assert!(result.is_ok(), "Shutdown did not complete within the expected timeout");
}</Rust>
