<imports> 
use glean_core::traits;
use glean_core::Glean;
</imports>
<Rust>
#[test]
fn test_expose_glean_core_traits() {
  let glean = Glean::new();
  let exposed_traits: Vec<&str> = vec!["boolean", "counter", "custom_distribution", "datetime", "event", "labeled", "memory_distribution", "quantity", "string", "string_list", "timespan", "timing_distribution", "uuid"];
  let actual_traits: Vec<&str> = traits::get_traits();
  assert_eq!(exposed_traits, actual_traits);
}
