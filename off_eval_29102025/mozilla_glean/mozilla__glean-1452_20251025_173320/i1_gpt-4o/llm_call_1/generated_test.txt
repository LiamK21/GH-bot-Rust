#[test]
fn test_expose_glean_core_traits() {
    struct TestBooleanMetric;
    impl BooleanMetric for TestBooleanMetric {
        fn set(&self, _glean: &glean_core::Glean, _value: bool) {}
    }

    let test_metric = TestBooleanMetric;
    let glean = glean_core::global_glean();
    test_metric.set(&glean, true);
}
