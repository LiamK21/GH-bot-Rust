<imports>
use super::validate_dynamic_label;
use crate::Glean;
use crate::CommonMetricData;
use crate::error::ErrorKind;
use crate::Configuration;
</imports>
<Rust>
#[test]
fn test_validate_dynamic_label() {
  let config = Configuration::new();
  let glean = Glean::new(config).unwrap();
  let meta = CommonMetricData {
    dynamic_label: Some("test_label".to_string()),
    ..Default::default()
  };
  let base_identifier = "test_base_identifier";
  let label = "test_label";
  let result = validate_dynamic_label(&glean, &meta, base_identifier, label);
  let expected_result = "test_base_identifier.test_label";
  assert_eq!(result, expected_result);
}
</Rust>
