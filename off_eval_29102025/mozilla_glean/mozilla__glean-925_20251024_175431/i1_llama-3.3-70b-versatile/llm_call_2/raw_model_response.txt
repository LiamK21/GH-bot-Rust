<imports>
use crate::Lifetime;
use crate::MemoryUnit;
use crate::TimeUnit;

<Rust>
#[test]
fn test_lifetime_enum() {
  let lifetime: i32 = Lifetime::Ping as i32;
  let memory_unit: i32 = MemoryUnit::Byte as i32;
  let time_unit: i32 = TimeUnit::Nanosecond as i32;
  assert_eq!(lifetime, 0);
  assert_eq!(memory_unit, 0);
  assert_eq!(time_unit, 0);
}
