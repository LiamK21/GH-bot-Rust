<imports>use super::glean_new_datetime_metric;
use crate::{Lifetime, TimeUnit};
use ffi_support::FfiStr;
use std::ptr;</imports>
<Rust>
#[test]
fn test_create_datetime_metric_with_time_unit() {
    let category = FfiStr::from_str("test_category");
    let name = FfiStr::from_str("test_name");
    let send_in_pings = ptr::null();
    let send_in_pings_len = 0;
    let lifetime = Lifetime::Ping;
    let disabled = 0;
    let time_unit = TimeUnit::Nanosecond;
    
    let metric_id = glean_new_datetime_metric(
        category,
        name,
        send_in_pings,
        send_in_pings_len,
        lifetime,
        disabled,
        time_unit,
    );
    
    assert!(metric_id > 0);
}</Rust>
