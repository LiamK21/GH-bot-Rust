#[test]
fn test_standalone_uploader_rate_limiter() {
    let data_dir = CString::new("/tmp").unwrap();
    let ffi_str = FfiStr::from_cstr(data_dir.as_c_str());
    let result = unsafe { glean_initialize_standalone_uploader(ffi_str) };
    assert_eq!(result, 1);
