<imports>
use super::MemoryDistributionMetric;
use glean_core::metrics::DistributionData;
use glean_core::metrics::MemoryUnit;
use std::collections::HashMap;
</imports>
<Rust>
#[test]
fn test_memory_distribution_metric() {
  let meta = glean_core::CommonMetricData {
    name: "test_metric".to_string(),
    category: "test_category".to_string(),
    send_in_pings: vec!["test_ping".to_string()],
    ..Default::default()
  };
  let metric = MemoryDistributionMetric::new(meta, MemoryUnit::Byte);
  metric.accumulate(100);
  let expected = DistributionData { values: HashMap::new(), sum: 100 };
  let actual = metric.test_get_value(None).unwrap();
  assert_eq!(actual.sum, expected.sum);
}
</Rust>
