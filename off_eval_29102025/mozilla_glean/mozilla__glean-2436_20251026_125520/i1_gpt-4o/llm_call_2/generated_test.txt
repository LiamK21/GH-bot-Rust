#[test]
fn test_create_date_header_value_format() {
    let current_time = Utc::now();
    let expected_format = format!(
        "{}, {:02} {} {} {:02}:{:02}:{:02} GMT",
        current_time.format("%a"),
        current_time.day(),
        current_time.format("%b"),
        current_time.year(),
        current_time.hour(),
        current_time.minute(),
        current_time.second()
    );
    let actual = create_date_header_value(current_time);
    assert_eq!(expected_format, actual);
}
