<imports>
use super::QuantityMetric;
use glean_core::metrics::MetricType;
use glean_core::ErrorType;
use std::sync::Arc;
use glean_core::CommonMetricData;
</imports>
<Rust>
#[test]
fn test_quantity_metric_set_and_get() {
  let meta = CommonMetricData::new("test_quantity", MetricType::Quantity, "ping_name".to_string());
  let quantity_metric = QuantityMetric::new(meta);
  let value: i64 = 10;
  quantity_metric.set(value);
  let actual_value = quantity_metric.test_get_value(None);
  assert_eq!(actual_value, Some(value));
}
</Rust>
