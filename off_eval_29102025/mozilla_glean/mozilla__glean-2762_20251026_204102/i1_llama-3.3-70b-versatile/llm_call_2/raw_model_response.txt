<imports>
use std::env;
use anyhow::{bail, Context};
use camino::Utf8PathBuf;
use uniffi::{generate_bindings, TargetLanguage};
use uniffi_bindgen::BindingGeneratorDefault;
</imports>
<Rust>
#[test]
fn test_binding_generator_default() {
  let target_languages = vec![uniffi::TargetLanguage::Rust];
  let generator = BindingGeneratorDefault {
    target_languages: target_languages.clone(),
    try_format_code: false,
  };
  let expected_target_languages = target_languages;
  assert_eq!(generator.target_languages, expected_target_languages);
}
</Rust>
