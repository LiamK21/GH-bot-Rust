<imports>
use std::collections::HashMap;
use glean_core::private::event::EventMetric;
use glean_core::get_timestamp_ms;
use glean_core::CommonMetricData;
use glean_core::Lifetime;
</imports>
<Rust>
#[test]
fn test_record_with_time() {
  let meta = CommonMetricData {
    name: "test_event".to_string(),
    category: "test_category".to_string(),
    lifetime: Lifetime::Application,
    disabled: false,
    dynamic_label: None,
    send_in_pings: vec![],
  };
  let event = EventMetric::new(meta);
  let timestamp = get_timestamp_ms();
  let extra = HashMap::new();
  event.record_with_time(timestamp, extra.clone());
  assert!(true);
}
</Rust>
