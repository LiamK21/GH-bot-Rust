<imports>
use super::TimingDistributionMetric;
use std::sync::Arc;
use std::sync::Mutex;
use std::sync::atomic::AtomicUsize;
use std::sync::atomic::Ordering;
</imports>
<Rust>
#[test]
fn test_timer_id_starts_at_one() {
    let metric = TimingDistributionMetric {
        meta: Arc::new(Default::default()),
        time_unit: Default::default(),
        next_id: Arc::new(AtomicUsize::new(1)),
        start_times: Arc::new(Mutex::new(Default::default())),
    };
    
    let expected = 1;
    let actual = metric.next_id.load(Ordering::SeqCst);
    
    assert_eq!(expected, actual);
}
</Rust>
