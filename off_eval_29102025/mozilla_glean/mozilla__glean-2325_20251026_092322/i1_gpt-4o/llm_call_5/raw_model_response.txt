<imports>
use super::get_windows_build_number;
use std::i64;
</imports>
<Rust>
#[test]
fn test_get_windows_build_number() {
    #[cfg(target_os = "windows")]
    {
        let expected = Some(22000); // Example build number for testing
        let actual = get_windows_build_number();
        assert_eq!(expected, actual);
    }
    
    #[cfg(not(target_os = "windows"))]
    {
        let expected = None;
        let actual = get_windows_build_number();
        assert_eq!(expected, actual);
    }
}
</Rust>
