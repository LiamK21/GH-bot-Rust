<imports>
use super::system;
use super::ClientInfoMetrics;
use glean_core::ClientInfoMetrics as GleanClientInfoMetrics;
</imports>
<Rust>
#[test]
fn test_windows_build_number_included_in_client_info() {
  let os_version = system::get_os_version();
  let windows_build_number = system::get_windows_build_number();
  let client_info = ClientInfoMetrics {
    app_display_version: "".to_string(),
    channel: Some("".to_string()),
    os_version,
    windows_build_number,
    architecture: "".to_string(),
    android_sdk_version: None,
    device_manufacturer: None,
    device_model: None,
    locale: None,
  };
  let glean_client_info: GleanClientInfoMetrics = client_info.into();
  let expected_build_number = system::get_windows_build_number();
  assert_eq!(glean_client_info.windows_build_number, expected_build_number);
}
</Rust>
