<imports>
use std::collections::HashMap;
use super::PingRequest;
use serde_json::json;
</imports>
<Rust>
#[test]
fn test_user_agent_customization() {
  let document_id = "test_document_id";
  let path = "test_path";
  let body = json!({});
  let ping_request = PingRequest::new(document_id, path, body);
  let headers = ping_request.headers;
  let expected_header = ("User-Agent", "Glean");
  assert!(headers.contains_key(&expected_header.0));
}
</Rust>
