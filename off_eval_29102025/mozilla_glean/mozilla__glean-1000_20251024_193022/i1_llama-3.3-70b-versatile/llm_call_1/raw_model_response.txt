<imports> 
use glean_core::upload::request::PingRequest;
use glean_core::upload::request::JsonValue;
</imports>
<Rust>
#[test]
fn test_user_agent_customization() {
  let document_id = "test_document_id";
  let path = "test_path";
  let body = JsonValue::Object(serde_json::json!({}).as_object().unwrap().clone());
  let ping_request = PingRequest::new(document_id, path, body);
  let headers = ping_request.headers;
  let expected_header = ("User-Agent", "Glean");
  assert!(headers.contains(&expected_header));
}</Rust>
