#[test]
fn test_user_agent_customization_removal() {
    let document_id = "test_doc_id";
    let path = "/test/path";
    let body = json!({"key": "value"});

    let request = PingRequest::new(document_id, path, body);

    let expected_headers = vec![
        ("Content-Encoding", "gzip"),
        ("Content-Type", "application/json"),
    ];

    for (key, value) in expected_headers {
        assert!(request.headers.contains(&(key.to_string(), value.to_string())));
    }

    assert!(!request.headers.iter().any(|(key, _)| key == "userAgent"));
}
