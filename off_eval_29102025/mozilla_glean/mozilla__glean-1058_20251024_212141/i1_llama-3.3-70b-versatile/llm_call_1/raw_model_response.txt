use super::glean_set_log_pings;

use std::ptr::null_mut;

<Rust>
#[test]
fn test_glean_get_upload_task_and_log_pings() {
  let mut result = std::ptr::null_mut();
  let log_pings = 1u8;
  glean_set_log_pings(log_pings);
  glean_get_upload_task(&mut result);
  assert_ne!(result, null_mut());
  glean_set_log_pings(0);
  glean_get_upload_task(&mut result);
  assert_ne!(result, null_mut());
}</Rust>
