#[test]
fn test_glean_get_upload_task_and_log_pings() {
  let mut result = std::ptr::null_mut();
  let log_pings = 1u8;
  unsafe { glean_set_log_pings(log_pings); }
  unsafe { glean_get_upload_task(&mut result); }
  assert_ne!(result, null_mut());
  let log_pings = 0u8;
  unsafe { glean_set_log_pings(log_pings); }
  let mut result2 = std::ptr::null_mut();
  unsafe { glean_get_upload_task(&mut result2); }
  assert_ne!(result2, null_mut());
}
