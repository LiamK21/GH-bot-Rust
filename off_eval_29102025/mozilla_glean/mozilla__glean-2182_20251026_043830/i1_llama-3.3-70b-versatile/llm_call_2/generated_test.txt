#[test]
fn test_uuid_encoding() {
  let glean = Glean::new();
  let metric = UuidMetric::new("test_metric");
  let value = "123e4567e89b12d3a456b7710000111";
  let expected = "123e4567-e89b-12d3-a456-b7710000111";
  let actual = metric.record(glean, value).unwrap().to_string();
  assert_eq!(actual, expected);
}
