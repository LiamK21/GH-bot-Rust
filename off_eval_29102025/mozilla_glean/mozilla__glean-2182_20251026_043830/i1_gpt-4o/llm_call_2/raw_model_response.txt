<imports>
use super::UuidMetric;
use uuid::Uuid;
use crate::Glean; 
use crate::storage::StorageManager;
</imports>
<Rust>
#[test]
fn test_uuid_hyphenated_encoding() {
    let glean = Glean::new(StorageManager::new());
    let uuid_metric = UuidMetric::new("test_category", "test_name", "test_send_in_pings", false);
    let input_uuid = "123e4567e89b12d3a456426655440000";
    let expected_uuid = "123e4567-e89b-12d3-a456-426655440000";

    uuid_metric.set(&glean, input_uuid);

    let stored_value = glean.storage().get("test_category", "test_name").unwrap();
    assert_eq!(stored_value, expected_uuid);
}
</Rust>
