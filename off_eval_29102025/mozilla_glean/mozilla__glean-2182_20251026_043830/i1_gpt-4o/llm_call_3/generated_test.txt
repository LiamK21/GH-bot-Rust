#[test]
fn test_uuid_hyphenated_encoding() {
    let glean = Glean::new(Database::new());
    let common_metric_data = CommonMetricData::new("test_category", "test_name", "test_send_in_pings");
    let uuid_metric = UuidMetric::new(common_metric_data);
    let input_uuid = "123e4567e89b12d3a456426655440000".to_string();
    let expected_uuid = "123e4567-e89b-12d3-a456-426655440000";

    uuid_metric.set(input_uuid);

    let stored_value = glean.storage().get("test_category", "test_name").unwrap();
    assert_eq!(stored_value, expected_uuid);
}
