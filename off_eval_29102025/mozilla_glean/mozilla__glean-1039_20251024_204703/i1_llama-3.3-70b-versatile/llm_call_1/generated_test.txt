#[test]
fn test_upload_error_reporting() {
  let cfg = Configuration {
    data_path: "/tmp/glean_test".to_string(),
    application_id: "glean_test".to_string(),
    upload_enabled: true,
    ..Default::default()
  };
  let glean = Glean::new(cfg).unwrap();
  let task = FfiPingUploadTask::Upload {
    document_id: ptr::null(),
    headers: ptr::null(),
    body: ptr::null(),
  };
  let status = 200;
  glean_process_ping_upload_response(&mut task, status);
  assert!(true);
}
