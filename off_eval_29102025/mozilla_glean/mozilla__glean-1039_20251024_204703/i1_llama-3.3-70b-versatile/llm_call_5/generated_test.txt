#[test]
fn test_upload_error_reporting() {
  let cfg = Configuration {
    data_path: "/tmp/glean_test".to_string(),
    application_id: "glean_test".to_string(),
    language_binding_name: "test_language".to_string(),
    delay_ping_lifetime_io: false,
    upload_enabled: true,
    max_events: None,
  };
  let glean = Glean::new(cfg).unwrap();
  let mut task = FfiPingUploadTask::Upload {
    document_id: std::ptr::null_mut(),
    headers: std::ptr::null_mut(),
    body: std::ptr::null_mut(),
    path: std::ptr::null_mut(),
  };
  let status = 200;
  glean_process_ping_upload_response(&mut task, status);
  assert!(true);
}
