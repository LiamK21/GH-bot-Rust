#[test]
fn test_ping_upload_task_variant() {
  let upload_manager = UploadManager::new();
  let expected_variant = FfiPingUploadTask::Upload;
  let actual_variant: FfiPingUploadTask = FfiPingUploadTask::Upload;
  assert_eq!(expected_variant, actual_variant);
  let _unused = upload_manager.clear_ping_queue();
}
