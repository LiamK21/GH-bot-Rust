#[test]
fn test_client_id_is_set_to_random_value_when_uploading_enabled_at_start() {
    let (glean, _dir) = new_glean(None);
    let expected_client_id = glean.core_metrics.client_id.test_get_value(&glean, "metrics").unwrap();
    let actual_client_id = glean.core_metrics.client_id.test_get_value(&glean, "metrics").unwrap();
    assert_eq!(expected_client_id, actual_client_id);
}
