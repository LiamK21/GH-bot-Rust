<imports> 
use glean_core::rlb::private::object::ObjectMetric;
use serde_json::json;
use std::collections::HashMap;
</imports>
<Rust>
#[test]
fn test_set_string_api_with_null_values() {
  let metric: ObjectMetric<HashMap<String, String>> = ObjectMetric::new(Default::default());
  let arr_str = json!({
    "key1": "value1",
    "key2": null,
  })
  .to_string();
  metric.set_string(arr_str);
  let data = metric.test_get_value(None).expect("no object recorded");
  let expected = json!({
    "key1": "value1"
  });
  assert_eq!(expected, data);
  assert_eq!(
    0,
    metric.test_get_num_recorded_errors(glean_core::ErrorType::InvalidValue)
  );
}</Rust>
