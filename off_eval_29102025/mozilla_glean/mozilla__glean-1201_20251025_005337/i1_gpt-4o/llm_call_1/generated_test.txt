struct TestLogger {
    messages: Mutex<Vec<String>>,
}

impl Log for TestLogger {
    fn enabled(&self, metadata: &Metadata) -> bool {
        metadata.level() <= Level::Info
    }

    fn log(&self, record: &Record) {
        if self.enabled(record.metadata()) {
            let mut messages = self.messages.lock().unwrap();
            messages.push(format!("{}", record.args()));
        }
    }

    fn flush(&self) {}
}

#[test]
fn test_glean_disabled_logging() {
    let test_logger = TestLogger {
        messages: Mutex::new(Vec::new()),
    };
    set_logger(Box::leak(Box::new(test_logger))).unwrap();
    set_max_level(Level::Info.to_level_filter());

    let glean = Glean::new(); // Assuming a constructor or method to create a Glean instance
    glean.submit_ping(&PingType::default(), None).unwrap(); // Assuming PingType::default() is valid

    let messages = test_logger.messages.lock().unwrap();
    let expected_message = "Glean disabled: not submitting any pings.";
    assert!(messages.contains(&expected_message.to_string()));
}
