#[test]
fn test_shutdown_functionality() {
  initialize(super::Configuration {}, super::ClientInfoMetrics {});
  let handle = thread::spawn(move || {
    shutdown();
  });
  handle.join().unwrap();
  // Since we can't directly test if the dispatcher is shut down, 
  // we can test if the shutdown function runs without panicking.
  info!("Shutdown function ran without panicking");
}
