<imports> 
use super::glean_flush_rlb_dispatcher;
use log::info;
use std::sync::atomic::{AtomicBool, Ordering};
use std::sync::Arc;
</imports>
<Rust>
#[test]
fn test_glean_flush_rlb_dispatcher() {
  let initialized = Arc::new(AtomicBool::new(false));
  let initialized_clone = Arc::clone(&initialized);
  std::thread::spawn(move || {
    glean_flush_rlb_dispatcher();
    initialized_clone.store(true, Ordering::SeqCst);
  });
  while !initialized.load(Ordering::SeqCst) {}
  info!("Test glean_flush_rlb_dispatcher passed");
}</Rust>
