#[test]
fn test_string_metric_set_and_get_value() {
  let metric = StringMetric::new(glean_core::metrics::CommonMetricData {
    name: "string_metric".into(),
    category: "test".into(),
    send_in_pings: vec!["store1".into()],
    lifetime: glean_core::Lifetime::Application,
    disabled: false,
    dynamic_label: None,
  });
  let sample_string = "test_string";
  metric.set(sample_string);
  let actual_value = metric.test_get_value(None).unwrap();
  assert_eq!(sample_string, actual_value);
}
