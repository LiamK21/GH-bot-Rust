<imports>
use super::Builder;
use crate::error::ErrorKind;
</imports>
<Rust>
#[test]
fn test_ping_request_body_exceeds_max_size() {
    let builder = Builder::new("Rust", 1)
        .document_id("test_id")
        .path("/test/path")
        .body("This body is definitely larger than 1 byte");

    let result = builder.build();

    assert!(matches!(result, Err(e) if matches!(e.kind(), ErrorKind::PingBodyOverflow(_))));
}</Rust>
