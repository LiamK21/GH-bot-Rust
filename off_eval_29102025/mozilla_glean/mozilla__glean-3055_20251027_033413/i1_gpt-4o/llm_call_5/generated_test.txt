#[test]
fn test_pingtype_metadata_access() {
    let ping_type = PingType::new(
        "test_ping",
        true,
        true,
        true,
        true,
        Arc::new(AtomicBool::new(true)),
        Arc::new(AtomicBool::new(true)),
        vec!["ping1".to_string(), "ping2".to_string()],
        vec!["reason1".to_string(), "reason2".to_string()],
    );

    let expected_name = "test_ping";
    let expected_include_client_id = true;
    let expected_send_if_empty = true;
    let expected_precise_timestamps = true;
    let expected_include_info_sections = true;
    let expected_naively_enabled = true;
    let expected_follows_collection_enabled = true;
    let expected_schedules_pings = vec!["ping1".to_string(), "ping2".to_string()];
    let expected_reason_codes = vec!["reason1".to_string(), "reason2".to_string()];

    let actual_name = ping_type.name();
    let actual_include_client_id = ping_type.include_client_id();
    let actual_send_if_empty = ping_type.send_if_empty();
    let actual_precise_timestamps = ping_type.precise_timestamps();
    let actual_include_info_sections = ping_type.include_info_sections();
    let actual_naively_enabled = ping_type.naively_enabled();
    let actual_follows_collection_enabled = ping_type.follows_collection_enabled();
    let actual_schedules_pings = ping_type.schedules_pings();
    let actual_reason_codes = ping_type.reason_codes();

    assert_eq!(actual_name, expected_name);
    assert_eq!(actual_include_client_id, expected_include_client_id);
    assert_eq!(actual_send_if_empty, expected_send_if_empty);
    assert_eq!(actual_precise_timestamps, expected_precise_timestamps);
    assert_eq!(actual_include_info_sections, expected_include_info_sections);
    assert_eq!(actual_naively_enabled, expected_naively_enabled);
    assert_eq!(actual_follows_collection_enabled, expected_follows_collection_enabled);
    assert_eq!(actual_schedules_pings, &expected_schedules_pings[..]);
    assert_eq!(actual_reason_codes, &expected_reason_codes[..]);
}
