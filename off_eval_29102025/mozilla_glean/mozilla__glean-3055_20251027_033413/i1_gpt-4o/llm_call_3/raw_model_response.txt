<imports>
use super::PingType;
use std::sync::atomic::AtomicBool;
use std::sync::Arc;
</imports>
<Rust>
#[test]
fn test_pingtype_metadata_access() {
    let ping_type = PingType::new(
        "test_ping",
        true,
        true,
        true,
        true,
        Arc::new(AtomicBool::new(true)),
        Arc::new(AtomicBool::new(true)),
        vec!["ping1".to_string(), "ping2".to_string()],
        vec!["reason1".to_string(), "reason2".to_string()],
    );

    let expected_name = "test_ping";
    let expected_include_client_id = true;
    let expected_send_if_empty = true;
    let expected_precise_timestamps = true;
    let expected_include_info_sections = true;
    let expected_naively_enabled = true;
    let expected_follows_collection_enabled = true;
    let expected_schedules_pings = vec!["ping1".to_string(), "ping2".to_string()];
    let expected_reason_codes = vec!["reason1".to_string(), "reason2".to_string()];

    assert_eq!(ping_type.name(), expected_name);
    assert_eq!(ping_type.include_client_id(), expected_include_client_id);
    assert_eq!(ping_type.send_if_empty(), expected_send_if_empty);
    assert_eq!(ping_type.precise_timestamps(), expected_precise_timestamps);
    assert_eq!(ping_type.include_info_sections(), expected_include_info_sections);
    assert_eq!(ping_type.naively_enabled(), expected_naively_enabled);
    assert_eq!(ping_type.follows_collection_enabled(), expected_follows_collection_enabled);
    assert_eq!(ping_type.schedules_pings(), &expected_schedules_pings[..]);
    assert_eq!(ping_type.reason_codes(), &expected_reason_codes[..]);
}
</Rust>
