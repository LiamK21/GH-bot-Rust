<imports>
use super::PingType;
use std::sync::atomic::AtomicBool;
use std::sync::Arc;
</imports>
<Rust>
#[test]
fn test_pingtype_metadata_access() {
    let ping_type = PingType::new(
        "test_ping",
        true,
        true,
        true,
        true,
        Arc::new(AtomicBool::new(true)),
        Arc::new(AtomicBool::new(true)),
        vec!["ping1".to_string(), "ping2".to_string()],
        vec!["reason1".to_string(), "reason2".to_string()],
    );

    assert_eq!(ping_type.name(), "test_ping");
    assert!(ping_type.include_client_id());
    assert!(ping_type.send_if_empty());
    assert!(ping_type.precise_timestamps());
    assert!(ping_type.include_info_sections());
    assert!(ping_type.naively_enabled());
    assert!(ping_type.follows_collection_enabled());
    assert_eq!(ping_type.schedules_pings(), &["ping1".to_string(), "ping2".to_string()]);
    assert_eq!(ping_type.reason_codes(), &["reason1".to_string(), "reason2".to_string()]);
}
</Rust>
