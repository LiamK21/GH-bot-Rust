<imports>
use malloc_size_of::MallocSizeOf;
use malloc_size_of::MallocSizeOfOps;
use std::ptr;
</imports>
<Rust>
#[test]
fn test_common_metric_data_size() {
  let common_metric_data = super::CommonMetricData {
    name: "test_name".to_string(),
    ..Default::default()
  };
  let expected_size = 0;
  let actual_size = common_metric_data.size_of(&mut MallocSizeOfOps::new(ptr::null_mut(), None));
  assert_eq!(expected_size, actual_size);
}
</Rust>
