<imports> 
use glean_core::common_metric_data::CommonMetricData;
use malloc_size_of::MallocSizeOf;
use std::mem;
</imports>
<Rust>
#[test]
fn test_common_metric_data_size() {
  let common_metric_data = CommonMetricData {
    name: "test_name".to_string(),
    ..Default::default()
  };
  let expected_size = mem::size_of::<String>() + mem::size_of::<Option<String>>() + mem::size_of::<Option<String>>();
  let actual_size = common_metric_data.size_of(&mut malloc_size_of::MallocSizeOfOps::new());
  assert_eq!(expected_size, actual_size);
}
</Rust>
