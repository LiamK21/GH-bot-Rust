<imports> 
use std::collections::HashMap;
use std::num::NonZeroU64;
</imports>
<Rust>
#[test]
fn test_merge_histograms_without_panic() {
  let mut histogram1 = Histogram::new(LinearOrExponential::Linear(NonZeroU64::new(1).unwrap()));
  let mut histogram2 = Histogram::new(LinearOrExponential::Linear(NonZeroU64::new(1).unwrap()));
  histogram1.accumulate(100);
  histogram2.accumulate(u64::MAX);
  let result = std::panic::catch_unwind(|| histogram1.merge(&histogram2));
  assert!(result.is_ok());
}</Rust>
