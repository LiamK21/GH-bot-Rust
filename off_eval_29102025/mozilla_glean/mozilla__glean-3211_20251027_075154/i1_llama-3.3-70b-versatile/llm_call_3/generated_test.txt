#[test]
fn test_merge_histograms_without_panic() {
  let mut histogram1 = Histogram::linear(0, 100, 10);
  let mut histogram2 = Histogram::linear(0, 100, 10);
  histogram1.accumulate(100);
  histogram2.accumulate(u64::MAX);
  let result = catch_unwind(|| histogram1.merge(&histogram2));
  assert!(result.is_ok());
}
