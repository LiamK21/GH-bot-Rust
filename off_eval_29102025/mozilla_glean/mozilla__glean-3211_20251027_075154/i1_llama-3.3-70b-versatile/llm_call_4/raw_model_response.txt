<imports>
use super::Histogram;
use std::num::NonZeroU64;
use std::panic::catch_unwind;
</imports>
<Rust>
#[test]
fn test_merge_histograms_without_panic() {
  let mut histogram1 = Histogram::linear(0, 100, 10);
  let mut histogram2 = Histogram::linear(0, 100, 10);
  histogram1.accumulate(100);
  histogram2.accumulate(u64::MAX);
  let result = catch_unwind(|| {
    let mut temp = histogram1.clone();
    temp.merge(&histogram2);
  });
  assert!(result.is_ok());
}
</Rust>
