<imports>
use super::BooleanMetric;
use glean_core::traits::Boolean;
use std::sync::Arc;
use crate::dispatcher;
use glean_core::metrics::BooleanMetric as CoreBooleanMetric;
</imports>
<Rust>
#[test]
fn test_boolean_metric_set() {
    let core_metric = Arc::new(CoreBooleanMetric::new("category", "name", "send_in_pings", true, true));
    let metric = BooleanMetric(core_metric);
    let expected_value = true;
    metric.set(expected_value);
    let actual_value = metric.test_get_value(None);
    assert_eq!(Some(expected_value), actual_value);
}
</Rust>
