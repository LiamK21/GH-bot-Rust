<imports>
use log::LevelFilter;
use android_logger::Filter;
use android_logger::FilterBuilder;
use super::glean_enable_logging;
</imports>
<Rust>
#[test]
fn test_glean_enable_logging_filters_modules() {
  glean_enable_logging();
  let filter = FilterBuilder::new()
    .filter_module("glean_ffi", LevelFilter::Debug)
    .filter_module("glean_core", LevelFilter::Debug)
    .filter_module("glean", LevelFilter::Debug)
    .build();
  let expected_modules = vec!["glean_ffi", "glean_core", "glean"];
  let actual_modules: Vec<String> = filter.modules().into_iter().collect();
  assert_eq!(expected_modules, actual_modules);
}
</Rust>
