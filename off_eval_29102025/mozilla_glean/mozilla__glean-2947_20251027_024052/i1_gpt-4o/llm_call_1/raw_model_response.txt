<imports>
use super::Builder;
use std::process::Command;
</imports>
<Rust>
#[test]
fn test_builder_generates_correct_rust_code() {
    let builder = Builder {};
    let output = Command::new("glean_parser")
        .arg("--version")
        .output()
        .expect("Failed to execute command");

    let expected_version = "15.0.1";
    let actual_version = String::from_utf8_lossy(&output.stdout).trim();

    assert_eq!(expected_version, actual_version);
}
