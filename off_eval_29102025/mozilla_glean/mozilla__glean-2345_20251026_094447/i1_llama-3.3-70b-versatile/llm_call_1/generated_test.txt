#[test]
fn test_config_option() {
  let mut args = vec!["--config".to_string(), "test_config.json".to_string()];
  let result_before = std::panic::catch_unwind(|| main());
  assert!(result_before.is_err());

  // Apply the patch
  let mut args_after = vec!["--config".to_string(), "test_config.json".to_string()];
  let result_after = std::panic::catch_unwind(|| main());
  assert!(result_after.is_ok());
}
