<imports>
use crate::metrics::BooleanMetric;
use crate::metrics::CommonMetricData;
use std::sync::Arc;
</imports>
<Rust>
#[test]
fn test_boolean_metric_test_get_value() {
  let common_metric_data = CommonMetricData::new("test_metric", "test_category", "test_description", None, None);
  let boolean_metric = BooleanMetric::new(common_metric_data);
  let expected_value = Some(true);
  boolean_metric.set(true);
  let actual_value = boolean_metric.test_get_value(None);
  assert_eq!(actual_value, expected_value);
}
