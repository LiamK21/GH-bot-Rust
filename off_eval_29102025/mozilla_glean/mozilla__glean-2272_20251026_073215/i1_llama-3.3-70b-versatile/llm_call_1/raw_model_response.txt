<imports> 
use std::borrow::Cow;
use super::LabeledMetric;
use std::collections::HashMap;
</imports>
<Rust>
#[test]
fn test_labeled_metric_creation() {
  let meta = CommonMetricData { ..Default::default() };
  let labels = Some(vec![Cow::from("status_code_4xx"), Cow::from("status_code_5xx")]);
  let labeled_metric: LabeledMetric<u64> = LabeledMetric::new(meta, labels);
  let expected_labels = Some(vec![Cow::from("status_code_4xx"), Cow::from("status_code_5xx")]);
  assert_eq!(labeled_metric.labels, expected_labels);
}</Rust>
