<imports>
use std::thread;
use std::thread::Builder;
use std::thread::Result;
</imports>
<Rust>
#[test]
fn test_thread_name() {
  let handle_before_patch = thread::spawn(|| {});
  let handle_after_patch = Builder::new().name("glean.dispatcher".into()).spawn(|| {});
  assert_ne!(handle_before_patch.thread().unwrap().name(), Some("glean.dispatcher"));
  assert_eq!(handle_after_patch.thread().unwrap().name(), Some("glean.dispatcher"));
}
</Rust>
