<imports>
use std::thread;
use std::thread::Builder;
use std::thread::Result;
use std::thread::Thread;
use std::thread::JoinHandle;
</imports>
<Rust>
#[test]
fn test_thread_name() {
  let handle_before_patch: JoinHandle<()> = thread::spawn(|| {});
  let handle_after_patch: JoinHandle<()> = Builder::new().name("glean.dispatcher".into()).spawn(|| {});
  let handle_before_patch_thread_name = handle_before_patch.thread().map(|t| t.name());
  let handle_after_patch_thread_name = handle_after_patch.thread().map(|t| t.name());
  assert_ne!(handle_before_patch_thread_name, Some("glean.dispatcher"));
  assert_eq!(handle_after_patch_thread_name, Some("glean.dispatcher"));
}
</Rust>
