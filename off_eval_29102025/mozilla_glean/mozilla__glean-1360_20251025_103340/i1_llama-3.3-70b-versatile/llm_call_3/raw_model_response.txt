<imports>
use std::thread;
use std::thread::Builder;
use std::thread::Result;
use std::thread::Thread;
</imports>
<Rust>
#[test]
fn test_thread_name() {
  let handle_before_patch = thread::spawn(|| {});
  let handle_after_patch = Builder::new().name("glean.dispatcher".into()).spawn(|| {});
  let handle_before_patch_thread: Thread = handle_before_patch.thread().unwrap();
  let handle_after_patch_thread: Thread = handle_after_patch.thread().unwrap();
  assert_ne!(handle_before_patch_thread.name(), Some("glean.dispatcher"));
  assert_eq!(handle_after_patch_thread.name(), Some("glean.dispatcher"));
}
</Rust>
