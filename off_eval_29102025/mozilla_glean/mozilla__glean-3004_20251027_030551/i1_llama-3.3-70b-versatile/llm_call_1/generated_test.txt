#[test]
fn test_debug_view_tag_and_log_pings() {
  let tag = "test_tag";
  let _ = set_debug_view_tag(tag);
  let actual_tag = glean_get_debug_view_tag();
  let expected_tag = Some(tag.to_string());
  assert_eq!(actual_tag, expected_tag);

  let log_pings_value = true;
  let _ = set_log_pings(log_pings_value);
  let actual_log_pings = glean_get_log_pings();
  let expected_log_pings = log_pings_value;
  assert_eq!(actual_log_pings, expected_log_pings);

  let actual_ping_names = get_registered_ping_names();
  let expected_ping_names: Vec<String> = vec![];
  assert!(actual_ping_names.len() >= expected_ping_names.len());
}
