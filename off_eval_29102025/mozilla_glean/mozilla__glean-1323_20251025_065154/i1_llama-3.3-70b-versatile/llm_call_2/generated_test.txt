#[test]
fn test_counter_metric_recording() {
  let _lock = std::sync::Mutex::new(()).lock().unwrap();
  destroy_glean(true);
  let metric = CounterMetric::new(glean_core::CommonMetricData {
    name: "counter_metric".into(),
    category: "test".into(),
    send_in_pings: vec!["store1".into()],
    lifetime: glean_core::Lifetime::Application,
    disabled: false,
    dynamic_label: None,
  });
  metric.add(1);
  let _ = new_glean(None, false);
  assert_eq!(1, metric.test_get_value(None).unwrap());
}
