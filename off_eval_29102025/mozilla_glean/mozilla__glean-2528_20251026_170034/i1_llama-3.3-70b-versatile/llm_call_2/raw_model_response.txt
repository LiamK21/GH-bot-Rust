<imports>
use super::configure_ping_throttling;
use crate::PingRateLimit;
</imports>
<Rust>
#[test]
fn test_configure_ping_throttling_rate_limit() {
  configure_ping_throttling();
  let expected_rate_limit = PingRateLimit {
    seconds_per_interval: 1,
    pings_per_interval: 10,
  };
  let actual_rate_limit = PingRateLimit {
    seconds_per_interval: 1,
    pings_per_interval: 10,
  };
  assert_eq!(expected_rate_limit, actual_rate_limit);
}
</Rust>
