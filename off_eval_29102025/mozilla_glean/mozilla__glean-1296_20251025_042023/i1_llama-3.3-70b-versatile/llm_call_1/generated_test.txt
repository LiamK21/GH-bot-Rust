#[test]
fn test_database_delete_error_logging() {
  let mut db = Database::new();
  let error = io::Error::new(io::ErrorKind::Other, "Test error");
  let expected_log_level = Level::Warn;
  let actual_log_level = log::max_level();
  log::set_max_level(expected_log_level);
  db.delete("test_metric", error);
  assert_eq!(actual_log_level, expected_log_level);
}
