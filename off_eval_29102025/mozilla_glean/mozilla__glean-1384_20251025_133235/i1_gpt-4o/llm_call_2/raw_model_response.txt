<imports>
use super::DatetimeMetric;
use crate::common_test::lock_test;
use crate::common_test::new_glean;
use glean_core::metrics::TimeUnit;
use glean_core::ErrorType;
use chrono::prelude::*;
use crate::CommonMetricData;
</imports>
<Rust>
#[test]
fn test_datetime_metric_set_and_get() {
    let _lock = lock_test();
    let _t = new_glean(None, true);

    let metric: DatetimeMetric = DatetimeMetric::new(
        CommonMetricData {
            name: "datetime".into(),
            category: "test".into(),
            send_in_pings: vec!["test1".into()],
            ..Default::default()
        },
        TimeUnit::Day,
    );

    let sample_date = FixedOffset::east(0).ymd(2023, 10, 5).and_hms(14, 30, 0);
    metric.set(Some(sample_date));

    let expected_date = FixedOffset::east(0).ymd(2023, 10, 5).and_hms(0, 0, 0);
    let actual_date = metric.test_get_value(None).unwrap();
    assert_eq!(actual_date, expected_date);

    assert_eq!(
        metric.test_get_num_recorded_errors(ErrorType::InvalidValue, None),
        0
    );
}
</Rust>
