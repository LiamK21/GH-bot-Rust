use std::fs;
use remove_dir_all::remove_dir_all;
use std::io;

<Rust>
#[test]
fn test_remove_dir_all() {
  let temp_dir = Path::new("temp_dir");
  fs::create_dir_all(temp_dir).unwrap();
  let _file = fs::File::create(temp_dir.join("file.txt")).unwrap();
  let result = std::panic::catch_unwind(|| remove_dir_all::remove_dir_all(temp_dir));
  assert!(result.is_ok());
  assert!(!temp_dir.exists());
}</Rust>
