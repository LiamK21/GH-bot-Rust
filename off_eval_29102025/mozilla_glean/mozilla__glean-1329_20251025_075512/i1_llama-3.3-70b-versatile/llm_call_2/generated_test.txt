#[test]
fn test_uuid_metric_set_and_get() {
  let metric = UuidMetric::new(glean_core::CommonMetricData::new("test_metric", "test_description", "test_ping"));
  let uuid = Uuid::new_v4();
  metric.set(uuid);
  let actual = metric.test_get_value(None);
  let expected = Some(uuid);
  assert_eq!(actual, expected);
}
