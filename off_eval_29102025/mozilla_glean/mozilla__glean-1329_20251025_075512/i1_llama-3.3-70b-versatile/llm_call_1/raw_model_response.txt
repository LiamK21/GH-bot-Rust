<imports> 
use glean_core::metrics::UuidMetric;
use glean_core::traits::Uuid;
use uuid::Uuid;
use std::sync::Arc;
</imports>
<Rust>
#[test]
fn test_uuid_metric_set_and_get() {
  let metric = UuidMetric::new(glean_core::CommonMetricData::new("test_metric", "test_description"));
  let uuid = Uuid::new_v4();
  metric.set(uuid);
  let actual = metric.test_get_value(None);
  assert_eq!(actual, Some(uuid));
}</Rust>
