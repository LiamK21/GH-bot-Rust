#[test]
fn test_boolean_get_num_recorded_errors_without_ping() {
    let metric = BooleanMetric::new(CommonMetricData {
        category: "".into(),
        name: "test".into(),
        send_in_pings: vec!["default".into()],
        ..Default::default()
    });

    let result = metric.test_get_num_recorded_errors(ErrorType::InvalidState);

    assert_eq!(result, 0);
}
