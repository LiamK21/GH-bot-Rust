#[test]
fn test_test_get_num_recorded_errors() {
  let glean = Glean::new();
  let meta = CommonMetricData {
    base_identifier: "test_metric".to_string(),
    send_in_pings: vec!["metrics".to_string()],
    ..Default::default()
  };
  let error = ErrorType::InvalidValue;
  let result = test_get_num_recorded_errors(&glean, &meta, error);
  assert_eq!(result.unwrap_or(0), 0);
}
