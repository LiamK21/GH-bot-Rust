<imports>
use super::test_get_num_recorded_errors;
use crate::metrics::CommonMetricData;
use crate::Glean;
use crate::ErrorType;
</imports>
<Rust>
#[test]
fn test_get_num_recorded_errors_assumes_metrics_store() {
    let glean = Glean::new();
    let meta = CommonMetricData::new("test_category", "test_name", "test_send_in_pings", false, false, None, None, None);
    let error = ErrorType::InvalidValue;

    let expected = Err("No error recorded for test_category.test_name in 'metrics' store".to_string());
    let actual = test_get_num_recorded_errors(&glean, &meta, error);

    assert_eq!(expected, actual);
}</Rust>
