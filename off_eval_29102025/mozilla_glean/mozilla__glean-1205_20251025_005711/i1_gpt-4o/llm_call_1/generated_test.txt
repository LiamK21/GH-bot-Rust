#[test]
fn test_glean_initialize_for_subprocess_upload_disabled() {
    let config = Configuration {
        data_path: "test_path".into(),
        application_id: "test_app".into(),
        upload_enabled: false,
        language_binding_name: "test_binding".into(),
    };
    let config_ptr = &config as *const Configuration as *const FfiConfiguration;
    let result = unsafe { glean_initialize_for_subprocess(config_ptr) };
    assert_eq!(result, 1);
}
