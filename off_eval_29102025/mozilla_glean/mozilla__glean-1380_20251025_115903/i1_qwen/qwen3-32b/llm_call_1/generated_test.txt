#[test]
fn test_send_in_pings_not_set_records_invalid_state_error() {
    let _lock = lock_test();
    let _glean = new_glean(None, true);

    let metric = StringListMetric::new(CommonMetricData {
        name: "event".into(),
        category: "test".into(),
        send_in_pings: vec![],

    });

    metric.add("invalid");

    assert_eq!(

        metric.test_get_num_recorded_errors(ErrorType::InvalidState, None)
    );
}
