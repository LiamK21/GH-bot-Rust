<imports>
use std::collections::HashMap;
use super::Timings;
use super::TimerId;
</imports>
<Rust>
#[test]
fn test_multiple_timers_can_run_simultaneously() {
    let mut timings = Timings {
        next_id: TimerId(0),
        start_times: HashMap::new(),
    };

    let start_time_1 = 100;
    let start_time_2 = 200;

    let timer_id_1 = timings.set_start(start_time_1);
    let timer_id_2 = timings.set_start(start_time_2);

    assert!(timings.start_times.contains_key(&timer_id_1));
    assert!(timings.start_times.contains_key(&timer_id_2));
    assert_ne!(timer_id_1, timer_id_2);
}
</Rust>
