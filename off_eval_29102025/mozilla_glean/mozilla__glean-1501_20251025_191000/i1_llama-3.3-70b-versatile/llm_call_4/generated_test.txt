#[test]
fn test_multiple_timers() {
  let mut timings = Timings::new();
  let timer_id1 = set_start(&mut timings, 1);
  let timer_id2 = set_start(&mut timings, 2);
  let timer_id3 = set_start(&mut timings, 3);
  assert_ne!(timer_id1, timer_id2);
  assert_ne!(timer_id2, timer_id3);
  assert_ne!(timer_id1, timer_id3);
  let stopped_time1 = stop_and_accumulate(&mut timings, timer_id1);
  let stopped_time2 = stop_and_accumulate(&mut timings, timer_id2);
  let stopped_time3 = stop_and_accumulate(&mut timings, timer_id3);
  assert_eq!(stopped_time1, 1);
  assert_eq!(stopped_time2, 2);
  assert_eq!(stopped_time3, 3);
}
