<imports>
use crate::metrics::timing_distribution::Timings;
use crate::metrics::timing_distribution::TimerId;
</imports>
<Rust>
#[test]
fn test_multiple_timers() {
  let mut timings = Timings::new();
  let timer_id1 = timings.set_start(1);
  let timer_id2 = timings.set_start(2);
  let timer_id3 = timings.set_start(3);
  assert_ne!(timer_id1, timer_id2);
  assert_ne!(timer_id2, timer_id3);
  assert_ne!(timer_id1, timer_id3);
  let stopped_time1 = timings.stop_and_accumulate(timer_id1);
  let stopped_time2 = timings.stop_and_accumulate(timer_id2);
  let stopped_time3 = timings.stop_and_accumulate(timer_id3);
  assert_eq!(stopped_time1, 1);
  assert_eq!(stopped_time2, 2);
  assert_eq!(stopped_time3, 3);
}
</Rust>
