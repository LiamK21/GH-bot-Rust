<imports>
use chrono::Local;
use super::local_now_with_offset;
</imports>
<Rust>
#[test]
fn test_local_now_with_offset_correct_timezone() {
    let now = Local::now();
    let fixed_now = local_now_with_offset();
    assert_eq!(now.with_timezone(now.offset()).timestamp(), fixed_now.timestamp());
}
</Rust>
