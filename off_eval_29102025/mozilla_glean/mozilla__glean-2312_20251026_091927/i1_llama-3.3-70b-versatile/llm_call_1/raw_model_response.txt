<imports> 
use super::glean_enable_logging;
use log::LevelFilter;
use std::env;
</imports>
<Rust>
#[test]
fn test_glean_enable_logging() {
  let original_log = env::var("RUST_LOG");
  env::set_var("RUST_LOG", "debug");
  glean_enable_logging();
  let log_level = log::max_level();
  assert_eq!(log_level, LevelFilter::Debug);
  env::set_var("RUST_LOG", "off");
  if let Ok(val) = original_log {
    env::set_var("RUST_LOG", val);
  } else {
    env::remove_var("RUST_LOG");
  }
}</Rust>
