<imports>
use super::sending_of_foreground_background_pings;
use super::sending_deletion_ping_if_disabled_outside_of_run;
use std::time::Duration;
use std::thread;
use std::sync::mpsc::channel;
</imports>
<Rust>
#[test]
fn test_sending_of_foreground_background_pings_timeout() {
    let (tx, rx) = channel();
    let handle = thread::spawn(move || {
        sending_of_foreground_background_pings();
        tx.send(()).unwrap();
    });
    let result = rx.recv_timeout(Duration::from_secs(10));
    assert!(result.is_err(), "Test should timeout before the patch");
    handle.join().unwrap();
}
