<imports> 
use super::PingUploadTask;
use std::thread;
use std::time::Duration;

<Rust>
#[test]
fn test_wait_task_contains_expected_wait_time_when_rate_limited() {
  let wait_time = 1000;
  let task = PingUploadTask::Wait(wait_time);
  let expected_time = wait_time;
  let actual_time = match task {
    PingUploadTask::Wait(time) => time,
    _ => panic!("Expected upload manager to return a wait task!"),
  };
  assert_eq!(expected_time, actual_time);
}</Rust>
