use std::collections::HashMap;
use super::Histogram;
use super::PrecomputedExponential;
use super::PrecomputedLinear;

<Rust>
#[test]
fn test_snapshot_values_without_zero_count_buckets() {
  let bucketing = PrecomputedExponential { min: 1, max: 100, bucket_count: 10 };
  let mut values = HashMap::new();
  values.insert(10, 5);
  values.insert(50, 3);
  let histogram = Histogram { values, bucketing, count: 8, sum: 53 };
  let actual = histogram.snapshot_values();
  let expected = HashMap::from([(10, 5), (50, 3)]);
  assert_eq!(actual, expected);
}
</Rust>
