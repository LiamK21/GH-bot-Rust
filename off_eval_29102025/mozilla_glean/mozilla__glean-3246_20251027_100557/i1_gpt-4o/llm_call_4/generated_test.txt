#[test]
fn test_snapshot_values_no_infill_zero_buckets() {
    let mut histogram = Histogram::linear(0, 40, 5);
    histogram.accumulate(15);
    histogram.accumulate(25);

    let expected: HashMap<u64, u64> = [(0, 0), (10, 1), (20, 1), (30, 0), (40, 0)].iter().cloned().collect();
    let actual = histogram.snapshot_values();

    assert_eq!(expected, actual);
}
