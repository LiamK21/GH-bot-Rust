<imports>
use super::DualLabeledCounterMetric;
use crate::metrics::MetricType;
use crate::metrics::CommonMetricDataInternal;
use std::borrow::Cow;
use crate::metrics::CommonMetricData;
</imports>
<Rust>
#[test]
fn test_dual_labeled_counter_metric_implements_metric_type() {
    let meta = CommonMetricData::new("test_category", "test_name", "test_send_in_pings", false, None, None, None);
    let keys = Some(vec![Cow::Borrowed("key1"), Cow::Borrowed("key2")]);
    let categories = Some(vec![Cow::Borrowed("category1"), Cow::Borrowed("category2")]);
    let metric = DualLabeledCounterMetric::new(meta, keys, categories);
    let expected_meta: &CommonMetricDataInternal = metric.counter.meta();
    let actual_meta = metric.meta();
    assert_eq!(expected_meta, actual_meta);
}
</Rust>
