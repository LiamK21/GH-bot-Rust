#[test]
fn test_dual_labeled_counter_metric_type() {
  let counter = super::CounterMetric::new();
  let metric = DualLabeledCounterMetric { 
    counter: counter, 
    categories: Default::default(), 
    dual_label_map: Default::default(), 
    keys: Default::default() 
  };
  let expected: &dyn MetricType = &metric;
  let actual: &dyn MetricType = &metric;
  assert!(std::any::type_name::<dyn MetricType>() == std::any::type_name::<&DualLabeledCounterMetric>());
  assert_eq!(metric.meta().disabled.load(std::sync::atomic::Ordering::SeqCst), 0);
}
