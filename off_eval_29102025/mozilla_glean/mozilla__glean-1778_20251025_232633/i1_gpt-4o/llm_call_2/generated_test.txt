#[test]
fn test_correct_order_with_url() {
    let metric = Metric::Url("https://example.com".to_string());
    let expected_disc = 15;
    let actual_disc = match metric {
        Metric::Url(_) => 15,
        _ => panic!("Unexpected metric type"),
    };
    assert_eq!(expected_disc, actual_disc);
}
