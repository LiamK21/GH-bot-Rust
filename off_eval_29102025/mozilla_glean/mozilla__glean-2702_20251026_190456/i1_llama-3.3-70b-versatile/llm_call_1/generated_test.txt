#[test]
fn test_dynamic_experimentation_id() {
  let initial_id = "initial_id".to_string();
  let new_id = "new_id".to_string();
  set_experimentation_id(initial_id.clone());
  let actual_initial_id = test_get_experimentation_id().unwrap();
  assert_eq!(actual_initial_id, initial_id);
  set_experimentation_id(new_id.clone());
  let actual_new_id = test_get_experimentation_id().unwrap();
  assert_eq!(actual_new_id, new_id);
}
