#[test]
fn test_event_max_events_overflow_triggers_upload() {
    let cfg = InternalConfiguration::default();
    let glean = Glean::new(cfg).unwrap();
    let meta = CommonMetricData::default();
    let allowed_extra_keys = Vec::new();
    let event_metric = EventMetric::new(meta, allowed_extra_keys);
    let timestamp = 123456789;
    let extra = HashMap::new();

    let expected = true;

    let actual = event_metric.record_sync(&glean, timestamp, extra);

    assert_eq!(expected, actual);
}
