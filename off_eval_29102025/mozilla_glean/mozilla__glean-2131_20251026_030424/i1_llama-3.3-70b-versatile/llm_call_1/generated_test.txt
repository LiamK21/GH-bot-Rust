#[test]
fn test_initialize_core_metrics_sets_client_info_fields() {
  let mut glean = Glean::new();
  let client_info = ClientInfoMetrics {
    android_sdk_version: Some("29".to_string()),
    device_manufacturer: Some("Google".to_string()),
    device_model: Some("Pixel 4".to_string()),
    locale: Some("en-US".to_string()),
  };
  initialize_core_metrics(&mut glean, &client_info);
  assert_eq!(glean.get_string("android_sdk_version"), "29");
  assert_eq!(glean.get_string("device_manufacturer"), "Google");
  assert_eq!(glean.get_string("device_model"), "Pixel 4");
  assert_eq!(glean.get_string("locale"), "en-US");
}
