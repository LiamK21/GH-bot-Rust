#[test]
fn test_uploader_capabilities_instrumentation() {
    let ping = Ping::new("test_ping", true, true, vec!["ohttp".to_string()]);
    let ping_maker = PingMaker;

    let expected = json!({
        "string_list": {
            "glean.ping.uploader_capabilities": ["ohttp"]
        }
    });

    let actual = ping_maker.make_metrics_data(&ping);

    assert_eq!(expected, actual);
}
