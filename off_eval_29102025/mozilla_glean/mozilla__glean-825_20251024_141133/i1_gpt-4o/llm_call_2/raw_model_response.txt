<imports>
use std::collections::VecDeque;
use std::sync::{Arc, RwLock};
use super::PingUploadManager;
use serde_json::json;
</imports>
<Rust>
#[test]
fn test_enqueue_ping_logs_correctly() {
    let manager = PingUploadManager {
        queue: Arc::new(RwLock::new(VecDeque::new())),
        directory_manager: None,
        processed_pending_pings: Arc::new(RwLock::new(VecDeque::new())),
    };
    let uuid = "test-uuid";
    let path = "/test/path";
    let body = json!({"test": "data"});

    manager.enqueue_ping(uuid, path, body);

    let queue = manager.queue.read().unwrap();
    assert_eq!(queue.len(), 1);
    assert_eq!(queue.front().unwrap().uuid, uuid);
    assert_eq!(queue.front().unwrap().path, path);
}
</Rust>
