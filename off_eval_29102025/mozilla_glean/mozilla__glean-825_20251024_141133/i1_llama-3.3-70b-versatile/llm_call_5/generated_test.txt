#[test]
fn test_deletion_request_ping_queue_clearing() {
  let ping_upload_manager = PingUploadManager::new(PathBuf::from("test"));
  let ping_request1 = PingRequest { uuid: "1".to_string(), path: "path1".to_string(), body: json!({}), headers: HashMap::new() };
  let ping_request2 = PingRequest { uuid: "2".to_string(), path: "path2".to_string(), body: json!({}), headers: HashMap::new() };
  let ping_request3 = PingRequest { uuid: "3".to_string(), path: "path3".to_string(), body: json!({}), headers: HashMap::new() };
  ping_upload_manager.enqueue_ping("1", "path1", json!({}));
  ping_upload_manager.enqueue_ping("2", "path2", json!({}));
  ping_upload_manager.enqueue_ping("3", "path3", json!({}));
  let expected_queue_len = 3;
  let actual_queue_len = ping_upload_manager.clear_ping_queue().len();
  assert_eq!(expected_queue_len, actual_queue_len);
}
