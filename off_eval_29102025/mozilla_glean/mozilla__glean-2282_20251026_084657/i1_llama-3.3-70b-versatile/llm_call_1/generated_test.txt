#[test]
fn test_set_metrics_disabled_config() {
  let json = json!({
    "category.string_metric": true
  }).to_string();
  let config = MetricsDisabledConfig::try_from(json).unwrap();
  glean_set_metrics_disabled_config(config.to_string());
  // Test that the config was set correctly
  let new_config = MetricsDisabledConfig::try_from(json).unwrap();
  assert_eq!(new_config.metrics_disabled, MetricsDisabledConfig::try_from(json).unwrap().metrics_disabled);
}
