<imports> 
use std::path::Path;
use std::fs;
use super::rkv_new;
use rkv::StoreError;
</imports>
<Rust>
#[test]
fn test_rkv_new_logging() {
  let path = Path::new("/tmp/test_rkv");
  fs::create_dir_all(path).unwrap();
  let safebin = path.join("data.safe.bin");
  fs::File::create(safebin).unwrap();
  let result = rkv_new(path);
  match result {
    Err(rkv::StoreError::DatabaseCorrupted) => assert!(true),
    _ => assert!(false),
  }
  fs::remove_dir_all(path).unwrap();
}</Rust>
