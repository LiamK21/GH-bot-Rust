<imports>
use serde_json;
use super::CommonMetricData;
use super::Lifetime;
use std::fmt::Debug;
</imports>
<Rust>
#[test]
fn test_common_metric_data_deserialization() {
  let json = r#"{"name": "test_metric", "category": "", "disabled": false, "dynamic_label": null, "send_in_pings": [], "lifetime": "ping"}"#;
  let expected: CommonMetricData = CommonMetricData { 
    name: "test_metric".to_string(), 
    category: "".to_string(), 
    disabled: false, 
    dynamic_label: None, 
    send_in_pings: vec![], 
    lifetime: Lifetime::Ping, 
  };
  let actual: Result<CommonMetricData, _> = serde_json::from_str(json);
  assert!(actual.is_ok());
  let actual = actual.unwrap();
  assert_eq!(expected.name, actual.name);
  assert_eq!(expected.category, actual.category);
  assert_eq!(expected.disabled, actual.disabled);
  assert_eq!(expected.dynamic_label, actual.dynamic_label);
  assert_eq!(expected.send_in_pings, actual.send_in_pings);
  assert_eq!(expected.lifetime, actual.lifetime);
}
</Rust>
