<imports> 
use serde_json;
use super::CommonMetricData;
use super::Lifetime;
</imports>
<Rust>
#[test]
fn test_common_metric_data_deserialization() {
  let json = r#"{"name": "test_metric"}"#;
  let expected: CommonMetricData = CommonMetricData { name: "test_metric".to_string() };
  let actual: CommonMetricData = serde_json::from_str(json).unwrap();
  assert_eq!(expected, actual);
}
</Rust>
