<imports>
use std::env;
use anyhow::Result;
use uniffi::TargetLanguage;
use super::parse_language;
</imports>
<Rust>
#[test]
fn test_parse_language() {
  let lang = "kotlin";
  let expected = Ok(TargetLanguage::Kotlin);
  let actual = parse_language(lang);
  assert_eq!(expected.as_ref().map(|x| format!("{:?}", x)), actual.as_ref().map(|x| format!("{:?}", x)));
}
</Rust>
