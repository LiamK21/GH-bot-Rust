<imports>
use super::timing_distribution_truncation;
use std::collections::HashSet;
use std::iter::FromIterator;
</imports>
<Rust>
#[test]
fn test_timing_distribution_truncation() {
  let hist = timing_distribution_truncation();
  let expected_keys: HashSet<u64> = HashSet::from([1, 2, 3, 4, 5]);
  let actual_keys: HashSet<u64> = HashSet::from_iter(hist.values().keys().cloned());
  assert_eq!(expected_keys, actual_keys);
}
</Rust>
