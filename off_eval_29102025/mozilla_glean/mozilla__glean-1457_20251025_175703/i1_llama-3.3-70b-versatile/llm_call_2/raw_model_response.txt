<imports>
use super::timing_distribution_truncation;
use std::collections::HashSet;
</imports>
<Rust>
#[test]
fn test_timing_distribution_truncation() {
  let mut hist = super::timing_distribution_truncation();
  let expected_keys: HashSet<u64> = HashSet::from([1, 2, 3, 4, 5]);
  let actual_keys: HashSet<u64> = hist.values().keys().cloned().collect();
  assert_ne!(expected_keys, actual_keys);
}
</Rust>
