#[test]
fn test_glean_initialize() {
  let cfg = FfiConfiguration {
    debug_logger: None,
    data_path: CString::new("/tmp/data").unwrap().into_raw(),
    max_pings: 10,
  };
  let result = unsafe { super::glean_initialize(&cfg as *const _ as *const FfiConfiguration) };
  assert_eq!(result, 0);
}
