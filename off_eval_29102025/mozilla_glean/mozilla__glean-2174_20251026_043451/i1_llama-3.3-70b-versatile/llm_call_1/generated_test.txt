#[test]
fn test_initialize_core_metrics_os_version() {
  let mut glean = Glean::new();
  let client_info = ClientInfoMetrics {
    os_version: "Test OS Version".to_string(),
    architecture: "Test Architecture".to_string(),
    ..Default::default()
  };
  initialize_core_metrics(&mut glean, &client_info);
  let expected_os_version = "Test OS Version";
  let actual_os_version = glean.get_string("os_version").unwrap();
  assert_eq!(expected_os_version, actual_os_version);
}
