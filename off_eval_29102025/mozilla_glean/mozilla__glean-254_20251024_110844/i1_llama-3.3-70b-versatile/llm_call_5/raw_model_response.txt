#[test]
fn test_experiment_metric_limits_violations() {
  let config = Configuration { ..Default::default() };
  let glean = Glean::new(config).unwrap();
  let experiment_id = "a".repeat(31);
  let branch = "b".repeat(31);
  let extra = HashMap::from([("key".to_string(), "value".to_string())]);
  let metric = ExperimentMetric::new(&glean, experiment_id);
  metric.set_active(&glean, branch, Some(extra.clone()));
  let metric_before_patch = ExperimentMetric::new(&glean, experiment_id);
  metric_before_patch.set_active(&glean, branch, Some(extra));
  info!("Test passed if no errors are logged for the metric after the patch");
}
