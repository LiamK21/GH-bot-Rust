<imports>
use super::ExperimentMetric;
use crate::Glean;
use crate::storage::StorageManager;
use std::collections::HashMap;
</imports>
<Rust>
#[test]
fn test_experiment_metric_error_reporting() {
    let glean = Glean::new("test_app_id", "test_app_version", "test_channel", "test_build_id", None);
    let extra: HashMap<String, String> = [("key".repeat(31), "value".repeat(31))]
        .iter()
        .cloned()
        .collect();

    let metric = ExperimentMetric::new(&glean, "some-experiment".repeat(31));

    metric.set_active(&glean, "test-branch".repeat(31), Some(extra));
    let snapshot = StorageManager::snapshot_as_json(&glean, "experiments", false, "test");

    assert!(snapshot.contains("Value length 372 for experiment id exceeds maximum of 30"));
    assert!(snapshot.contains("Value length 341 for branch exceeds maximum of 30"));
    assert!(snapshot.contains("Extra key length 93 exceeds maximum of 30"));
    assert!(snapshot.contains("Extra value length 155 exceeds maximum of 100"));
}</Rust>
