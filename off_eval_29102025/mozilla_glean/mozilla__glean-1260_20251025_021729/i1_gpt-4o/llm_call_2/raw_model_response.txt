<imports>
use std::fs;
use std::path::Path;
use tempfile::tempdir;
use super::Database;
use super::rkv_new;
</imports>
<Rust>
#[test]
fn test_database_initialization_with_lmdb_backend() {
    let temp_dir = tempdir().unwrap();
    let db_path = temp_dir.path().join("db");
    fs::create_dir_all(&db_path).unwrap();

    let result = rkv_new(&db_path);

    assert!(result.is_ok());
    let rkv = result.unwrap();
    assert!(rkv.open_single("test_store", rkv::StoreOptions::create()).is_ok());
}
</Rust>
