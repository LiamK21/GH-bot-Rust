#[test]
fn test_open_rkv_with_latest_rkv_version() {
  let path = Path::new("test_db");
  fs::create_dir_all(&path).unwrap();
  let result_after_patch = Database::open_rkv(&path);
  assert!(result_after_patch.is_ok());
  let rkv_after_patch = result_after_patch.unwrap();
  let type_name_after_patch = std::any::type_name::<rkv::Rkv<rkv::backend::LmdbEnvironment>>();
  let type_name_rkv = std::any::type_name::<_>();
  assert_eq!(type_name_after_patch, type_name_rkv);
}
