#[test]
fn test_join_init_before_get_attribution() {
  glean_init();
  let handle = thread::spawn(move || {
    glean_test_get_attribution();
  });
  handle.join().unwrap();
  let handle2 = thread::spawn(move || {
    glean_test_get_distribution();
  });
  handle2.join().unwrap();
  assert!(true);
}
