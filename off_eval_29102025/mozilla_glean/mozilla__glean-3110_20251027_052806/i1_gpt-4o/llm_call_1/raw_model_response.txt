<imports>
use super::glean_test_get_attribution;
use super::glean_test_get_distribution;
use super::glean_update_attribution;
use super::glean_update_distribution;
use crate::AttributionMetrics;
use crate::DistributionMetrics;
</imports>
<Rust>
#[test]
fn test_get_attribution_and_distribution_after_init() {
    let attribution = AttributionMetrics::default();
    let distribution = DistributionMetrics::default();
    
    glean_update_attribution(attribution.clone());
    glean_update_distribution(distribution.clone());
    
    let expected_attribution = attribution;
    let expected_distribution = distribution;
    
    let actual_attribution = glean_test_get_attribution();
    let actual_distribution = glean_test_get_distribution();
    
    assert_eq!(expected_attribution, actual_attribution);
    assert_eq!(expected_distribution, actual_distribution);
}
</Rust>
