<imports>
use super::rkv_new;
use std::path::Path;
use std::fs;
use tempfile::tempdir;
</imports>
<Rust>
#[test]
fn test_rkv_new_initializes_database() {
    let dir = tempdir().unwrap();
    let database_dir = dir.path().join("db");
    fs::create_dir_all(&database_dir).expect("create dir");

    let result = rkv_new(&database_dir);
    assert!(result.is_ok(), "Database should initialize successfully");
}
</Rust>
