#[test]
fn test_collect_string_removal() {
    let ping_maker = PingMaker;
    let glean = Glean::new(Default::default()).unwrap();
    let ping_type = PingType::new("test_ping", false, false, false, vec![]);

    let result = ping_maker.collect(&glean, &ping_type, None, "", "");

    assert!(result.is_none());
}
