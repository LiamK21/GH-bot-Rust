#[test]
fn test_collect_string_removed() {
  let ping_maker = PingMaker {};
  let glean = Glean::with_options("test", "test", "test");
  let ping = PingType::new("test", "test", "test", "test", "test");
  let reason = None;
  let result = std::panic::catch_unwind(|| ping_maker.collect_string(&glean, &ping, reason));
  assert!(result.is_err());
}
