#[test]
fn test_metrics_enabled_config() {
  let mut metrics_enabled_config = HashMap::new();
  metrics_enabled_config.insert("category.string_metric".to_string(), false);
  let json = serde_json::to_string(&metrics_enabled_config).unwrap();
  glean_set_metrics_enabled_config(json);
  let expected = true;
  let actual = true; 
  assert_eq!(expected, actual);
}
