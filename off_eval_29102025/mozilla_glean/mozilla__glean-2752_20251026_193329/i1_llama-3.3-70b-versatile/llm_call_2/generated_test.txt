#[test]
fn test_test_get_value_returns_json_value() {
  let metric = ObjectMetric::new(glean_core::CommonMetricData::new("test_metric", "test_description", "test_category", "test_type", "test_lifetime", "test_send_in_pings"));
  let expected = json!({"key": "value"});
  metric.set(expected.clone());
  let actual = metric.test_get_value(None).unwrap();
  assert_eq!(expected, actual);
}
