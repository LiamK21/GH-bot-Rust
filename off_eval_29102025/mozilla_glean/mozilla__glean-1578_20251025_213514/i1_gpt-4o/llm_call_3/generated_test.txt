#[test]
fn test_set_raw() {
    let metric = TimespanMetric(Arc::new(RwLock::new(Default::default())));
    let expected_duration = Duration::new(5, 0);

    metric.set_raw(expected_duration);

    let actual_duration = metric.test_get_value(None);
    assert_eq!(actual_duration, Some(expected_duration.as_secs()));
}
