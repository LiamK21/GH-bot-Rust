#[test]
fn test_set_raw_timespan() {
  let timespan = super::TimespanMetric::new(glean_core::CommonMetricData::new("test_timespan", glean_core::metrics::MetricType::Timespan), TimeUnit::Milliseconds);
  let elapsed = Duration::from_millis(100);
  timespan.set_raw(elapsed);
  let actual = timespan.test_get_value(None);
  let expected = Some(100);
  assert_eq!(actual, expected);
}
