<imports>
use glean_core::traits::Timespan;
use std::time::Duration;
use super::TimespanMetric;
</imports>
<Rust>
#[test]
fn test_set_raw_timespan() {
  let timespan = TimespanMetric::new(glean_core::CommonMetricData::new("test_timespan", glean_core::metrics::MetricType::Timespan, "test_ping"), glean_core::metrics::TimeUnit::Millisecond);
  let elapsed = Duration::from_millis(100);
  timespan.set_raw(elapsed);
  let actual = timespan.test_get_value(None);
  let expected = Some(100);
  assert_eq!(actual, expected);
}
</Rust>
