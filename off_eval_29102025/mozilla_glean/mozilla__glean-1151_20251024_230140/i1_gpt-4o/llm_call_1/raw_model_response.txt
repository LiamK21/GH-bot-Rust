<imports>
use super::glean_enable_logging;
use log::LevelFilter;
use oslog::OsLogger;
</imports>
<Rust>
#[test]
fn test_logging_initialization() {
    let logger = OsLogger::new("org.mozilla.glean", "glean_core").level_filter(LevelFilter::Debug);
    let expected = logger.init().is_ok();
    
    glean_enable_logging();
    
    let actual = logger.init().is_err();
    assert_eq!(expected, actual);
}
</Rust>
