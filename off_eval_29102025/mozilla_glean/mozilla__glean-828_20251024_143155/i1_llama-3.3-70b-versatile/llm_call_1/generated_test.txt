#[test]
fn test_lazy_static_replacement() {
  let expected_regex = Regex::new("^[a-z_][a-z0-9_-]{0,29}(\\.[a-z_][a-z0-9_-]{0,29})*$").unwrap();
  let actual_regex: &Lazy<Regex> = &LABEL_REGEX;
  let actual_regex_value = actual_regex.force();
  assert_eq!(expected_regex.as_str(), actual_regex_value.as_str());
}
