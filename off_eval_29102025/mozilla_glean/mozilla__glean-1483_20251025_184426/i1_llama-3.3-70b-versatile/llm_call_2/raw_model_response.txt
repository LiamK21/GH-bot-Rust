<imports>
use super::shutdown;
use super::with_glean_mut;
use crate::block_on_dispatcher;
use crossbeam_channel::bounded;
use crossbeam_channel::TryRecvError;
use std::path::Path;
use std::fs;
use std::thread;
use super::dispatcher;
</imports>
<Rust>
#[test]
fn test_shutdown_clears_dirty_bit() {
  let (s, r) = bounded(1);
  let data_dir = Path::new("/tmp/glean_test");
  fs::create_dir_all(data_dir).unwrap();
  dispatcher::launch(move || {
    with_glean_mut(|glean| {
      glean.set_dirty_flag(true);
    })
  });
  crate::block_on_dispatcher();
  shutdown();
  dispatcher::launch(move || {
    with_glean_mut(|glean| {
      assert_eq!(glean.set_dirty_flag(false), ());
    })
  });
  crate::block_on_dispatcher();
  dispatcher::launch(move || {
    with_glean_mut(|glean| {
      assert_eq!(glean.set_dirty_flag(true), ());
    })
  });
  crate::block_on_dispatcher();
  shutdown();
  dispatcher::launch(move || {
    with_glean_mut(|glean| {
      assert_eq!(glean.set_dirty_flag(false), ());
    })
  });
  crate::block_on_dispatcher();
}
</Rust>
