<imports> 
use super::shutdown;
use super::with_glean_mut;
use crate::block_on_dispatcher;
use crossbeam_channel::bounded;
use crossbeam_channel::TryRecvError;
use glean_core::rlb::dispatcher;
use glean_core::rlb::net;
use std::thread;
</imports>
<Rust>
#[test]
fn test_shutdown_clears_dirty_bit() {
  let (s, r) = bounded(1);
  let data_dir = std::path::Path::new("/tmp/glean_test");
  std::fs::create_dir_all(data_dir).unwrap();
  dispatcher::launch(move || {
    with_glean_mut(|glean| {
      glean.set_dirty_flag(true);
    })
  });
  block_on_dispatcher();
  shutdown();
  dispatcher::launch(move || {
    with_glean_mut(|glean| {
      assert_eq!(glean.get_dirty_flag(), false);
    })
  });
  block_on_dispatcher();
}
</Rust>
