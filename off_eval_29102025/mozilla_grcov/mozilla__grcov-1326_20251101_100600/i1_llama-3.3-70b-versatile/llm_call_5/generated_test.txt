#[test]
fn test_jacoco_report_parsing_for_kotlin() {
  let f = File::open("./test/jacoco/kotlin-jacoco-report.xml").expect("Failed to open xml file");
  let mut file = BufReader::new(f);
  let package = "";
  let mut buf = Vec::new();
  let result = parse_jacoco_report_package(&mut file, &mut buf, package);
  assert!(result.is_ok());
}
