<imports>
use super::parse_version;
use std::version::Version;
</imports>
<Rust>
#[test]
fn test_parse_version_after_removing_check_gcov_version() {
  let gcov_output = "gcov version 4.9.0";
  let expected_version = Version { major: 4, minor: 9, patch: 0, pre: vec![], build: vec![] };
  let actual_version = parse_version(gcov_output);
  assert_eq!(actual_version, expected_version);
}
</Rust>
