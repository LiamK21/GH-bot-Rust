use std::path::Path;

#[test]
fn test_parse_gcov_gz_with_floating_point_error() {
    let result = parse_gcov_gz(Path::new("./test/mozillavpn_serverconnection.gcno.gcov.json.gz"));
    assert!(result.is_ok());
    let results = result.unwrap();
    assert_eq!(results.len(), 37);
    let (ref source_name, ref result) = results[0];
    assert_eq!(source_name, "server/serverconnection.cpp");
    assert_eq!(result.lines.len(), 37);
    assert!(result.functions.contains_key("ServerConnection::readData()"));
    let func = result.functions.get("ServerConnection::readData()").unwrap();
    assert_eq!(func.start, 188);
    assert!(!func.executed);
}
