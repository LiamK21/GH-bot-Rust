<imports>
use serde::{Deserialize, Deserializer};
use serde_json::Number;
use std::io::Cursor;
use super::deserialize_counter;
</imports>
<Rust>
#[test]
fn test_deserialize_counter() {
  let json = r#"
    {
      "count": 123.0
    }
  "#;
  let cursor = Cursor::new(json);
  let deserializer = serde_json::Deserializer::from_reader(cursor);
  let result: u64 = deserialize_counter(deserializer).unwrap();
  assert_eq!(result, 123);
}
</Rust>
