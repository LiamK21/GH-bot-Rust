<imports>
use serde_json::Deserializer;
use serde_json::de::StrRead;
use std::io::Cursor;
use super::deserialize_counter;
</imports>
<Rust>
#[test]
fn test_deserialize_counter() {
  let json = r#"
    {
      "count": 123.0
    }
  "#;
  let cursor = Cursor::new(json);
  let mut deserializer = Deserializer::from_reader(cursor);
  let result: u64 = deserialize_counter(&mut deserializer).unwrap();
  assert_eq!(result, 123);
}
</Rust>
