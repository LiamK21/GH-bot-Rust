#[test]
fn test_parse_lcov_error_handling() {
  let info_content = "SF:file\nDA:1,0,-\n";
  let buffer = Cursor::new(info_content);
  let lcov_reader = std::io::BufReader::new(buffer);
  let result = parse_lcov(lcov_reader, false);
  match result {
    Err(ParserError::Parse(_)) => assert!(true),
    _ => assert!(false),
  }
}
