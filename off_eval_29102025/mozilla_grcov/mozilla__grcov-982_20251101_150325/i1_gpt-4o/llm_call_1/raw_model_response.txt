<imports>
use super::output_html;
use std::fs;
use std::path::Path;
</imports>
<Rust>
#[test]
fn test_create_subpath_in_output_path() {
    let output_dir = Path::new("coverage/subpath");
    let results = &[];
    let num_threads = 1;
    let branch_enabled = false;
    let output_config_file = None;

    if output_dir.exists() {
        fs::remove_dir_all(output_dir).unwrap();
    }

    output_html(results, Some(output_dir), num_threads, branch_enabled, output_config_file);

    assert!(output_dir.exists());

    fs::remove_dir_all("coverage").unwrap();
}
</Rust>
