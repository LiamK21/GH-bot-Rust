<imports>
use super::Config;
use super::HtmlResources;
use std::path::Path;
use std::fs;
use super::gen_bundled_resources;
</imports>
<Rust>
#[test]
fn test_bundled_resources() {
  let config = Config { 
    hi_limit: 90.0, 
    med_limit: 75.0, 
    date: None, 
    branch_enabled: false, 
    precision: 2, 
    html_resources: HtmlResources::Bundled 
  };
  let output = Path::new("test_output");
  fs::create_dir_all(output).unwrap();
  gen_bundled_resources(&config, output);
  let bulma = output.join("bulma.min.css");
  assert!(bulma.exists());
}
