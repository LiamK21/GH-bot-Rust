#[test]
fn test_coverage_returns_false_when_no_lines_covered() {
    let result = CovResult {
        lines: HashMap::new(),
        functions: HashMap::from([(
            "top-level".to_string(),
            Function {
                executed: false,
                start: 0,
                end: 0,
                name_location: None,
                is_declared: false,
                exec_count: 0,
                is_incomplete: false,
            },
        )]),
        branches: HashMap::new(),
    };
    assert_eq!(super::is_covered(&result), false);
}
