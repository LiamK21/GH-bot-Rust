<imports>
use super::is_covered;
use defs::CovResult;
use std::collections::HashMap;
</imports>
<Rust>
#[test]
fn test_is_covered_early_return() {
  let mut result = CovResult {
    lines: HashMap::new(),
    functions: HashMap::new(),
    branches: HashMap::new(),
  };
  result.lines.insert("1".to_string(), 0);
  result.lines.insert("2".to_string(), 0);
  let expected = false;
  let actual = is_covered(&result);
  assert_eq!(expected, actual);
}
</Rust>
