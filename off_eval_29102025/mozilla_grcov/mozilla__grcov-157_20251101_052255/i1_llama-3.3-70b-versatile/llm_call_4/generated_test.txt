#[test]
fn test_is_covered_early_return() {
  let mut result = CovResult {
    lines: BTreeMap::new(),
    functions: BTreeMap::new(),
    branches: BTreeMap::new(),
  };
  result.lines.insert(1, 0);
  result.lines.insert(2, 0);
  let expected = false;
  let actual = is_covered(&result);
  assert_eq!(expected, actual);
}
