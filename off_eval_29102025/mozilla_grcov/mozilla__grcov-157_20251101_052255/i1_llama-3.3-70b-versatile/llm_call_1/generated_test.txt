#[test]
fn test_is_covered_early_return() {
  let mut result = CovResult {
    lines: HashMap::new(),
    functions: vec![],
  };
  result.lines.insert("line1".to_string(), 0);
  result.lines.insert("line2".to_string(), 0);
  let expected = false;
  let actual = is_covered(&result);
  assert_eq!(expected, actual);
}
