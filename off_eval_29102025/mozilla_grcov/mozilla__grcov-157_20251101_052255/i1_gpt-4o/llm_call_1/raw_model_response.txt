<imports>
use super::is_covered;
use crate::CovResult;
use std::collections::HashMap;
</imports>
<Rust>
#[test]
fn test_is_covered_early_return_when_all_lines_uncovered() {
    let mut lines = HashMap::new();
    lines.insert("line1".to_string(), 0);
    lines.insert("line2".to_string(), 0);

    let result = CovResult {
        lines,
        functions: HashMap::new(),
    };

    let expected = false;
    let actual = is_covered(&result);
    assert_eq!(expected, actual);
}
</Rust>
