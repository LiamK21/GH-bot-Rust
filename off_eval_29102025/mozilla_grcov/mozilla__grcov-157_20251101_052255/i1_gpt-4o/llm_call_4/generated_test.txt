#[test]
fn test_is_covered_early_return_when_all_lines_uncovered() {
    let mut lines = BTreeMap::new();
    lines.insert(1, 0);
    lines.insert(2, 0);

    let result = CovResult {
        lines,
        functions: BTreeMap::new(),
        branches: BTreeMap::new(),
    };

    let expected = false;
    let actual = is_covered(&result);
    assert_eq!(expected, actual);
}
