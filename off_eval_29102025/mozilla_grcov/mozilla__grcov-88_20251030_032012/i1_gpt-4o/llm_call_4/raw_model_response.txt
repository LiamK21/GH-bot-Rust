<imports>
use std::collections::HashMap;
use std::path::PathBuf;
use crate::defs::CovResult;
use crate::empty_result;
use super::rewrite_paths;
</imports>
<Rust>
#[test]
fn test_ignore_multiple_directories() {
    let mut result_map: HashMap<String, CovResult> = HashMap::new();
    result_map.insert("main.cpp".to_string(), empty_result!());
    result_map.insert("mydir/prova.h".to_string(), empty_result!());
    result_map.insert("mydir2/prova.h".to_string(), empty_result!());
    let expected: Vec<(PathBuf, PathBuf, CovResult)> = vec![(PathBuf::from("main.cpp"), PathBuf::from("main.cpp"), empty_result!())];
    let results: Vec<_> = rewrite_paths(result_map, None, "", "", false, false, vec!["mydir".to_string(), "mydir2".to_string()]).collect();
    assert_eq!(results, expected);
}
</Rust>
