#[test]
fn test_output_path() {
  let opt = Opt {
    output_path: Some(PathBuf::from("lcov.info")),
    output_types: vec![OutputType::Lcov],
    ..Default::default()
  };
  let output_types = opt.output_types;
  let output_path = match output_types.len() {
    0 => panic!("Output types has a default value"),
    1 => opt.output_path.as_deref(),
    _ => opt.output_path.as_deref(),
  };
  assert!(output_path.is_some());
  assert_eq!(output_path.unwrap().to_str(), Some("lcov.info"));
}
