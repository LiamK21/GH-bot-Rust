<imports>
use std::process::Command;
use std::fs;
use std::path::Path;
</imports>
<Rust>
#[test]
fn test_output_path_creation() {
    let output_file = "lcov.info";
    if Path::new(output_file).exists() {
        fs::remove_file(output_file).expect("Failed to remove existing output file");
    }

    let status = Command::new("cargo")
        .args(&[
            "run",
            "--",
            "target/coverage",
            "--binary-path",
            "target/debug/",
            "--source-dir",
            ".",
            "--output-path",
            output_file,
            "--output-type",
            "lcov",
        ])
        .status()
        .expect("Failed to execute command");

    assert!(status.success());
    assert!(Path::new(output_file).exists());

    if Path::new(output_file).exists() {
        fs::remove_file(output_file).expect("Failed to remove output file after test");
    }
}
</Rust>
