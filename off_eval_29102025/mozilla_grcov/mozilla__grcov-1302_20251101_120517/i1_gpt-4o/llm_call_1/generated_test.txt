#[test]
fn test_output_html_with_abs_link_prefix() {
    let results = &[];
    let output_dir = Some(Path::new("output"));
    let num_threads = 1;
    let branch_enabled = false;
    let output_config_file = None;
    let precision = 2;
    let abs_link_prefix = Some(String::from("http://example.com/"));
    let no_date = false;

    let expected_prefix = Some(Path::new("http://example.com/").to_path_buf());

    output_html(
        results,
        output_dir,
        num_threads,
        branch_enabled,
        output_config_file,
        precision,
        &abs_link_prefix,
        no_date,
    );

    let global_stats = HtmlGlobalStats {
        abs_prefix: expected_prefix.clone(),
        ..Default::default()
    };

    let stats = Arc::new(Mutex::new(global_stats));
    let actual_prefix = stats.lock().unwrap().abs_prefix.clone();

    assert_eq!(expected_prefix, actual_prefix);
}
