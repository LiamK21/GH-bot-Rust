<imports>
use std::path::Path;
use std::sync::{Arc, Mutex};
use crate::defs::HtmlGlobalStats;
use super::output_html;
</imports>
<Rust>
#[test]
fn test_output_html_with_abs_link_prefix() {
    let results = &[];
    let output_dir = Some(Path::new("output"));
    let num_threads = 1;
    let branch_enabled = false;
    let output_config_file = None;
    let precision = 2;
    let abs_link_prefix = Some(String::from("http://example.com/"));
    let no_date = false;

    output_html(
        results,
        output_dir,
        num_threads,
        branch_enabled,
        output_config_file,
        precision,
        &abs_link_prefix,
        no_date,
    );

    let global_stats = HtmlGlobalStats {
        abs_prefix: abs_link_prefix.clone().map(PathBuf::from),
        ..Default::default()
    };

    let stats = Arc::new(Mutex::new(global_stats));
    let actual_prefix = stats.lock().unwrap().abs_prefix.clone();

    assert_eq!(abs_link_prefix.map(PathBuf::from), actual_prefix);
}
</Rust>
