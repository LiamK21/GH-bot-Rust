#[test]
fn test_parse_gcov_and_lcov_with_edition_2018() {
  let gcov_path = Path::new("test.gcov");
  let mut file = File::create(gcov_path).unwrap();
  file.write_all(b"1:0:call    0").unwrap();
  let result = parse_gcov(gcov_path);
  assert!(result.is_ok());
  drop(file);
  let mut lcov_data = b"SF:src/main.rs\nFN:1:main\nFNF:1\nFNH:1\nend_of_record\n";
  let mut lcov_reader = BufReader::new(Cursor::new(lcov_data));
  let result = parse_lcov(lcov_reader, true);
  assert!(result.is_ok());
}
