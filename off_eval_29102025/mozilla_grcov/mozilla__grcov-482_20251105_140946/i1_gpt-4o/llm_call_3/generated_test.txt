#[test]
fn test_unexpected_version_handling() {
    let buffer = vec![0x80, 0x80, 0x80, 0x80];
    let reader = GcovReaderBuf { buffer, pos: 0, stem: String::from("test") };
    let result = reader.check_version();
    assert!(matches!(result, Err(GcovError::Str(err)) if err.contains("Unexpected version")));
}
