<imports>
use std::str;
use super::GcovReaderBuf;
use super::GcovError;
</imports>
<Rust>
#[test]
fn test_version_parsing_error() {
  let mut reader = GcovReaderBuf::new("test_file", vec![]);
  let bytes = b"adcg";
  let result = std::str::from_utf8(&bytes);
  let expected_error = GcovError::Str(format!("Unexpected version: {} in {}", String::from_utf8_lossy(&bytes), "test_file"));
  let actual_error = reader.parse_version(&bytes).unwrap_err();
  assert_eq!(format!("{}", expected_error), format!("{}", actual_error));
}
</Rust>
