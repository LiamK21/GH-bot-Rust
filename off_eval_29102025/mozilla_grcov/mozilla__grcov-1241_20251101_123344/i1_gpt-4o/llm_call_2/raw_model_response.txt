<imports>
use super::parse_lcov;
use std::vec::Vec;
</imports>
<Rust>
#[test]
fn test_lcov_parser_invalid_key() {
    let buf = "XX:152,4\nDA:153,4\nDA:154,8\nDA:156,12\n".as_bytes().to_vec();
    let result = parse_lcov(buf, true);
    assert!(result.is_err());
    let error = result.unwrap_err();
    assert_eq!(error.to_string(), "Invalid key at line 1");
}
</Rust>
