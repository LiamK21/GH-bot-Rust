<imports> 
use super::parse_lcov;
use super::ParserError;
</imports>
<Rust>
#[test]
fn test_lcov_parser_malformed_data() {
  let buffer = "SF:/path/to/file
DA:152,4
DA:153,4
DA:154,8
DA:156,12
DA
END:
".as_bytes().to_vec();
  let result = parse_lcov(buffer, true);
  assert!(result.is_err());
  let error = result.unwrap_err();
  assert_eq!(error.to_string(), "Invalid record: 'DA at line 6'");
}</Rust>
