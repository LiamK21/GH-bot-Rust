<imports>
use std::path::PathBuf;
use crate::defs::CovResult;
use crate::output::output_coveralls;
use crate::utils::convert_to_json_string;
</imports>
<Rust>
#[test]
fn test_output_coveralls_with_custom_branch() {
    let results: Vec<(PathBuf, PathBuf, CovResult)> = Vec::new();
    let repo_token = "dummy_token";
    let service_name = "dummy_service";
    let service_number = "dummy_number";
    let service_job_number = "dummy_job_number";
    let commit_sha = "dummy_commit_sha";
    let with_function_info = false;
    let output_file = None;
    let vcs_branch = "feature-branch";

    let mut output = Vec::new();
    output_coveralls(
        Box::new(results.into_iter()),
        repo_token,
        service_name,
        service_number,
        service_job_number,
        commit_sha,
        with_function_info,
        output_file,
        vcs_branch,
    );

    let output_str = convert_to_json_string(&output).unwrap();
    assert!(output_str.contains(r#""branch": "feature-branch""#));
}
</Rust>
