<imports> 
use super::parse_jacoco_xml_report;
use std::fs::File;
use std::io::BufReader;
use std::path::Path;
</imports>
<Rust>
#[test]
fn test_jacoco_xml_parsing() {
  let file_path = Path::new("./test/jacoco/basic-report.xml");
  let file = File::open(file_path).expect("Failed to open xml file");
  let file_reader = BufReader::new(&file);
  let result = parse_jacoco_xml_report(file_reader).unwrap();
  let expected = vec![(String::from("hello.java"), super::CovResult {
    lines: [(1, 0), (4, 1), (6, 1)].iter().cloned().collect(),
    branches: [(3, 0, true), (3, 1, true)].iter().cloned().collect(),
    functions: [
      ("hello#<init>".to_string(), super::Function { start: 1, executed: false }),
      ("hello#main".to_string(), super::Function { start: 3, executed: true }),
    ].iter().cloned().collect(),
  })];
  assert_eq!(result, expected);
}
</Rust>
