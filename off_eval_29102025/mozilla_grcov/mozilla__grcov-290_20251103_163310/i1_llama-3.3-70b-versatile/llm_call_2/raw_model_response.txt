<imports>
use std::path::PathBuf;
use super::get_abs_path;
</imports>
<Rust>
#[test]
fn test_get_abs_path_normalization() {
  let source_dir = Some(PathBuf::from("/home/worker/firefox/"));
  let rel_path = PathBuf::from("../bar/oof/");
  let mut cache = None;
  let expected: Option<(PathBuf, PathBuf)> = None;
  let actual = get_abs_path(&source_dir, rel_path, &mut cache);
  assert_eq!(actual, expected);
}
