#[test]
fn test_rewrite_paths_with_invalid_path() {
    let result_map = HashMap::new();
    let path_mapping = None;
    let source_dir = Some(PathBuf::from("/invalid/source/dir"));
    let prefix_dir = None;
    let ignore_not_existing = false;
    let mut to_ignore_dirs = vec![];
    let filter_option = None;

    let result = rewrite_paths(
        result_map,
        path_mapping,
        source_dir,
        prefix_dir,
        ignore_not_existing,
        &mut to_ignore_dirs,
        filter_option,
    );

    assert!(result.next().is_none());
}
