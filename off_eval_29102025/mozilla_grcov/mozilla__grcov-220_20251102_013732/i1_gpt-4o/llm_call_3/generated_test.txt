#[test]
fn test_thread_naming() {
    let handle_producer = thread::Builder::new().name(String::from("Producer")).spawn(|| {}).unwrap();
    let producer_thread_name = handle_producer.thread().name().unwrap().to_string();
    assert_eq!(producer_thread_name, "Producer");

    let handle_consumer = thread::Builder::new().name(String::from("Consumer 0")).spawn(|| {}).unwrap();
    let consumer_thread_name = handle_consumer.thread().name().unwrap().to_string();
    assert_eq!(consumer_thread_name, "Consumer 0");
}
