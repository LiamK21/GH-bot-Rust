<imports>use std::collections::HashMap;
use std::path::PathBuf;
use super::rewrite_paths;
use serde_json::Value;</imports>
<Rust>
#[test]
fn test_rewrite_paths_skips_symlinks() {
    let result_map = HashMap::new();
    let path_mapping = None;
    let source_dir = Some(PathBuf::from("/path/to/source"));
    let prefix_dir = None;
    let ignore_not_existing = false;
    let to_ignore_dirs = vec![];
    let filter_option = None;

    let expected: Vec<(PathBuf, PathBuf, ())> = vec![]; // Expected result after skipping symlinks

    let actual: Vec<_> = rewrite_paths(
        result_map,
        path_mapping,
        source_dir,
        prefix_dir,
        ignore_not_existing,
        to_ignore_dirs,
        filter_option,
    ).collect();

    assert_eq!(expected, actual);
}</Rust>
