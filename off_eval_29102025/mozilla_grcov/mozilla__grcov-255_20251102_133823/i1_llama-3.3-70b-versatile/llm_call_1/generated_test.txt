#[test]
fn test_output_file_option() {
  let output_file = Some("test_output.txt");
  let output_file_none = None;
  let results = vec![(String::new(), Path::new("test.txt"), String::new())];
  let mut file = File::create("test_output.txt").unwrap();
  file.write_all(b"test").unwrap();
  let expected_output = "test";
  let mut actual_output = String::new();
  {
    let mut writer = io::BufWriter::new(File::open("test_output.txt").unwrap());
    writer.write_all(b"test").unwrap();
    let mut file = File::open("test_output.txt").unwrap();
    io::BufReader::new(file).read_to_string(&mut actual_output).unwrap();
  }
  assert_eq!(expected_output, actual_output);
  std::fs::remove_file("test_output.txt").unwrap();
}
