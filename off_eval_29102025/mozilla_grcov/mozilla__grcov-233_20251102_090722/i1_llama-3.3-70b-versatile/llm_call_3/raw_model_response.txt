<imports>
use std::path::PathBuf;
use std::process::Command;
use std::io;
use super::run_gcov;
</imports>
<Rust>
#[test]
fn test_gcov_failure_output() {
  let gcno_path = PathBuf::from("non_existent_gcno_file.gcno");
  let working_dir = PathBuf::from(".");
  let _ = std::fs::remove_file(&gcno_path);
  let result = std::panic::catch_unwind(|| run_gcov(&gcno_path, false, &working_dir));
  assert!(result.is_err());
  if let Err(panic_msg) = result {
    let panic_str = panic_msg.downcast::<&str>().unwrap();
    assert!(panic_str.contains("gcov execution failed"));
  }
}
</Rust>
