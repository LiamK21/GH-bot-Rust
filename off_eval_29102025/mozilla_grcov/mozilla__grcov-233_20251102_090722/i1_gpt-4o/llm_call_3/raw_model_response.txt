<imports>
use std::path::PathBuf;
use std::env;
use std::process::Command;
use std::panic;
use super::run_gcov;
</imports>
<Rust>
#[test]
fn test_gcov_failure_output() {
    let gcno_path = PathBuf::from("invalid_path.gcno");
    let working_dir = PathBuf::from(".");
    let branch_enabled = false;

    env::set_var("GCOV", "false_gcov_command");

    let result = panic::catch_unwind(|| {
        run_gcov(&gcno_path, branch_enabled, &working_dir);
    });

    assert!(result.is_err());
    if let Err(err) = result {
        let err_string = format!("{:?}", err);
        assert!(err_string.contains("gcov execution failed on"));
    }
}
</Rust>
