#[test]
fn test_get_abs_path_normalizes_path() {
    let source_dir = Some(PathBuf::from("/source"));
    let rel_path = PathBuf::from("./foo//bar");
    let mut cache = None;

    let expected_abs_path = PathBuf::from("/source/foo/bar");
    let expected_rel_path = PathBuf::from("foo/bar");

    let (actual_abs_path, actual_rel_path) = get_abs_path(&source_dir, rel_path, &mut cache);

    assert_eq!(actual_abs_path, expected_abs_path);
    assert_eq!(actual_rel_path, expected_rel_path);
}
