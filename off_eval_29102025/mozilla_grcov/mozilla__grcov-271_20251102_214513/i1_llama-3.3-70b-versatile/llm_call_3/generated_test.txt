#[test]
fn test_rewrite_paths_without_sorting() {
  let mut result_map: HashMap<String, (), std::collections::hash_map::DefaultHasher> = HashMap::new();
  let path_mapping: Option<String> = None;
  let source_dir: Option<PathBuf> = None;
  let prefix_dir: Option<PathBuf> = None;
  let ignore_not_existing = false;
  let mut to_ignore_dirs_unsorted = vec!["bar/*".to_string(), "foo/*".to_string()];
  let mut to_ignore_dirs_sorted = vec!["bar/*".to_string(), "foo/*".to_string()];
  to_ignore_dirs_sorted.sort_unstable();
  let filter_option: Option<bool> = None;
  let expected = rewrite_paths(result_map.clone(), path_mapping.clone(), source_dir.clone(), prefix_dir.clone(), ignore_not_existing, to_ignore_dirs_sorted, filter_option.clone());
  let actual = rewrite_paths(result_map, path_mapping, source_dir, prefix_dir, ignore_not_existing, to_ignore_dirs_unsorted, filter_option);
  assert!(expected.count() > 0);
  assert!(actual.count() > 0);
  assert_eq!(expected.count(), actual.count());
}
