#[test]
fn test_rewrite_paths_without_sorting() {
  let result_map: HashMap<String, String> = HashMap::new();
  let path_mapping = None;
  let source_dir = None;
  let prefix_dir = None;
  let ignore_not_existing = false;
  let mut to_ignore_dirs = vec!["bar/*".to_string(), "foo/*".to_string()];
  let filter_option = None;
  let expected = rewrite_paths(result_map.clone(), path_mapping.clone(), source_dir.clone(), prefix_dir.clone(), ignore_not_existing, vec!["bar/*".to_string(), "foo/*".to_string()], filter_option.clone());
  let actual = rewrite_paths(result_map, path_mapping, source_dir, prefix_dir, ignore_not_existing, to_ignore_dirs, filter_option);
  // Since the return type is an iterator, we can't directly compare the two.
  // However, we can check if the iterator is empty or not.
  assert!(expected.count() > 0);
  assert!(actual.count() > 0);
}
