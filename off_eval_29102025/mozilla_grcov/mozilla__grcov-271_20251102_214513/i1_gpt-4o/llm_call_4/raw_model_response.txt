<imports>
use std::collections::HashMap;
use std::path::PathBuf;
use crate::defs::CovResult;
use crate::defs::CovResultIter;
use crate::defs::CovResultMap;
use super::rewrite_paths;
use serde_json::Value;
use rustc_hash::FxHasher;
use std::hash::BuildHasherDefault;
</imports>
<Rust>
#[test]
fn test_rewrite_paths_without_sorting() {
    let result_map: HashMap<String, CovResult, BuildHasherDefault<FxHasher>> = HashMap::default();
    let path_mapping = None;
    let source_dir = None;
    let prefix_dir = None;
    let ignore_not_existing = false;
    let mut to_ignore_dirs = vec!["b/*".to_string(), "a/*".to_string()];
    let filter_option = None;

    let expected: Vec<String> = vec!["b/*".to_string(), "a/*".to_string()];

    let actual: Vec<String> = rewrite_paths(
        result_map,
        path_mapping,
        source_dir,
        prefix_dir,
        ignore_not_existing,
        to_ignore_dirs.clone(),
        filter_option,
    )
    .map(|(_, _, res)| format!("{:?}", res))
    .collect();

    assert_eq!(expected, actual);
}
</Rust>
