#[test]
fn test_profraws_to_lcov_parallelism() {
  let temp_dir = PathBuf::from("/tmp");
  fs::create_dir_all(&temp_dir).unwrap();
  let profraw_path = temp_dir.join("test.profraw");
  let binary_path = temp_dir.join("test");
  fs::File::create(&profraw_path).unwrap();
  fs::File::create(&binary_path).unwrap();
  let profraw_paths = vec![profraw_path.clone()];
  let binary_path = binary_path;
  let working_dir = temp_dir.clone();
  let result = profraws_to_lcov(&profraw_paths, &binary_path, &working_dir);
  assert!(result.is_ok());
  fs::remove_dir_all(temp_dir).unwrap();
}
