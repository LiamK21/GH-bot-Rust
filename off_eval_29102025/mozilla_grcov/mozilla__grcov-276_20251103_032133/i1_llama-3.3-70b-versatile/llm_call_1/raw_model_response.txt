<imports> 
use clap::{App, Arg};
use std::process;
</imports>
<Rust>
#[test]
fn test_cli_options() {
  let result = App::new("grcov")
    .arg(Arg::with_name("paths")
         .help("Sets the input paths to use")
         .required(true)
         .multiple(true)
         .takes_value(true))
    .get_matches_from(vec!["grcov", "path1", "path2"]);
  assert!(result.is_ok());
  let result = App::new("grcov")
    .arg(Arg::with_name("paths")
         .help("Sets the input paths to use")
         .required(true)
         .multiple(true)
         .takes_value(true))
    .get_matches_from(vec!["grcov"]);
  assert!(result.is_err());
}</Rust>
