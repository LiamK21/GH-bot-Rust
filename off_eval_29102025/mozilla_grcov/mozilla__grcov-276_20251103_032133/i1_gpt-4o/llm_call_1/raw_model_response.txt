#[test]
fn test_cli_options_with_clap() {
    let output = Command::new("cargo")
        .arg("run")
        .arg("--")
        .arg("test_dir")
        .arg("-t")
        .arg("lcov")
        .output()
        .expect("Failed to execute command");

    let expected_status = true;
    let actual_status = output.status.success();

    assert_eq!(expected_status, actual_status);
}
