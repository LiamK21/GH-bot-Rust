<imports> 
use std::path::PathBuf;
use super::rewrite_paths;
use super::CovResultMap;
use super::CovResultIter;
use super::CovResult;
use std::collections::HashMap;
</imports>
<Rust>
#[test]
fn test_rewrite_paths_for_same_file_names() {
  let mut result_map: CovResultMap = HashMap::new();
  result_map.insert("lib.rs".to_string(), CovResult{});
  result_map.insert("lib.rs".to_string(), CovResult{});
  let results = rewrite_paths(
    result_map,
    None,
    Some(PathBuf::from("src")),
    None,
    true,
    Vec::new(),
    None,
  );
  let mut results: Vec<(PathBuf, PathBuf, CovResult)> = results.collect();
  assert!(results.len() == 2);
  let (abs_path1, rel_path1, _result1) = results.remove(0);
  let (abs_path2, rel_path2, _result2) = results.remove(0);
  assert_ne!(abs_path1, abs_path2);
}</Rust>
