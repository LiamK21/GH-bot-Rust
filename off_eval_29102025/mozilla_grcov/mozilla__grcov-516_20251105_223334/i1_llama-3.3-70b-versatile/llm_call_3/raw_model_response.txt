<imports>
use std::path::PathBuf;
use std::process::Command;
use std::fs;
use super::profraws_to_lcov;
</imports>
<Rust>
#[test]
fn test_profraws_to_lcov_multiple_files() {
  let tmp_dir = PathBuf::from("/tmp");
  fs::create_dir_all(&tmp_dir).unwrap();
  let profraw_path1 = tmp_dir.join("test1.profraw");
  let profraw_path2 = tmp_dir.join("test2.profraw");
  let binary_path = "test_binary".to_string();
  let profraw_paths = vec![profraw_path1.clone(), profraw_path2.clone()];
  let expected_output = Command::new("rustc").arg("--version").output().unwrap();
  let actual_output = profraws_to_lcov(&profraw_paths, &binary_path, &tmp_dir).unwrap();
  assert!(actual_output.len() > 0);
  let profdata_path = tmp_dir.join("grcov.profdata");
  assert!(profdata_path.exists());
}
</Rust>
