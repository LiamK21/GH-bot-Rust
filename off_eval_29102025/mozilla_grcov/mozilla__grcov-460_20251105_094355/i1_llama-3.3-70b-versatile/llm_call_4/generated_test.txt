#[test]
fn test_branch_coverage_in_html_report() {
  let config = get_config();
  let result = CovResult {
    lines: BTreeMap::new(),
    functions: BTreeMap::new(),
    branches: BTreeMap::new(),
  };
  let stats = get_stats(&result);
  let expected_total_branches = 0;
  let expected_covered_branches = 0;
  assert_eq!(stats.total_branches, expected_total_branches);
  assert_eq!(stats.covered_branches, expected_covered_branches);
}
