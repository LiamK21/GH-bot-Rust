#[test]
fn test_branch_coverage_in_html_report() {
  let config = get_config();
  let mut result = CovResult {
    lines: HashMap::new(),
    functions: HashMap::new(),
    branches: HashMap::new(),
  };
  result.branches.insert("branch1".to_string(), vec![true, false]);
  let stats = get_stats(&result);
  let expected_total_branches = 2;
  let expected_covered_branches = 1;
  assert_eq!(stats.total_branches, expected_total_branches);
  assert_eq!(stats.covered_branches, expected_covered_branches);
}
