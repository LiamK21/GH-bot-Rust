#[test]
fn test_wildcard_exclusion() {
  let mut result_map: CovResultMap = HashMap::new();
  result_map.insert("vs2017_15.6.6/prova.h".to_string(), CovResult { /* initialize fields here */ });
  result_map.insert("vs2017_15.7.6/prova.h".to_string(), CovResult { /* initialize fields here */ });
  result_map.insert("main.cpp".to_string(), CovResult { /* initialize fields here */ });

  let expected_count = 1;
  let actual_count: usize = rewrite_paths(result_map, None, "", "", false, false, vec!["vs2017_*/".to_string()]).count();
  assert_eq!(expected_count, actual_count);
}
