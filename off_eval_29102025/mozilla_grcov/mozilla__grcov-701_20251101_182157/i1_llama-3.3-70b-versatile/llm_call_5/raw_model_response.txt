#[test]
fn test_make_context_returns_context() {
  let context = make_context();
  let expected_type = "Context&lt;CtxThreadSafe&gt;";
  let actual_type = any::type_name::<_>(&context);
  assert_eq!(actual_type, expected_type);
}
