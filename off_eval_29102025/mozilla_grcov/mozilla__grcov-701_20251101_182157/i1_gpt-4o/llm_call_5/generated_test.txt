#[test]
fn test_make_context_with_generic() {
    env::set_var("BULMA_VERSION", "0.9.3");
    let ctx: Context<CtxThreadSafe> = make_context();
    let expected_bulma_version = "0.9.3";
    let actual_bulma_version: &str = ctx.get("bulma_version").unwrap();
    assert_eq!(expected_bulma_version, actual_bulma_version);
}
