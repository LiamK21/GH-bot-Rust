<imports> 
use std::path::PathBuf;
use super::get_coveralls_git_info;
use serde_json::json;
</imports>
<Rust>
#[test]
fn test_get_coveralls_git_info() {
  let commit_sha = "b31f08d07ae564b08237e5a336e478b24ccc4a65";
  let vcs_branch = "master";
  let expected = json!({
    "head": {
      "id": commit_sha,
    },
    "branch": vcs_branch,
  });
  let actual = get_coveralls_git_info(commit_sha, vcs_branch);
  assert_ne!(actual, json!({}));
  assert!(actual["head"].has("id"));
  assert_eq!(actual["head"]["id"], commit_sha);
  assert_eq!(actual["branch"], vcs_branch);
}</Rust>
