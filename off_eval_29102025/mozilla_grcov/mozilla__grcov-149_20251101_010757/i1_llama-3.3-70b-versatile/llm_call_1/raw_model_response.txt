<imports> 
use std::path::PathBuf;
use super::rewrite_paths;
use std::collections::HashMap;
</imports>
<Rust>
#[test]
fn test_rewrite_paths_add_prefix() {
  let mut result_map: HashMap<String, ()> = HashMap::new();
  result_map.insert("org/example/Hello.java".to_string(), ());
  let results = rewrite_paths(
    result_map,
    None,
    None,
    None,
    Some(PathBuf::from("mobile/android/app/src/java")),
    false,
    false,
    Vec::new(),
    None,
  );
  let mut count = 0;
  for (abs_path, rel_path, _) in results {
    count += 1;
    assert_eq!(
      abs_path,
      PathBuf::from("mobile/android/app/src/java/org/example/Hello.java")
    );
    assert_eq!(
      rel_path,
      PathBuf::from("mobile/android/app/src/java/org/example/Hello.java")
    );
  }
  assert_eq!(count, 1);
}
</Rust>
