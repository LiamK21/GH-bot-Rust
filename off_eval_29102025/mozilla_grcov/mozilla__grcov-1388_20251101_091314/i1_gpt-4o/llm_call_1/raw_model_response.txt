<Rust>
#[test]
fn test_is_known_binary() {
    let elf_bytes = [0x7F, b'E', b'L', b'F', 0x00, 0x00, 0x00, 0x00];
    let pe_bytes = [0x4D, 0x5A, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00];
    let macho_bytes = [0xFE, 0xED, 0xFA, 0xCE, 0x00, 0x00, 0x00, 0x00];
    let coff_bytes = [0x4C, 0x01, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00];
    let wasm_bytes = [0x00, 0x61, 0x73, 0x6D, 0x00, 0x00, 0x00, 0x00];
    let llvm_bytes = [0x42, 0x43, 0xC0, 0xDE, 0x00, 0x00, 0x00, 0x00];
    let unknown_bytes = [0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00];

    assert!(is_known_binary(&elf_bytes));
    assert!(is_known_binary(&pe_bytes));
    assert!(is_known_binary(&macho_bytes));
    assert!(is_known_binary(&coff_bytes));
    assert!(is_known_binary(&wasm_bytes));
    assert!(is_known_binary(&llvm_bytes));
    assert!(!is_known_binary(&unknown_bytes));
}
</Rust>
