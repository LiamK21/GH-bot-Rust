<imports>
use log::Level;
use log::Log;
use std::io::Cursor;
use super::add_results;
</imports>
<Rust>
#[test]
fn test_execution_count_overflow_logging_before_patch() {
  let mut results: Vec<(String, ())> = Vec::new();
  let warn_overflow = true;
  let original_log = Log::log;
  Log::log = |log| {
    if log.level == Level::Error {
      assert_eq!(log.args.to_string(), "Execution count overflow detected.");
    }
    original_log(log)
  };
  add_results(&mut results, warn_overflow);
  Log::log = original_log;
}

#[test]
fn test_execution_count_overflow_logging_after_patch() {
  let mut results: Vec<(String, ())> = Vec::new();
  let warn_overflow = true;
  let original_log = Log::log;
  Log::log = |log| {
    if log.level == Level::Warn {
      assert_eq!(log.args.to_string(), "Execution count overflow detected.");
    }
    original_log(log)
  };
  add_results(&mut results, warn_overflow);
  Log::log = original_log;
}
</Rust>
