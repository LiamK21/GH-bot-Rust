use super::Version;

<Rust>
#[test]
fn test_parse_version_with_carriage_return() {
  let gcov_output = "gcov (GCC) 12.2.0\r";
  let expected_version = Version::new(12, 2, 0);
  let actual_version = parse_version(gcov_output);
  assert_eq!(actual_version, expected_version);
}</Rust>
