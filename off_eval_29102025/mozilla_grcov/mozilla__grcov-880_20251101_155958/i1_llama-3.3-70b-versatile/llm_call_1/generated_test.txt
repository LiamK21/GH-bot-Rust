#[test]
fn test_jacoco_xml_report_parsing() {
  let mut file = File::open("test_data/jacoco_report.xml").unwrap();
  let mut xml_data = String::new();
  file.read_to_string(&mut xml_data).unwrap();
  let xml_reader = Cursor::new(xml_data);
  let result = parse_jacoco_xml_report(xml_reader);
  assert!(result.is_ok());
}
