<imports>
use std::collections::HashMap;
use std::env;
use std::ffi::OsString;
use std::path::PathBuf;
use super::parse_args;
use super::Context;
use super::Action;
use super::Report;
</imports>
<Rust>
#[test]
fn test_parse_args_with_report_command() {
  let mut context = Context::default();
  context.args = vec![OsString::from("cargo-grcov"), OsString::from("report")];
  let expected = vec![Action::Report(Report {
    path: PathBuf::from("target/coverage/debug"),
    context: Context::default(),
    output_type: "html".to_string(),
  })];
  let actual = parse_args(context).unwrap();
  assert_eq!(expected, actual);
}
</Rust>
