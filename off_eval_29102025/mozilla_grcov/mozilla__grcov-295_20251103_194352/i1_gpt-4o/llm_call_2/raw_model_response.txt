<imports>
use std::vec::IntoIter;
use crate::parser::{ParserError, try_next};
</imports>
<Rust>
#[test]
fn test_try_next_with_none() {
    let mut iter: IntoIter<i32> = vec![].into_iter();
    let line = "test line";
    let result = (|| {
        try_next!(iter, line);
        Ok::<(), ParserError>(())
    })();
    assert_eq!(result, Err(ParserError::InvalidRecord(line.to_string())));
}
</Rust>
