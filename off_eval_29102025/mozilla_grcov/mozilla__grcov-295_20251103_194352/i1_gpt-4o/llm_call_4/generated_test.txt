#[test]
fn test_try_next_with_none() {
    let mut iter: IntoIter<i32> = vec![].into_iter();
    let line = "test line";
    let result = (|| {
        try_next!(iter, line);
        Ok::<(), ParserError>(())
    })();
    assert!(matches!(result, Err(ParserError::InvalidRecord(l)) if l == line.to_string()));
}
