<imports>
use tera::{Context, Tera, CtxThreadSafe};
use std::env;
use super::make_context;
</imports>
<Rust>
#[test]
fn test_make_context_fails_with_ctx_thread_safe() {
  let result = std::panic::catch_unwind(|| {
    let _ = make_context() as Context<CtxThreadSafe>;
  });
  assert!(result.is_err());
}
#[test]
fn test_make_context_succeeds_without_ctx_thread_safe() {
  let result = std::panic::catch_unwind(|| {
    let _ = make_context();
  });
  assert!(result.is_ok());
}
</Rust>
