<imports>
use super::make_context;
use tera::Context;
use std::env;
</imports>
<Rust>
#[test]
fn test_make_context_creates_context_with_bulma_version() {
    env::set_var("BULMA_VERSION", "0.9.3");
    let context = make_context();
    let expected_version = "0.9.3";
    let actual_version = context.get("bulma_version").unwrap();
    assert_eq!(expected_version, actual_version);
}
</Rust>
