#[test]
fn test_check_file_reader_jacoco() {
    let mut reader = Cursor::new(b"TN:");
    let result = super::check_file(Some(&mut reader), &|f: &mut dyn Read| {
        let mut buf = [0; 3];
        if let Err(_) = f.read_exact(&mut buf) {
            false
        } else {
            buf == [b'T', b'N', b':']
        }
    });
    assert!(result);
}
