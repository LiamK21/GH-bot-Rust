#[test]
fn test_call_parse_llvm_gcno_buf_with_gcno_file() {
  let working_dir = "test/llvm";
  let file_stem = "file";
  let gcno = vec![1, 2, 3];
  let gcda = vec![4, 5, 6];
  let branch_enabled = true;
  let expected: Vec<(String, CovResult)> = vec![];
  let actual = call_parse_llvm_gcno_buf(working_dir, file_stem, &gcno, &gcda, branch_enabled);
  assert_ne!(actual, expected);
}
