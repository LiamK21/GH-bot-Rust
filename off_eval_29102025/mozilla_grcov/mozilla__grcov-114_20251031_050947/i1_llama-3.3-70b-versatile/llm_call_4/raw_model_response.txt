<imports>
use std::path::PathBuf;
use std::collections::BTreeMap;
use std::collections::HashMap;
use super::call_parse_llvm_gcno_buf;
use super::CovResult;
use super::Function;
</imports>
<Rust>
#[test]
fn test_call_parse_llvm_gcno_buf_with_gcno_file() {
  let working_dir = "test/llvm";
  let file_stem = "file";
  let gcno = vec![1, 2, 3];
  let gcda = vec![4, 5, 6];
  let branch_enabled = true;
  let expected: Vec<(String, CovResult)> = vec![(file_stem.to_string(), CovResult {
    lines: BTreeMap::new(),
    branches: BTreeMap::new(),
    functions: HashMap::new(),
  })];
  let actual = call_parse_llvm_gcno_buf(working_dir, file_stem, &gcno, &gcda, branch_enabled);
  assert_eq!(actual.len(), expected.len());
  if let Some((file, result)) = actual.get(0) {
    assert_eq!(file, &expected[0].0);
    assert_eq!(result.lines, expected[0].1.lines);
    assert_eq!(result.branches, expected[0].1.branches);
    assert_eq!(result.functions, expected[0].1.functions);
  }
}
</Rust>
