#[test]
fn test_lcov_parser_with_invalid_record() {
  let mut f = File::open("./test/invalid_DA_record.info").expect("Failed to open lcov file");
  let mut buf = Vec::new();
  f.read_to_end(&mut buf).unwrap();
  let result_without_ignore_parsing_error = parse_lcov(buf.clone(), true, false);
  assert!(result_without_ignore_parsing_error.is_err());
  let result_with_ignore_parsing_error = parse_lcov(buf, true, true);
  assert!(result_with_ignore_parsing_error.is_ok());
}
