#[test]
fn test_make_context_returns_context_without_ctx_thread_safe() {
  let result = std::panic::catch_unwind(|| {
    let _ = make_context();
  });
  assert!(result.is_ok());
}

#[test]
fn test_make_context_fails_when_cast_to_ctx_thread_safe() {
  let result = std::panic::catch_unwind(|| {
    let _ = make_context() as Context;
    panic!("Should not be able to cast to Context with CtxThreadSafe");
  });
  assert!(result.is_err());
}
