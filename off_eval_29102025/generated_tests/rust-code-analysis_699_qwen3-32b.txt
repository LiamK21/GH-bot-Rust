#[test]
fn test_cyclomatic_merge_minmax() {
    let mut s1 = Stats::default();
    s1.cyclomatic = 1.0;
    s1.compute_minmax();

    let mut s2 = Stats::default();
    s2.cyclomatic = 3.0;
    s2.compute_minmax();

    s1.merge(&s2);

    assert_eq!(s1.cyclomatic_average(), 2.0);
    assert_eq!(s1.cyclomatic_min(), 1.0);
    assert_eq!(s1.cyclomatic_max(), 3.0);
}
