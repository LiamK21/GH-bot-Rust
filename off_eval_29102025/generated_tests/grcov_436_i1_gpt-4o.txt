#[test]
#[should_panic(expected = "The output file ./target/debug/coverage/ is a directory, but must be a regular file.")]
fn test_output_file_is_directory() {
    let output_dir = "./target/debug/coverage/";
    if !Path::new(output_dir).exists() {
        fs::create_dir_all(output_dir).unwrap();
    }
    get_target_output_writable(Some(output_dir));
}
