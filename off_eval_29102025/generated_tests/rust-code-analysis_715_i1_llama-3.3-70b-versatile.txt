#[test]
fn test_compute_minmax() {
  let mut stats = Stats::default();
  stats.fn_nargs = 5;
  stats.closure_nargs = 3;
  stats.compute_minmax();
  assert_eq!(stats.fn_nargs_min, 5);
  assert_eq!(stats.fn_nargs_max, 5);
  assert_eq!(stats.closure_nargs_min, 3);
  assert_eq!(stats.closure_nargs_max, 3);
  assert_eq!(stats.fn_nargs_sum, 5);
  assert_eq!(stats.closure_nargs_sum, 3);
}
