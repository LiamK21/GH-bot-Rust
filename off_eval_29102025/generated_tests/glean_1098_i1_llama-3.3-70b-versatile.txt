#[test]
fn test_ping_request_body_size_limit() {
  let builder = Builder::new("Rust", 10);
  let result = builder.document_id("id").path("path").body("body_with_size_larger_than_10").build();
  let expected_error = ErrorKind::PingBodyOverflow(23);
  assert!(matches!(result, Err(error) if matches!(error.kind(), expected_error)));
}
