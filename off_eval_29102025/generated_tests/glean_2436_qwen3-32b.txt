#[test]
fn test_date_header_not_included_during_build() {
    let request = Builder::new("rust-test", 1024)
        .document_id("test-document-id") // Required to avoid panic during build
        .path("/submit/test")
        .body("test body")
        .build()
        .unwrap();
    assert!(!request.headers.contains_key("Date"));
}
