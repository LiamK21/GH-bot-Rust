#[test]
fn test_handle_file_with_none() {
    let archive = Archive {
        name: String::from("test_archive"),
        item: RefCell::new(ArchiveType::Plain(vec![])),
    };
    let path = PathBuf::from("non_existent_file");
    let gcno_stem_archives = RefCell::new(FxHashMap::default());
    let gcda_stem_archives = RefCell::new(FxHashMap::default());
    let infos = RefCell::new(FxHashMap::default());
    let xmls = RefCell::new(FxHashMap::default());
    let linked_files_maps = RefCell::new(FxHashMap::default());

    archive.handle_file(
        None::<&mut File>,
        &path,
        &gcno_stem_archives,
        &gcda_stem_archives,
        &infos,
        &xmls,
        &linked_files_maps,
        false,
    );
}
