#[test]
fn test_create_datetime_metric_with_time_unit() {
    let category = CString::new("test_category").unwrap();
    let name = CString::new("test_name").unwrap();
    let send_in_pings = ptr::null();
    let send_in_pings_len = 0;
    let lifetime = Lifetime::Ping;
    let disabled = 0;
    let time_unit = TimeUnit::Nanosecond;

    let metric_id = unsafe {
        glean_new_datetime_metric(
            FfiStr::from_cstr(&category),
            FfiStr::from_cstr(&name),
            send_in_pings,
            send_in_pings_len,
            lifetime,
            disabled,
            time_unit,
        )
    };

    assert!(metric_id > 0);
}
