#[test]
fn test_lifetime_size() {
  let lifetime: Lifetime = Lifetime::Ping;
  let time_unit: TimeUnit = TimeUnit::Nanosecond;
  let memory_unit: MemoryUnit = MemoryUnit::Byte;
  assert_eq!(mem::size_of::<Lifetime>(), 4);
  assert_eq!(mem::size_of::<TimeUnit>(), 4);
  assert_eq!(mem::size_of::<MemoryUnit>(), 4);
}
