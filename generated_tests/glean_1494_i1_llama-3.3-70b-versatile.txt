#[test]
fn test_launch_with_glean_resolves_issue() {
  let experiment_id = "test_expid".to_string();
  let branch = "test_branch".to_string();
  let extra = Some(HashMap::from([("key".to_string(), "value".to_string())]));
  launch_with_glean(move |glean| {
    set_experiment_active(experiment_id.clone(), branch.clone(), extra.clone());
  });
  shutdown();
  assert!(true); // This test will fail if launch_with_glean doesn't resolve the issue
}
