#[test]
fn test_valid_session_id() {
  let mut conn = Http3Connection::new();
  let session_id = StreamId::from(1);
  let invalid_session_id = StreamId::from(0);

  // Test valid session ID
  assert!(session_id.is_client_initiated());
  assert!(session_id.is_bidi());
  assert!(conn.is_stream_id_allowed(session_id));

  // Test invalid session ID
  assert!(!invalid_session_id.is_client_initiated());
  assert!(invalid_session_id.is_bidi());
  assert!(conn.is_stream_id_allowed(invalid_session_id));

  // Check connection closure with H3_ID_ERROR error code
  let result = conn.stream_stop_sending(invalid_session_id, Error::HttpId.code());
  assert!(result.is_ok());
}
