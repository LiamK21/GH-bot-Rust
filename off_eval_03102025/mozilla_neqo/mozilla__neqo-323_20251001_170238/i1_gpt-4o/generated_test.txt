#[test]
fn test_initial_packet_padding() {
    let mut conn = Connection::new_client(
        "example.com",
        &["alpn"],
        Instant::now(),
    ).unwrap();
    conn.state = State::Connected;
    conn.role = Role::Client;
    let out = conn.output_pkt_for_path(Instant::now()).unwrap();
    assert!(out.is_some());
    let datagram = out.unwrap();
    assert_eq!(datagram.len(), conn.path.as_ref().unwrap().mtu());
}
