#[test]
fn test_version_negotiation_does_not_repeat_client_version() {
  let mut rng = rand::thread_rng();
  let dcid = [1; 8];
  let scid = [2; 8];
  let client_version: u32 = 0x0a0a0a0a;
  let versions = &[];
  let vn = PacketBuilder::version_negotiation(&dcid, &scid, client_version, versions);
  let grease_start = vn.len() - 4;
  assert_ne!(vn[grease_start..], [0x0a, 0x0a, 0x0a, 0x0a]);
}
