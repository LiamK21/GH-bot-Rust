<imports>
use super::PacketBuilder;
use crate::packet::Version;
</imports>
<Rust>
#[test]
fn test_vn_does_not_include_client_version() {
    let dcid = &[0x01, 0x02, 0x03, 0x04];
    let scid = &[0x05, 0x06, 0x07, 0x08];
    let client_version = 0x0a0a0a0a;
    let versions = &Version::all();
    
    let vn_packet = PacketBuilder::version_negotiation(dcid, scid, client_version, versions);
    
    let client_version_bytes = client_version.to_be_bytes();
    assert!(!vn_packet.windows(4).any(|window| window == client_version_bytes));
}
</Rust>
