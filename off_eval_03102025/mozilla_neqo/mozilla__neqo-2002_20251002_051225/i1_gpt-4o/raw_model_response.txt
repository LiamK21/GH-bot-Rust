#[test]
fn test_socket_nonblocking() {
    let result = socket();
    assert!(result.is_ok());
    let sock = result.unwrap();
    let nonblocking = sock.inner.nonblocking().unwrap();
    assert!(!nonblocking, "Socket should be blocking");
}
