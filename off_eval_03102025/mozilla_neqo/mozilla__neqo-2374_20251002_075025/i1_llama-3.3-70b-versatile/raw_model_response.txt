<imports>
use neqo_transport::src::tracking::RecvdPackets;
use std::time::Instant;
use std::time::Duration;
</imports>
<Rust>
#[test]
fn test_ack_frequency_immediate_ack() {
  let mut rp = RecvdPackets::new(neqo_transport::src::packet_number::PacketNumberSpace::ApplicationData);
  let now = Instant::now();
  rp.ack_freq(0, 1, Duration::from_millis(10), 1);
  rp.set_received(now, 2, true);
  assert_ne!(Some(now), rp.ack_time());
  rp.set_received(now, 3, true);
  assert_ne!(Some(now), rp.ack_time());
  rp.set_received(now, 1, true);
  assert!(rp.ack_time().is_some());
}</Rust>
