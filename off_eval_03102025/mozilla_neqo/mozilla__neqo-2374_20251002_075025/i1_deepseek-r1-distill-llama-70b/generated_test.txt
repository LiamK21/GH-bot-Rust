#[test]
fn test_default_reordering_threshold() {
    let mut rp = RecvdPackets::new(PacketNumberSpace::ApplicationData);
    let now = Instant::now();

    // Without any ACK_FREQUENCY frames, reordering_threshold should be 1.
    assert_eq!(rp.reordering_threshold, 1);

    // Receive packets out of order to trigger immediate ACK.
    rp.set_received(now, 1, true);
    rp.set_received(now, 3, true);

    // Since reordering_threshold is 1, this should elicit an immediate ACK.
    assert!(rp.ack_time().is_some());
}
