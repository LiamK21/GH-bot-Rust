#[test]
fn test_pto_period_inner() {
  let mut rtt = LossRecovery::new();
  let pto_state = Some((30, Duration::from_millis(10)));
  let fast_pto = 1;
  let pn_space = 1;
  let expected = Duration::from_millis(10);
  let actual = rtt.pto_period_inner(pto_state, fast_pto, pn_space).unwrap_or(Duration::from_secs(3600));
  assert_ne!(actual, Duration::from_nanos(0));
}
