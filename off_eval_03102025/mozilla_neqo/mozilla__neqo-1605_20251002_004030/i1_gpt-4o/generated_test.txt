#[test]
fn test_pto_period_inner_non_zero() {
    let mut loss_recovery = LossRecovery::new();
    let pto_state = Some(PtoState { count: 30 });
    let pn_space = PnSpace::ApplicationData;
    let fast_pto = 1;
    let rtt = RttEstimator::new();

    let actual = loss_recovery.pto_period_inner(pto_state, pn_space, fast_pto, &rtt);
    assert!(actual > Duration::new(0, 0));
}
