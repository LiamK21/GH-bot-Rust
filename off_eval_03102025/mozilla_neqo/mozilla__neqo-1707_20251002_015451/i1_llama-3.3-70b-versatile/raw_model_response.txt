<imports> 
use neqo_common::udp::Socket;
use std::net::SocketAddr;
use std::io;
</imports>
<Rust>
#[test]
fn test_recv_mut_socket() {
  let mut socket = Socket::bind("127.0.0.1:0").unwrap();
  let local_address: SocketAddr = "127.0.0.1:0".parse().unwrap();
  let result = socket.recv(&local_address);
  assert!(result.is_ok());
}</Rust>
