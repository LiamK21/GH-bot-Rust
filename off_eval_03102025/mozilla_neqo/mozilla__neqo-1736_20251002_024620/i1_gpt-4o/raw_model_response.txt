<imports>
use super::Datagram;
use std::net::SocketAddr;
</imports>
<Rust>
#[test]
fn test_datagram_into_vec_conversion() {
    let source: SocketAddr = "127.0.0.1:12345".parse().unwrap();
    let destination: SocketAddr = "127.0.0.1:54321".parse().unwrap();
    let data = vec![1, 2, 3, 4, 5];
    let datagram = Datagram::new(source, destination, Default::default(), None, data.clone());

    let converted: Vec<u8> = datagram.into();
    assert_eq!(converted, data);
}
</Rust>
