#[test]
fn test_aead_real_new() {
  let secret = Vec::new();
  let p = Vec::new();
  let mut ctx = ptr::null_mut();
  let result_before = RealAead::new(secret, p, &mut ctx);
  assert!(result_before.is_err());
  let result_after = RealAead::new(secret, p, &mut ctx);
  assert!(result_after.is_ok());
}
