<imports>
use super::ServersRunner;
use std::io;
use std::net::{SocketAddr, ToSocketAddrs};
use tokio::runtime::Runtime;
</imports>
<Rust>
#[test]
fn test_servers_runner_initialization() {
    let rt = Runtime::new().unwrap();
    rt.block_on(async {
        let args = Args {
            listen_addresses: vec!["127.0.0.1:0".to_socket_addrs().unwrap().next().unwrap()],
            ..Default::default()
        };
        let result = ServersRunner::new(args);
        assert!(result.is_ok());
    });
}
</Rust>
