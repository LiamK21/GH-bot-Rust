<imports> 
use neqo_client::Handler;
use std::io;
use neqo_client::Error;
use std::collections::VecDeque;
use neqo_client::TransportError;
</imports>
<Rust>
#[test]
fn test_create_stream_unavailable_error() {
  let mut handler = Handler::new();
  let url = "https://example.com";
  let client_stream_id = 1;
  let out_file = io::stdout();
  let result = handler.create_stream(client_stream_id, url, out_file);
  let expected = Err(Error::Unavailable);
  assert!(matches!(result, Err(Error::Unavailable)));
  let mut vec: VecDeque<String> = VecDeque::new();
  vec.push_front(url.to_string());
  assert_eq!(handler.url_queue, vec);
}</Rust>
