#[test]
fn test_acknowledged_logging_removal() {
    let mut packet_range = PacketRange {
        smallest: 100,
        largest: 200,
        ack_needed: true,
    };
    let other = PacketRange {
        smallest: 100,
        largest: 200,
        ack_needed: true,
    };

    packet_range.acknowledged(&other);

    assert_eq!(packet_range.ack_needed, false);
}
</Rust>
