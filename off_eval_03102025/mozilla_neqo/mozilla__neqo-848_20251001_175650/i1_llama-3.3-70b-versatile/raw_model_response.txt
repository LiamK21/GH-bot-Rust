<imports> 
use neqo_transport::tracking::PacketRange;
use std::sync::atomic::AtomicBool;
use std::sync::atomic::Ordering;
use std::sync::Arc;
</imports>
<Rust>
#[test]
fn test_acking_does_not_log() {
  let range1 = PacketRange { smallest: 1, largest: 10, ack_needed: true };
  let range2 = PacketRange { smallest: 1, largest: 10, ack_needed: true };
  let logged = Arc::new(AtomicBool::new(false));
  let original_qinfo = std::mem::replace(&mut qinfo, |_, msg| {
    logged.store(true, Ordering::SeqCst);
  });
  range1.acknowledged(&range2);
  assert!(!logged.load(Ordering::SeqCst));
  std::mem::drop(original_qinfo);
}</Rust>
