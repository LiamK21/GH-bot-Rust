<imports> 
use neqo_transport::send_stream::SendStreams;
use neqo_transport::send_stream::StreamId;
use neqo_transport::send_stream::SendStream;
</imports>
<Rust>
#[test]
fn test_send_stream_order() {
  let mut send_streams = SendStreams::default();
  let stream_id1 = StreamId::from(0);
  let stream_id2 = StreamId::from(1);
  let stream_id3 = StreamId::from(2);
  let stream_id4 = StreamId::from(3);
  send_streams.0.insert(stream_id1, SendStream::default());
  send_streams.0.insert(stream_id2, SendStream::default());
  send_streams.0.insert(stream_id3, SendStream::default());
  send_streams.0.insert(stream_id4, SendStream::default());
  let mut actual_stream_id = Vec::new();
  for (id, _) in send_streams.into_iter() {
    actual_stream_id.push(id.0);
  }
  let expected_stream_id = vec![0, 1, 2, 3];
  assert_eq!(actual_stream_id, expected_stream_id);
}</Rust>
