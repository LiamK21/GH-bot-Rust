<Rust>
#[test]
fn test_send_arbitrary_data() {
  let mut server = SimpleServer::new(&Args::default(), AntiReplay::default(), Rc::new(RefCell::new(RandomConnectionIdGenerator::default())));
  let mut stream = Http3OrWebTransportStream::default();
  let response = SimpleServer::MESSAGE.to_vec();
  let mut response_data = ResponseData::from(response);
  response_data.send(&mut stream);
  assert!(response_data.done());
}</Rust>
