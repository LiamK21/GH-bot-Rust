<imports> 
use neqo_http3::connection::Http3Connection;
use neqo_http3::connection::Error;
use neqo_qpack::Error as QpackError;
use std::result::Result;
</imports>
<Rust>
#[test]
fn test_qpack_error_handling() {
  let mut conn = Http3Connection::new();
  let result = conn.qpack_encoder.borrow_mut().send_encoder_updates(&mut conn);
  let expected = Result::<(), QpackError>::Ok(());
  let actual = result.map_err(|e| e);
  assert!(actual.is_ok());
}</Rust>
