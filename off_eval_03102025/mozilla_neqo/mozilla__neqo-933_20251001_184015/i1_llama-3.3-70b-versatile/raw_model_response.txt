<imports> 
use crate::cc::CongestionControl;
use std::time::{Duration, Instant};
use crate::tracking::SentPacket;
</imports>
<Rust>
#[test]
fn test_detect_persistent_congestion_first_rtt() {
  let mut cc = CongestionControl::default();
  let lost_packets = vec![SentPacket::new(0, 1, Instant::now(), true, std::rc::Rc::default(), 1000)];
  let pto = Duration::from_millis(100);
  let prev_largest_acked_sent = None;
  let first_rtt_sample_time = Some(Instant::now());

  cc.detect_persistent_congestion(first_rtt_sample_time, prev_largest_acked_sent, pto, &lost_packets);
  assert_eq!(cc.cwnd(), 14720);
}
</Rust>
