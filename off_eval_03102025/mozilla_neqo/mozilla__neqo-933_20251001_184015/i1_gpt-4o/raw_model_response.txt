<imports>
use super::CongestionControl;
use crate::tracking::SentPacket;
use std::rc::Rc;
use std::time::{Duration, Instant};
</imports>
<Rust>
#[test]
fn test_persistent_congestion_defer_start() {
    let mut cc = CongestionControl::default();
    let pto = Duration::from_millis(100);
    let now = Instant::now();
    let first_rtt_sample_time = Some(now + pto);
    let prev_largest_acked_sent = Some(now);
    
    let lost_packets = vec![
        SentPacket::new(
            crate::packet::PacketType::Short,
            1,
            now + pto * 2,
            true,
            Rc::default(),
            1000,
        ),
        SentPacket::new(
            crate::packet::PacketType::Short,
            2,
            now + pto * 4,
            true,
            Rc::default(),
            1000,
        ),
    ];

    cc.on_packets_lost(now, first_rtt_sample_time, prev_largest_acked_sent, pto, &lost_packets);
    assert_eq!(cc.cwnd(), crate::cc::CWND_MIN);
}
</Rust>
