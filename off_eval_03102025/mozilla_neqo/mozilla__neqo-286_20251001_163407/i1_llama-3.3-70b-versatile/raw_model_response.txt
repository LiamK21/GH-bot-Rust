#[test]
fn test_available_space() {
  let data = vec![1, 2, 3, 4, 5];
  let space = 10;
  let offset = 0;
  let stream_id = 5;
  let fin = false;

  let (frame, length) = Frame::new_stream(stream_id, offset, &data, fin, space);
  assert!(length <= space);
}
