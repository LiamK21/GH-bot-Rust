<imports> 
use neqo_transport::Connection;
use neqo_transport::Error;
use std::time::Instant;
use std::time::Duration;
</imports>
<Rust>
#[test]
fn test_idle_timeout_expires() {
  let mut client = Connection::new_client();
  let now = Instant::now();
  client.process_timer(now + Duration::from_secs(60));
  assert!(matches!(client.state(), neqo_transport::State::Closing{..}));
}</Rust>
