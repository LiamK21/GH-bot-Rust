#[test]
fn test_new_stream_type_reader() {
  let mut reader = NewStreamHeadReader::new(StreamType::BiDi, Role::Server);
  let input = [u64::from(H3_FRAME_TYPE_SETTINGS)];
  let result = reader.decode(&input, false);
  assert_eq!(result, Err(Error::HttpFrame));
}
