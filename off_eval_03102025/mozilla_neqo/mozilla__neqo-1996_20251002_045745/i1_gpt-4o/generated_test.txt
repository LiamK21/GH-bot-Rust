#[test]
fn test_http3_stream_type_reader() {
    let mut conn = Connection::new_client("example.com", &["alpn"], &Default::default()).unwrap();
    let mut reader = NewStreamHeadReader::new(StreamType::BiDi, Role::Server);
    let result: Res<(ReceiveOutput, bool)> = reader.receive(&mut conn);
    assert_eq!(result, Err(Error::HttpFrame));
}
