<imports> 
use neqo_transport::cc::CongestionControl;
use neqo_transport::cc::ClassicCongestionControl;
use neqo_transport::packet::PacketType;
use neqo_transport::packet::SentPacket;
use neqo_transport::time::Duration;
use neqo_transport::time::Instant;
use std::time::Duration as StdDuration;
</imports>
<Rust>
#[test]
fn test_app_limited_cwnd_increase() {
  let mut cc = ClassicCongestionControl::new(NewReno::default());
  let mut now = Instant::now();
  let mut next_pn = 0;
  let mut pkts = Vec::new();
  for _ in 0..5 {
    let p = SentPacket::new(
        PacketType::Short,
        next_pn,           // pn
        now,               // time sent
        true,              // ack eliciting
        Vec::new(),        // tokens
        1000,              // size
    );
    next_pn += 1;
    cc.on_packet_sent(&p);
    pkts.push(p);
  }
  assert_eq!(cc.bytes_in_flight(), 5000);
  now += Duration::from_millis(100);
  for (i, pkt) in pkts.into_iter().enumerate() {
    cc.on_packets_acked(&[pkt], Duration::from_millis(100), now);
    assert_eq!(
        cc.bytes_in_flight(),
        (5 - i - 1) * 1000
    );
    if i == 0 {
      assert_eq!(cc.congestion_window, 6000);
    } else {
      assert_eq!(cc.congestion_window, 6000 + (i * 1000));
    }
  }
}
</Rust>
