#[test]
fn test_resumption_token() {
  let mut client = Connection::new();
  let stream_id = StreamId::new(1);
  let mut handler = neqo_client::old::Handler::new();
  handler.read(&mut client, stream_id).unwrap();
  let handler_done = handler.handle(&mut client).unwrap();
  let resumption_token = handler.token.take();
  assert!(handler_done && resumption_token.is_some());
}
