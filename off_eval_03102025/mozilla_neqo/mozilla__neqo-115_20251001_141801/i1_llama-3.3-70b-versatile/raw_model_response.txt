<imports> 
use neqo_transport::Connection;
use neqo_transport::Packet;
use std::log;
</imports>
<Rust>
#[test]
fn test_retry_packet() {
  let mut conn = Connection::new();
  let token = vec![1, 2, 3];
  let hdr = Packet::retry(token.clone());
  let res = conn.handle_retry(hdr, token);
  assert!(res.is_ok());
  assert!(conn.retry_token.is_some());
  assert_eq!(conn.retry_token.unwrap(), token);
}</Rust>
