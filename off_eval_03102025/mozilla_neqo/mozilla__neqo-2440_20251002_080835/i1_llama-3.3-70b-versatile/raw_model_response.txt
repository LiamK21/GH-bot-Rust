<imports> 
use neqo_common::tos::IpTosEcn;
use strum::FromRepr;
</imports>
<Rust>
#[test]
fn test_from_repr_ip_tos_ecn() {
    let repr = 0b11;
    let actual: IpTosEcn = IpTosEcn::from_repr(repr).expect("all ECN values are covered");
    let expected = IpTosEcn::Ce;
    assert_eq!(actual, expected);
}</Rust>
