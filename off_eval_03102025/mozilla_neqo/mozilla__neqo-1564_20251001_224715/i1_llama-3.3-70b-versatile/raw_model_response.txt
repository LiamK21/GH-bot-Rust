<imports>
use super::StreamHandlerType;
use super::Url;
use super::Args;
use super::Http3Client;
use super::StreamId;
use std::path::PathBuf;

<Rust>
#[test]
fn test_stream_close_send() {
  let url = Url::parse("http://example.com").unwrap();
  let args = Args { output_dir: PathBuf::from("/tmp") };
  let mut all_paths = Vec::new();
  let mut client = Http3Client::new();
  let client_stream_id = StreamId::new(1);
  let handler_type = StreamHandlerType::Download;
  let stream_handler = StreamHandlerType::create_stream_handler(handler_type, &url, &args, &mut all_paths, &mut client, client_stream_id);
  let result = client.stream_close_send(client_stream_id);
  assert!(result.is_ok());
}
</Rust>
