#[test]
fn test_idle_timeout_expiry() {
    let mut idle_timeout = IdleTimeout::new(Duration::from_secs(10));
    let now = Instant::now();
    let pto = Duration::from_secs(1);

    let expected_expiry = now + Duration::from_secs(10);
    let actual_expiry = idle_timeout.expiry(now, pto);

    assert_eq!(expected_expiry, actual_expiry);
}
