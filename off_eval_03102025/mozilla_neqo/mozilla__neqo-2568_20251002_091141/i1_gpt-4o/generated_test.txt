#[test]
fn test_is_empty_behavior() {
    let count = Count::new(0, 0, 0, 0);
    assert!(count.is_empty());

    let count = Count::new(1, 0, 0, 0);
    assert!(!count.is_empty());

    let count = Count::new(0, 1, 0, 0);
    assert!(!count.is_empty());

    let count = Count::new(0, 0, 1, 0);
    assert!(!count.is_empty());

    let count = Count::new(0, 0, 0, 1);
    assert!(!count.is_empty());
}
