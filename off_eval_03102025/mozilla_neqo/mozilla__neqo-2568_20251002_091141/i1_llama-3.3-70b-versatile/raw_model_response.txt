<imports> 
use neqo_transport::ecn::Count;
use neqo_transport::ecn::IpTosEcn;
use std::collections::EnumMap;
</imports>
<Rust>
#[test]
fn test_ecn_count_behavior() {
  let not_ect = 0;
  let ect1 = 1;
  let ect0 = 0;
  let ce = 0;
  let count = Count(EnumMap::from_array([not_ect, ect1, ect0, ce]));
  assert!(count.is_some());
  assert!(!count.is_empty());
  let not_ect = 0;
  let ect1 = 0;
  let ect0 = 0;
  let ce = 0;
  let count = Count(EnumMap::from_array([not_ect, ect1, ect0, ce]));
  assert!(!count.is_some());
  assert!(count.is_empty());
}
</Rust>
