use neqo_transport::connection::Connection;

<Rust>
#[test]
fn test_resend_0rtt_after_rejection() {
  let mut connection = Connection::new();
  let now = Instant::now();
  connection.zero_rtt_state = neqo_transport::connection::ZeroRttState::Sending;
  connection.client_0rtt_rejected(now);
  assert!(connection.loss_recovery.drop_0rtt(&connection.paths.primary().unwrap(), now).len() > 0);
}</Rust>
