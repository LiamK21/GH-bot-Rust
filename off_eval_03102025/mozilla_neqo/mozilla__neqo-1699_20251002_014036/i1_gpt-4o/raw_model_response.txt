<imports>
use super::init;
use log::LevelFilter;
use std::sync::Once;
</imports>
<Rust>
#[test]
fn test_init_no_op_when_static_max_level_off() {
    static INIT_ONCE: Once = Once::new();
    INIT_ONCE.call_once(|| {
        // Set STATIC_MAX_LEVEL to Off for testing
        log::set_max_level(LevelFilter::Off);
    });

    // Call init and check if it returns immediately
    init();

    // If STATIC_MAX_LEVEL is Off, init should return immediately without setting up the logger
    // We can't directly test the logger setup, but we can ensure no panic or error occurs
    assert!(true);
}
</Rust>
