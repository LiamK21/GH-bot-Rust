#[test]
fn test_unused_methods() {
    let mut conn = Connection::new();
    let mut send_order = Some(SendOrder::default());
    let mut fairness = true;

    let result_sendorder = set_sendorder(&mut conn, send_order);
    let result_fairness = set_fairness(&mut conn, fairness);

    assert!(result_sendorder.is_err());
    assert!(result_fairness.is_err());
}
