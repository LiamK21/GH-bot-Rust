#[test]
fn test_headers_valid_with_invalid_pseudo_headers() {
    let headers = vec![
        Header::new(":method", "GET"),
        Header::new(":scheme", "https"),
        Header::new(":path", "/"),
        Header::new(":status", "200"),
    ];
    let result = headers_valid(&headers, MessageType::Request);
    assert_eq!(result, Err(Error::InvalidHeader));
}
