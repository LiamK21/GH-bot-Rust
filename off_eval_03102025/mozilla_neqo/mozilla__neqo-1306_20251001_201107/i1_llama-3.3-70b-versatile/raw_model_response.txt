#[test]
fn test_headers_valid_pseudo_headers() {
  let headers = vec![
    Header { name: ":method", value: "GET" },
    Header { name: ":scheme", value: "https" },
    Header { name: ":path", value: "/path" },
    Header { name: "host", value: "example.com" },
  ];
  let result = headers_valid(&headers, MessageType::Request);
  assert!(result.is_ok());
  let modified_headers = vec![
    Header { name: ":method", value: "GET" },
    Header { name: ":method", value: "POST" },
  ];
  let modified_result = headers_valid(&modified_headers, MessageType::Request);
  assert!(modified_result.is_err());
}</Rust>
