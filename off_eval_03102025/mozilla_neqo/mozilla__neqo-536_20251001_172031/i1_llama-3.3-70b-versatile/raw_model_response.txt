<imports> 
use std::time::Instant;
use neqo_transport::connection::Connection;
use neqo_transport::connection::Output;
use neqo_transport::connection::State;
</imports>
<Rust>
#[test]
fn test_output_on_idle_timer() {
  let mut client = Connection::new_client();
  let now = Instant::now();
  client.process_input(b"test", now);
  let output = client.process_output(now);
  assert_ne!(output, Output::Nothing);
  let output = client.process_output(now + std::time::Duration::from_millis(100));
  assert_ne!(output, Output::Nothing);
}
</Rust>
