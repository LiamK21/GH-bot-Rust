<imports>
use super::Connection;
use std::time::{Duration, Instant};
use crate::Output;
use crate::State;
</imports>
<Rust>
#[test]
fn test_process_output_triggers_timer() {
    let mut client = Connection::new_client();
    let now = Instant::now();
    
    // Simulate the client being in a state where a timer would trigger a state change
    client.set_state(State::Confirmed);
    
    // Call process_output and expect it to handle the timer internally
    let output = client.process_output(now + Duration::from_secs(10));
    
    // Verify that the state has changed to Closed due to the timer
    assert!(matches!(client.state(), State::Closed(_)));
    // Verify that the output is as expected
    assert!(matches!(output, Output::Callback(_)));
}
</Rust>
