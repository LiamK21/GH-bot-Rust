<imports> 
use neqo_http3::headers_checks::is_interim;
use neqo_http3::Error;
use neqo_http3::Header;
</imports>
<Rust>
#[test]
fn test_is_interim_returns_error_for_101_status() {
  let headers = vec![Header::new(":status", "101")];
  let result = is_interim(&headers);
  assert!(result.is_err());
  assert_eq!(result.unwrap_err(), Error::InvalidHeader);
}
</Rust>
