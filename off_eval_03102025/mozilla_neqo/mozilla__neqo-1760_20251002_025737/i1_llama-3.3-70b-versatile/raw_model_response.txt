<imports> 
use std::error::Error;
use neqo_crypto::agent::SecretAgent;
use neqo_crypto::ssl;
</imports>
<Rust>
#[test]
fn test_greasing_option() {
  let mut agent = SecretAgent::new();
  let grease = true;
  let result_before_patch = agent.set_option(ssl::Opt::Grease, grease);
  assert!(result_before_patch.is_err());
  let result_after_patch = agent.set_option(ssl::Opt::Grease, grease);
  assert!(result_after_patch.is_ok());
}</Rust>
