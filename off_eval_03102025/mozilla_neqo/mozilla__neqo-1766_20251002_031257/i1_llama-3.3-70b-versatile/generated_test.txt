#[test]
fn test_print_stats_after_close() {
  let mut args = Args { upload_size: 100, stats: true };
  let client = super::Connection {};
  let runner = Runner { args, client, handler: super::Handler {} };
  runner.args.stats = true;
  let stats = Stats {};
  let result_stats = runner.client.stats();
  assert_eq!(result_stats, stats);
}
