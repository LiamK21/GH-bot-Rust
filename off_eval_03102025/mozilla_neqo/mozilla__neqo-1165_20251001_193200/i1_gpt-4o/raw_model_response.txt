<imports>
use super::LossRecoverySpace;
use std::time::{Duration, Instant};
</imports>
<Rust>
#[test]
fn test_detect_lost_packets_no_overflow() {
    let mut recovery_space = LossRecoverySpace::new();
    let now = Instant::now();
    let loss_delay = Duration::from_secs(1);
    let packet_time_sent = now - Duration::from_secs(2); // Simulate a packet sent 2 seconds ago

    // Simulate adding a packet with a specific time_sent
    recovery_space.sent_packets.insert(1, Packet { time_sent: packet_time_sent });

    // Call the function under test
    recovery_space.detect_lost_packets(now, loss_delay, Some(1));

    // Check if the packet is considered lost
    assert!(recovery_space.lost_packets.contains(&1));
}
</Rust>
