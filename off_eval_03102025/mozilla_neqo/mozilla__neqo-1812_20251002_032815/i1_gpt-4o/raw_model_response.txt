<imports>
use super::ClassicCongestionControl;
use super::CongestionControl;
use crate::cc::classic_cc::WindowAdjustment;
use crate::cc::classic_cc::Packet;
</imports>

<Rust>
#[test]
fn test_rebinding_does_not_panic() {
    struct DummyWindowAdjustment;
    impl WindowAdjustment for DummyWindowAdjustment {}

    let mut cc = ClassicCongestionControl::<DummyWindowAdjustment>::new();
    cc.bytes_in_flight = 0;

    let pkt = Packet {
        pn: 1,
        size: 100,
    };

    cc.on_packets_lost(&[pkt]);

    assert_eq!(cc.bytes_in_flight, 0);
}
</Rust>
