<imports> 
use neqo_transport::connection::Connection;
use neqo_transport::packet::PacketHdr;
use neqo_transport::frame::StreamType;
use neqo_common::matches;
</imports>
<Rust>
#[test]
fn test_zero_rtt_after_rejection() {
  let mut client = Connection::new_client("127.0.0.1", &["hq-22"]);
  let stream_id = client.stream_create(StreamType::UniDi).unwrap();
  let msg = &[1, 2, 3];
  client.stream_send(stream_id, msg).unwrap();
  client.zero_rtt_state = Connection::ZeroRttState::Rejected;
  let client_out = client.process(None, 0);
  assert!(client_out.as_dgram_ref().is_some());
  let stream_id_after_reject = client.stream_create(StreamType::UniDi).unwrap();
  assert_eq!(stream_id, stream_id_after_reject);
  client.stream_send(stream_id_after_reject, msg).unwrap();
  let client_after_reject = client.process(None, 0);
  assert!(client_after_reject.as_dgram_ref().is_some());
}</Rust>
