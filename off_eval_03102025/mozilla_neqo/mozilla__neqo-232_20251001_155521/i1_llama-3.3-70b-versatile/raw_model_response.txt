<imports> 
use super::*;
use crate::connection::State;
use crate::stream_id::StreamId;
use std::collections::VecDeque;
</imports>
<Rust>
#[test]
fn test_event_ordering() {
  let mut evts = ConnectionEvents::default();
  evts.new_stream(4.into());
  evts.connection_state_change(State::Connected);
  let mut events: VecDeque<ConnectionEvent> = VecDeque::new();
  while let Some(event) = evts.next_event() {
    events.push_back(event);
  }
  assert_eq!(events[0], ConnectionEvent::StateChange(State::Connected));
  assert_eq!(events[1], ConnectionEvent::NewStream { stream_id: 4, stream_type: StreamType::BiDi });
}
</Rust>
