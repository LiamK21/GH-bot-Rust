#[test]
fn test_cli_options() {
  let _ = env::set_var("CARGO_MANIFEST_DIR", ".");
  let result = App::new("grcov")
    .arg(Arg::with_name("paths")
         .help("Sets the input paths to use")
         .required(true)
         .multiple(true)
         .takes_value(true))
    .try_get_matches_from vec!["grcov", "test"];
  assert!(result.is_ok());
}
