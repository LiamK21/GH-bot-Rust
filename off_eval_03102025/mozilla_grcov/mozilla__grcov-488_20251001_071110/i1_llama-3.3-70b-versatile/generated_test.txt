#[test]
fn test_handle_file() {
  let archive = Archive { name: "test".to_string() };
  let mut file = File::open("./test/jacoco/basic-report.xml").ok().unwrap();
  let path = PathBuf::from("./test/jacoco/basic-report.xml");
  let gcno_stem_archives: &std::cell::RefCell<std::collections::HashMap<String, &Archive>> = &std::cell::RefCell::new(std::collections::HashMap::new());
  let gcda_stem_archives: &std::cell::RefCell<std::collections::HashMap<String, Vec<&Archive>>> = &std::cell::RefCell::new(std::collections::HashMap::new());
  archive.handle_file(Some(&mut file), &path, gcno_stem_archives, gcda_stem_archives);
}
