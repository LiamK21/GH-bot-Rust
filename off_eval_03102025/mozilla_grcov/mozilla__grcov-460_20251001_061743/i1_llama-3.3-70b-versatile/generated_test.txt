#[test]
fn test_branch_coverage_in_html_report() {
  let config = get_config();
  let mut result = CovResult {
    lines: HashMap::new(),
    functions: HashMap::new(),
    branches: HashMap::new(),
  };
  result.lines.insert("file".to_string(), 10);
  result.functions.insert("func".to_string(), true);
  result.branches.insert("file".to_string(), vec![true, false]);
  let stats = get_stats(&result);
  let html_stats = HtmlStats {
    total_lines: stats.total_lines,
    covered_lines: stats.covered_lines,
    total_funs: stats.total_funs,
    covered_funs: stats.covered_funs,
    total_branches: stats.total_branches,
    covered_branches: stats.covered_branches,
  };
  assert!(html_stats.total_branches > 0);
  assert!(html_stats.covered_branches > 0);
}
