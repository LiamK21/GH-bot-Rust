#[test]
fn test_warn_overflow() {
  let mut results: Vec<(String, HashMap<String, String>)> = Vec::new();
  let warn_overflow = true;
  let original_log = std::mem::replace(log::log_enabled, |level| level == Level::Error);
  add_results(&mut results, warn_overflow);
  assert!(log::log_enabled(Level::Warn));
  std::mem::replace(&mut log::log_enabled, original_log);
}
