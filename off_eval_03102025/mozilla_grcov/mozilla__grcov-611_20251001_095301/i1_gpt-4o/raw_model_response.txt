<imports>
use super::add_results;
use log::Level;
use log::Log;
use log::Metadata;
use log::Record;
use std::sync::Mutex;
</imports>

<Rust>
struct TestLogger {
    pub messages: Mutex<Vec<String>>,
}

impl Log for TestLogger {
    fn enabled(&self, metadata: &Metadata) -> bool {
        metadata.level() <= Level::Warn
    }

    fn log(&self, record: &Record) {
        if self.enabled(record.metadata()) {
            self.messages.lock().unwrap().push(format!("{}", record.args()));
        }
    }

    fn flush(&self) {}
}

#[test]
fn test_execution_count_overflow_warning() {
    let test_logger = TestLogger {
        messages: Mutex::new(Vec::new()),
    };
    log::set_boxed_logger(Box::new(test_logger)).unwrap();
    log::set_max_level(Level::Warn.to_level_filter());

    let mut results = Vec::new();
    add_results(&mut results, true);

    let messages = log::logger().downcast_ref::<TestLogger>().unwrap().messages.lock().unwrap();
    assert!(messages.contains(&"Execution count overflow detected.".to_string()));
}
