<imports> 
use super::get_coverage;
use std::collections::BTreeMap;
use std::path::PathBuf;
use grcov::CovResultIter;
use grcov::DemangleOptions;
use grcov::Coverage;
</imports>
<Rust>
#[test]
fn test_cobertura_report_contains_full_class_level_line_coverage() {
  let results: CovResultIter = BTreeMap::new().into_iter();
  let sources: Vec<String> = vec![];
  let demangle: bool = false;
  let demangle_options: DemangleOptions = DemangleOptions::new();
  let coverage: Coverage = get_coverage(results, sources, demangle, demangle_options);
  let expected_lines: Vec<super::Line> = vec![];
  assert!(coverage.classes.iter().all(|class| !class.lines.is_empty()));
}</Rust>
