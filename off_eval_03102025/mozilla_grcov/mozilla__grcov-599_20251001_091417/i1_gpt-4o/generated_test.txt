#[test]
fn test_cobertura_output_type() {
    let output = Command::new("cargo")
        .arg("run")
        .arg("--")
        .arg("--output-type")
        .arg("cobertura")
        .output()
        .expect("Failed to execute command");

    assert!(output.status.success());
    let stdout = String::from_utf8_lossy(&output.stdout);
    assert!(stdout.contains("cobertura"));
}
