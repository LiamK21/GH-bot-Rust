<imports> 
use super::parse_lcov;
use std::fs::File;
use std::io::Read;
</imports>
<Rust>
#[test]
fn test_lcov_parser_with_parsing_error() {
  let mut f = File::open("./test/invalid_DA_record.info").expect("Failed to open lcov file");
  let mut buf = Vec::new();
  f.read_to_end(&mut buf).unwrap();
  let result = parse_lcov(buf, true, false);
  assert!(result.is_err());
  let result_ignoring_error = parse_lcov(buf, true, true);
  assert!(result_ignoring_error.is_ok());
}</Rust>
