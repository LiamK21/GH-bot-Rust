#[test]
fn test_bundled_resources() {
  let config = Config {
    hi_limit: 90.0,
    med_limit: 75.0,
    date: None,
    branch_enabled: false,
    precision: 0,
    html_resources: HtmlResources::Bundled,
  };

  let output = PathBuf::from("test");
  super::gen_bundled_resources(&config, &output);
  let bulma_path = output.join("bulma.min.css");
  assert!(bulma_path.exists());
}
