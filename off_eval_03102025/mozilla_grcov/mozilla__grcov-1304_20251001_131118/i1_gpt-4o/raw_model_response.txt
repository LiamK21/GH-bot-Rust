<imports>
use super::gen_bundled_resources;
use super::Config;
use super::HtmlResources;
use std::path::Path;
use std::fs;
</imports>

<Rust>
#[test]
fn test_gen_bundled_resources_creates_bulma_file() {
    let config = Config {
        hi_limit: 90.0,
        med_limit: 75.0,
        low_limit: 50.0,
        date: None,
        branch_enabled: false,
        precision: 2,
        html_resources: HtmlResources::Bundled,
    };

    let output_path = Path::new("test_output");
    fs::create_dir_all(output_path).unwrap();

    gen_bundled_resources(&config, output_path);

    let bulma_path = output_path.join("bulma.min.css");
    assert!(bulma_path.exists());

    fs::remove_file(bulma_path).unwrap();
    fs::remove_dir(output_path).unwrap();
}
</Rust>
