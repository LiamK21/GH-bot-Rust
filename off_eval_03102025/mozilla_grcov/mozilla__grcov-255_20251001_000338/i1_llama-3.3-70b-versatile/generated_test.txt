#[test]
fn test_output_activedata_etl_with_output_file() {
  let mut output = Vec::new();
  let output_file = Some("test_output.txt");
  let results = [(("file1.rs", "path1"), vec![1, 0, 1]), (("file2.rs", "path2"), vec![1, 1, 0])].into_iter();
  output_activedata_etl(results, output_file);
  let mut file = std::fs::File::open("test_output.txt").unwrap();
  let mut contents = String::new();
  file.read_to_string(&mut contents).unwrap();
  assert!(contents.len() > 0);
  std::fs::remove_file("test_output.txt").unwrap();
}
