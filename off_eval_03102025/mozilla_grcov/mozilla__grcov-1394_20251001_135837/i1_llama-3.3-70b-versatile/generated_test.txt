#[test]
fn test_llvm_profiles_to_lcov_with_threads() {
  let profile_paths = vec![PathBuf::from("test.profraw")];
  let binary_path = Path::new("binary");
  let working_dir = Path::new("working_dir");
  let num_threads = 4;
  let result_with_threads = llvm_profiles_to_lcov(&profile_paths, binary_path, working_dir, num_threads).unwrap();
  let result_without_threads = llvm_profiles_to_lcov(&profile_paths, binary_path, working_dir, 1).unwrap();
  assert_ne!(result_with_threads, result_without_threads);
}
