#[test]
fn test_output_html_subpath_creation() {
  let results: Vec<(&str, &str, &str)> = vec![("file1", "content1", "content2")];
  let output_dir = Path::new("coverage/subpath");
  let num_threads = 1;
  let branch_enabled = true;
  let output_config_file = None;
  fs::remove_dir_all("coverage").ok();
  output_html(&results, Some(&output_dir), num_threads, branch_enabled, output_config_file);
  assert!(output_dir.exists());
}
