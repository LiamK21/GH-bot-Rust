<imports>
use super::output_html;
use std::fs::{self, File};
use std::path::Path;
</imports>

<Rust>
#[test]
fn test_create_subpath_in_output_path() {
    let output_dir = Path::new("coverage/subpath");
    let _ = fs::remove_dir_all(output_dir);

    output_html(&[], Some(output_dir), 1, false, None);

    assert!(output_dir.exists() && output_dir.is_dir());

    let _ = fs::remove_dir_all("coverage");
}
</Rust>
