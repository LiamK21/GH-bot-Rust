#[test]
fn test_parallel_mode() {
  let results = Vec::new();
  let repo_token = "";
  let service_name = "";
  let service_number = "";
  let service_job_number = "";
  let commit_sha = "";
  let with_function_info = false;
  let output_file = None;
  let vcs_branch = "";
  let parallel = true;

  let expected_json = serde_json::json!({
    "service_name": service_name,
    "service_number": service_number,
    "service_job_number": service_job_number,
    "parallel": parallel,
  });

  let actual_json = {
    let mut map = HashMap::new();
    output_coveralls(results, repo_token, service_name, service_number, service_job_number, commit_sha, with_function_info, output_file, vcs_branch, parallel);
    map.insert("service_name", service_name);
    map.insert("service_number", service_number);
    map.insert("service_job_number", service_job_number);
    map.insert("parallel", parallel);
    serde_json::json!(map)
  };

  assert_eq!(expected_json, actual_json);
}
