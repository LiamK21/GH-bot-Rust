#[test]
fn test_parse_jacoco_xml_report_unexpected_version() {
    let xml_data = r#"
        <report name="Test Report">
            <package name="example">
                <class name="ExampleClass">
                    <method name="exampleMethod" type="METHOD" covered="1"/>
                </class>
            </package>
        </report>
    "#;
    let reader = BufReader::new(Cursor::new(xml_data.as_bytes()));
    let result = parse_jacoco_xml_report(reader);
    assert!(result.is_ok());
}
