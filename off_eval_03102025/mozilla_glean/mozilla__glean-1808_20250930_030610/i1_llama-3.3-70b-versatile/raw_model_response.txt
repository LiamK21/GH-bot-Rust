<imports> 
use std::panic;
use android_logger::FilterBuilder;
use log::LevelFilter;
use log::Level;
</imports>
<Rust>
#[test]
fn test_glean_enable_logging_filters_modules() {
  let filter = FilterBuilder::new()
    .filter_module("glean_ffi", LevelFilter::Debug)
    .filter_module("glean_core", LevelFilter::Debug)
    .filter_module("glean", LevelFilter::Debug)
    .build();
  let expected_log_level = Level::Debug;
  let actual_log_level = filter.filter_level(&"glean_ffi".to_string());
  assert_eq!(expected_log_level, actual_log_level);
}</Rust>
