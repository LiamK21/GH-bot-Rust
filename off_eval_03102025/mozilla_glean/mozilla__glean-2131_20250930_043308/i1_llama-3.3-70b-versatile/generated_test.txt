#[test]
fn test_client_info_fields() {
  let client_info = ClientInfoMetrics {
    android_sdk_version: Some("29".to_string()),
    device_manufacturer: Some("Google".to_string()),
    device_model: Some("Pixel 4 XL".to_string()),
    locale: Some("en-US".to_string()),
  };
  let glean = Arc::new(Glean::new("test_app"));
  initialize_core_metrics(glean.as_ref(), &client_info);
  assert!(glean.get_string("android_sdk_version").is_some());
  assert!(glean.get_string("device_manufacturer").is_some());
  assert!(glean.get_string("device_model").is_some());
  assert!(glean.get_string("locale").is_some());
}
