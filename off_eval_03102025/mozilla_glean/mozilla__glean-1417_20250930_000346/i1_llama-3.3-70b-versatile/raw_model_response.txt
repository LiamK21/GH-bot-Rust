fn test_timing_distribution_methods_do_not_require_mut_ref() {

  let metric_clone = Arc::clone(&metric);
  let timer_id = metric_clone.start();
  metric_clone.stop_and_accumulate(timer_id);
  metric_clone.cancel(timer_id);
