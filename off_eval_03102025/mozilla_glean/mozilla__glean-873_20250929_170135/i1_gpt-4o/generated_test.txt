struct TestLogger {
    logs: Mutex<Vec<String>>,
}

impl log::Log for TestLogger {
    fn enabled(&self, metadata: &Metadata) -> bool {
        metadata.level() <= Level::Info
    }

    fn log(&self, record: &Record) {
        if self.enabled(record.metadata()) {
            let mut logs = self.logs.lock().unwrap();
            logs.push(format!("{}", record.args()));
        }
    }

    fn flush(&self) {}
}

#[test]
fn test_glean_version_logging() {
    let test_logger = TestLogger {
        logs: Mutex::new(Vec::new()),
    };
    set_logger(Box::leak(Box::new(test_logger))).unwrap();
    set_max_level(Level::Info);

    let cfg = Configuration {
        application_id: "test_app".into(),
        data_path: "test_path".into(),
        upload_enabled: false,
        max_events: None,
    };

    let _ = Glean::new(cfg);

    let logs = test_logger.logs.lock().unwrap();
    let expected_log = format!("Creating new Glean v{}", crate::GLEAN_VERSION);
    assert!(logs.iter().any(|log| log.contains(&expected_log)));
}
