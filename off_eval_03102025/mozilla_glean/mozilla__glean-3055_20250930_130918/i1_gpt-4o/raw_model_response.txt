use super::PingType;

<Rust>
#[test]
fn test_ping_type_metadata_access() {
    let ping_type = PingType::new("test_ping", true, true, true, true, true, vec!["scheduled_ping".to_string()], vec!["reason_code".to_string()]);
    
    assert_eq!(ping_type.name(), "test_ping");
    assert!(ping_type.include_client_id());
    assert!(ping_type.send_if_empty());
    assert!(ping_type.precise_timestamps());
    assert!(ping_type.include_info_sections());
    assert!(ping_type.naively_enabled());
    assert!(ping_type.follows_collection_enabled());
    assert_eq!(ping_type.schedules_pings(), &["scheduled_ping".to_string()]);
    assert_eq!(ping_type.reason_codes(), &["reason_code".to_string()]);
}
</Rust>
