#[test]
fn test_pingtype_metadata() {
  let ping = InnerPing {
    name: "test_ping".to_string(),
    include_client_id: true,
    send_if_empty: false,
    precise_timestamps: true,
    include_info_sections: true,
    enabled: AtomicBool::new(true),
    follows_collection_enabled: AtomicBool::new(true),
    schedules_pings: vec![String::from("ping1"), String::from("ping2")],
    reason_codes: vec![String::from("reason1"), String::from("reason2")],
  };

  let ping_type = PingType { inner: ping };

  assert_eq!(ping_type.name(), "test_ping");
  assert_eq!(ping_type.include_client_id(), true);
  assert_eq!(ping_type.send_if_empty(), false);
  assert_eq!(ping_type.precise_timestamps(), true);
  assert_eq!(ping_type.include_info_sections(), true);
  assert_eq!(ping_type.naively_enabled(), true);
  assert_eq!(ping_type.follows_collection_enabled(), true);
  assert_eq!(ping_type.schedules_pings(), &vec![String::from("ping1"), String::from("ping2")]);
  assert_eq!(ping_type.reason_codes(), &vec![String::from("reason1"), String::from("reason2")]);
}
