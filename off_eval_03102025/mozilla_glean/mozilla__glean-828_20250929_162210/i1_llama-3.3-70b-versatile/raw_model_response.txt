<imports> 
use glean_core::metrics::labeled::{MAX_LABELS, OTHER_LABEL, MAX_LABEL_LENGTH};
use glean_core::metrics::labeled::LABEL_REGEX;
use regex::Regex;
</imports>
<Rust>
#[test]
fn test_label_regex() {
  let bad_label = "1.not_fine";
  let good_label = "this.is.fine";
  assert!(!LABEL_REGEX.is_initialized());
  let _ = LABEL_REGEX.clone();
  assert!(LABEL_REGEX.is_initialized());
  assert!(LABEL_REGEX.lock().unwrap().is_match(bad_label));
  assert!(LABEL_REGEX.lock().unwrap().is_match(good_label));
  assert_eq!(LABEL_REGEX.lock().unwrap().as_str(), "^[a-z_][a-z0-9_-]{0,29}(\\.[a-z_][a-z0-9_-]{0,29})*$");
}
</Rust>
