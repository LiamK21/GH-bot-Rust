#[test]
fn test_rkv_load_state() {
  let dir = tempfile::tempdir().unwrap();
  let path = dir.path().join("data.safe.bin");
  fs::write(path, "<broken>").expect("write to database file");

  let (rkv, load_state) = rkv_new(path.parent().unwrap()).unwrap();
  assert!(matches!(load_state, super::RkvLoadState::Err(_)));
}
