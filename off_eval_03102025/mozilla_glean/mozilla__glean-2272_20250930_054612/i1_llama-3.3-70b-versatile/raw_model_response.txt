<imports> 
use std::borrow::Cow;
use super::LabeledMetric;
use std::collections::HashMap;
</imports>
<Rust>
#[test]
fn test_labeled_metric_labels_cow() {
  let labels: Option<Vec<Cow<'static, str>>> = Some(vec![Cow::from("label1"), Cow::from("label2")]);
  let meta = CommonMetricData { ..Default::default() };
  let labeled_metric = LabeledMetric::new(meta, labels);
  assert!(labeled_metric.labels.is_some());
  assert_eq!(labeled_metric.labels.unwrap().len(), 2);
  assert_eq!(labeled_metric.labels.unwrap()[0], Cow::from("label1"));
  assert_eq!(labeled_metric.labels.unwrap()[1], Cow::from("label2"));
}</Rust>
