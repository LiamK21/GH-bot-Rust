<imports>
use super::QuantityMetric;
use glean_core::CommonMetricData;
use glean_core::traits::Quantity;
</imports>
<Rust>
#[test]
fn test_set_and_get_value() {
    let meta = CommonMetricData {
        name: "test_quantity".into(),
        category: "test".into(),
        send_in_pings: vec!["test_ping".into()],
        ..Default::default()
    };
    let quantity_metric = QuantityMetric::new(meta);

    quantity_metric.set(42);
    let value = quantity_metric.test_get_value("test_ping");

    assert_eq!(value, Some(42));
}
</Rust>
