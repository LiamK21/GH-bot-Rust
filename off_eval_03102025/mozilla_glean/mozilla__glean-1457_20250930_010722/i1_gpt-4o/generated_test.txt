#[test]
fn test_timing_distribution_truncation_keys() {
    let mut hist = HashSet::new();
    hist.insert(1);
    hist.insert(2);
    hist.insert(3);

    let expected_keys: HashSet<u64> = [1, 2, 3].iter().cloned().collect();
    let keys: HashSet<u64> = hist.iter().cloned().collect();

    assert_eq!(keys, expected_keys);
}
