#[test]
fn test_dynamic_label_with_invalid_label() {
  let glean = &Glean {};
  let meta = &CommonMetricData {};
  let base_identifier = "base";
  let label = "1.not_fine";
  let expected = true;
  let actual = dynamic_label(glean, meta, base_identifier, label);
  assert_eq!(actual, base_identifier);
}
