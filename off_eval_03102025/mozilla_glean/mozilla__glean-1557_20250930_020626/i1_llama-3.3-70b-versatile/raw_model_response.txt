fn test_dynamic_label_rename() {
  let glean = Glean::new();
  let meta = CommonMetricData { base_identifier: "test".to_string(), dynamic_label: Some("label".to_string()), ..Default::default() };
  let base_identifier = meta.base_identifier();
  let label = meta.dynamic_label.as_ref().unwrap();

  let result = validate_dynamic_label(&glean, &meta, &base_identifier, label);
  assert!(result.is_ok());
