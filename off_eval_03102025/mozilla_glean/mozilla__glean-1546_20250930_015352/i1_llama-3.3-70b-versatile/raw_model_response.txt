<imports> 
use super::get_timestamp_ms;
use std::thread;
use std::time::Duration;
</imports>
<Rust>
#[test]
fn test_get_timestamp_ms_increases() {
  let initial_timestamp = get_timestamp_ms();
  thread::sleep(Duration::from_millis(10));
  let later_timestamp = get_timestamp_ms();
  assert!(later_timestamp > initial_timestamp);
}</Rust>
