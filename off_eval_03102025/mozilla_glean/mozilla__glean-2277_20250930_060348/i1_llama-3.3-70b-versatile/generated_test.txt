#[test]
fn test_event_database_remove_dir_all() {
  let path = Path::new("test_dir");
  fs::create_dir(path).unwrap();
  let event_db = super::EventDatabase { path: path.to_path_buf(), file_lock: std::sync::RwLock::new(()) };
  let result = std::panic::catch_unwind(|| event_db.remove_dir_all());
  assert!(result.is_ok());
}
