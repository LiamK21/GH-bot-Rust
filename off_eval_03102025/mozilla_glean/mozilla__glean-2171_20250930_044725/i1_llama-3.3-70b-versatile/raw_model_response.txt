#[test]
fn test_sending_of_foreground_background_pings_finishes_within_time() {
  let start = std::time::Instant::now();
  sending_of_foreground_background_pings();
  let elapsed = start.elapsed().as_secs();
  assert!(elapsed < 60);
}
