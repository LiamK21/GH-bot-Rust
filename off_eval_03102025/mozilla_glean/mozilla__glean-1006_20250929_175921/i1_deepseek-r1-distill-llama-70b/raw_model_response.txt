#[test]
fn test_rate_limiter_is_configured() {
    let data_dir = "test_data_dir".into();
    let result = unsafe { glean_initialize_standalone_uploader(data_dir.into()) };
    assert_eq!(result, 0);

    let upload_manager = PingUploadManager::new("test_data_dir".to_string(), true);
    let rate_limiter = upload_manager.rate_limiter();
    
    let (interval, max_tasks) = rate_limiter.get_config();
    assert_eq!(interval, Duration::from_secs(60));
    assert_eq!(max_tasks, 10);
}
