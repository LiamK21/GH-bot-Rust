#[test]
fn test_accumulate_samples() {
  struct MockTimingDistribution {
    called: bool,
  }
  
  impl TimingDistribution for MockTimingDistribution {
    fn start(&self, _id: TimerId) {}
    fn stop(&self, _id: TimerId) {}
    fn cancel(&self, _id: TimerId) {}
    fn accumulate_samples(&mut self, _samples: Vec<i64>) {
      self.called = true;
    }
    fn accumulate_raw_samples_nanos(&mut self, _samples: Vec<u64>) {
      self.called = true;
    }
  }

  let mut metric: MockTimingDistribution = MockTimingDistribution { called: false };
  let samples = vec![1, 2, 3];
  metric.accumulate_samples(samples);
  assert!(metric.called);
}
