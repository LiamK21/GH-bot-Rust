#[test]
fn test_cognitive_min_max() {
  let mut stats = Stats::default();
  stats.structural = 5;
  stats.compute_minmax();
  assert_eq!(stats.cognitive_min(), 0.0);
  assert_eq!(stats.cognitive_max(), 5.0);
  let mut stats2 = Stats::default();
  stats2.structural = 10;
  stats2.compute_minmax();
  stats.merge(&stats2);
  assert_eq!(stats.cognitive_min(), 0.0);
  assert_eq!(stats.cognitive_max(), 10.0);
}
