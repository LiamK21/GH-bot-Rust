<imports>
use super::Cpp;
use std::collections::HashSet;
</imports>
<Rust>
#[test]
fn test_cpp_enum_values() {
    let mut cpp_values = HashSet::new();
    cpp_values.insert(Cpp::DASHGTSTAR as u32);
    cpp_values.insert(Cpp::LPARENRPAREN as u32);
    cpp_values.insert(Cpp::LBRACKRBRACK as u32);
    cpp_values.insert(Cpp::DQUOTEDQUOTE as u32);
    cpp_values.insert(Cpp::This as u32);
    cpp_values.insert(Cpp::Nullptr as u32);
    cpp_values.insert(Cpp::LiteralSuffix as u32);
    cpp_values.insert(Cpp::OperatorName as u32);

    let expected_values: HashSet<u32> = (157..=163).collect();
    assert_eq!(cpp_values, expected_values);
}
</Rust>
