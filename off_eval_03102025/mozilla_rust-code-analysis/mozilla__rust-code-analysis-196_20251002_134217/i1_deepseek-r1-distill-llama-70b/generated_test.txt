#[test]
fn test_ast_documentation() {
    let payload = AstPayload {
        id: "test".to_string(),
        file_name: "test.rs".to_string(),
        code: "fn test() {}".to_string(),
        comment: false,
        span: true,
    };
    let response = AstResponse {
        id: "test".to_string(),
        root: None,
    };
    assert_eq!(payload.id, response.id);
}
