#[test]
fn test_read_file_removes_trailing_newlines() {
  let mut file_data = vec![b' ', b'a', b' ', b'=', b' ', b'4', b'2', b'\n', b'\n', b'\n'];
  let path = PathBuf::from("foo.txt");
  let expected = vec![b' ', b'a', b' ', b'=', b' ', b'4', b'2', b'\n'];
  let actual = read_file(&path).unwrap();
  file_data.append(&mut vec![b'\n']);
  std::fs::write(&path, &file_data).unwrap();
  assert_eq!(expected, read_file(&path).unwrap());
}
