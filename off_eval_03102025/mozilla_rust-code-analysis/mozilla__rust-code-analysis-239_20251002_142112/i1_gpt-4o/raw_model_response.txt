<imports>
use super::read_file;
use std::path::PathBuf;
</imports>
<Rust>
#[test]
fn test_read_file_removes_trailing_blank_lines() {
    let path = PathBuf::from("test_file_with_trailing_blank_lines.txt");
    let expected = b"int a = 42;\n".to_vec();
    
    let actual = read_file(&path).unwrap();
    
    assert_eq!(actual, expected);
}
</Rust>
