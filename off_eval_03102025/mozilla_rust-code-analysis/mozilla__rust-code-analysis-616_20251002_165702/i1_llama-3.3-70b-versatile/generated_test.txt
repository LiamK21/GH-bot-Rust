#[test]
fn test_macro_annotations_on_class() {
    let input1 = "class AutoEntryMonitor";
    let input2 = "class MOZ_STACK_CLASS AutoEntryMonitor";
    let input3 = "class MOZ_STACK_CLASS JS_PUBLIC_API AutoEntryMonitor";

    let expected1 = "class_specifier";
    let expected2 = "class_specifier";
    let expected3 = "function_definition";

    let actual1 = KEYS.get(input1);
    let actual2 = KEYS.get(input2);
    let actual3 = KEYS.get(input3);

    assert_eq!(actual1.unwrap(), &Cpp::ClassSpecifier);
    assert_eq!(actual2.unwrap(), &Cpp::ClassSpecifier);
    assert_eq!(actual3.unwrap(), &Cpp::FunctionDefinition);
}
