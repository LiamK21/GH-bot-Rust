use super::Stats;
use serde::Serialize;

<Rust>
#[test]
fn test_exit_min_max() {
    let mut stats = Stats::default();
    stats.exit = 3;
    stats.compute_minmax();
    assert_eq!(stats.exit_min(), 3.0);
    assert_eq!(stats.exit_max(), 3.0);

    stats.exit = 1;
    stats.compute_minmax();
    assert_eq!(stats.exit_min(), 1.0);
    assert_eq!(stats.exit_max(), 3.0);

    stats.exit = 5;
    stats.compute_minmax();
    assert_eq!(stats.exit_min(), 1.0);
    assert_eq!(stats.exit_max(), 5.0);
}
</Rust>
